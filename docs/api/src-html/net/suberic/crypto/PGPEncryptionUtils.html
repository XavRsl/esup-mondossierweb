095      /**
096       * Decrypts a Message.
097       */
098      public javax.mail.internet.MimeMessage decryptMessage(Session s, javax.mail.internet.MimeMessage msg, java.security.Key key) throws MessagingException, IOException, java.security.GeneralSecurityException {
099        Object o = msg.getContent();
100        if (o instanceof Multipart) {
101          BodyPart decryptedPart = decryptMultipart((MimeMultipart) o, key);
102          MimeMessage m = new MimeMessage(s);
103          java.util.Enumeration hdrLines = msg.getAllHeaderLines();
104          while (hdrLines.hasMoreElements()) {
105            m.addHeaderLine((String) hdrLines.nextElement());
106          }
107          m.setDataHandler(decryptedPart.getDataHandler());
108          return m;
109        }
110    
111        return null;
112      }

117      public MimeBodyPart encryptPart(MimeBodyPart part, java.security.Key[] keys) throws MessagingException, java.security.GeneralSecurityException, IOException {
118    
119        ByteArrayOutputStream baos = new ByteArrayOutputStream();
120        part.writeTo(baos);
121        ByteArrayInputStream bais = new ByteArrayInputStream(baos.toByteArray());
122    
123        return createEncryptedPart(bais, keys);

127      /**
128       * Makes an encrypted BodyPart from the content in the given
129       * InputStream.
130       */
131      protected MimeBodyPart createEncryptedPart(java.io.InputStream stream, java.security.Key[] keys) throws MessagingException, java.io.IOException, java.security.GeneralSecurityException {
132        ContentType cType = new ContentType("multipart/encrypted");
133        cType.setParameter("protocol", "application/pgp-encrypted");
134        MimeMultipart mm = new MultipartEncrypted(cType);
135    
136        MimeBodyPart idPart = new MimeBodyPart();
137        DataSource ds = new ByteArrayDataSource(new String("Version: 1").getBytes(), "pgpapp", "application/pgp-encrypted");
138        idPart.setDataHandler(new javax.activation.DataHandler(ds));
139    
140        //idPart.setContent("Version: 1", "application/pgp-encrypted");
141    
142        MimeBodyPart encryptedContent = new MimeBodyPart();
143        byte[] encryptedBytes = encrypt(stream, keys);
144        ByteArrayDataSource dataSource = new ByteArrayDataSource(encryptedBytes, "message", "application/octet-stream");
145    
146        javax.activation.DataHandler dh = new javax.activation.DataHandler(dataSource);
147    
148        encryptedContent.setFileName("message");
149        encryptedContent.setDataHandler(dh);
150    
151        mm.addBodyPart(idPart);
152        mm.addBodyPart(encryptedContent);
153    
154        UpdatableMBP returnValue = new UpdatableMBP();
155        returnValue.setContent(mm);
156        returnValue.updateMyHeaders();
157    
158        return returnValue;
159    
160      }

196      /**
197       * Decrypts a Multipart.
198       */
199      public MimeBodyPart decryptMultipart(MimeMultipart mpart, java.security.Key key) throws MessagingException, IOException, java.security.GeneralSecurityException {
200    
201        // should have two internal parts.
202        if (mpart.getCount() != 2) {
203          throw new MessagingException ("error in content:  expected 2 parts, got " + mpart.getCount());
204        }
205    
206        // first part should be application/pgp-encrypted
207        BodyPart firstPart = mpart.getBodyPart(0);
208        String firstPartType = firstPart.getContentType();
209        ContentType firstCt = new ContentType(firstPartType);
210        if (firstPartType == null || ! firstCt.getBaseType().equalsIgnoreCase("application/pgp-encrypted")) {
211          throw new MessagingException ("error in content:  expected first part of type application/pgp-encrypted, got " + firstPartType);
212        }
213    
214        // don't bother checking the version for now.
215    
216        BodyPart secondPart = mpart.getBodyPart(1);
217        String secondPartType = secondPart.getContentType();
218        ContentType secondCt = new ContentType(secondPartType);
219        if (secondPartType == null || ! secondCt.getBaseType().equalsIgnoreCase("application/octet-stream")) {
220          throw new MessagingException ("error in content:  expected second part of type application/octet-stream, got " + secondPartType);
221        }
222    
223        java.io.InputStream is = secondPart.getInputStream();
224    
225        byte[] value = decrypt(is, key);
226    
227        ByteArrayInputStream bais = new ByteArrayInputStream(value);
228    
229        MimeBodyPart returnValue = new MimeBodyPart(bais);
230    
231        return returnValue;
232      }

234      /**
235       * Signs a MimeBodyPart.
236       * TODO: add header protect code
237       */
238      public MimeBodyPart signBodyPart(MimeBodyPart mbp, java.security.Key key)
239        throws MessagingException, IOException, java.security.GeneralSecurityException {
240    
241        MimeBodyPart p = MimeUtils.canonicalize(mbp);
242    
243        ContentType ct = new ContentType("multipart/signed");
244        ct.setParameter("protocol", "application/pgp-signature");
245        ct.setParameter("micalg", "pgp-sha1");
246        MimeMultipart mpart = new MultipartEncrypted(ct);
247    
248        ByteArrayOutputStream baos = new ByteArrayOutputStream();
249    
250        com.sun.mail.util.CRLFOutputStream los = new com.sun.mail.util.CRLFOutputStream(baos);
251        OutputStreamWriter osw = new OutputStreamWriter(los);
252    
253        BufferedWriter bw = new BufferedWriter(osw);
254    
255        java.util.Enumeration hdrLines = ((MimeBodyPart)p).getAllHeaderLines();
256    
257        while (hdrLines.hasMoreElements()) {
258          String nextLine = (String)hdrLines.nextElement();
259          bw.write(nextLine);
260          bw.newLine();
261        }
262    
263        bw.newLine();
264        bw.flush();
265    
266        p.getDataHandler().writeTo(javax.mail.internet.MimeUtility.encode(los, p.getEncoding()));
267    
268        InputStream is = new ByteArrayInputStream(baos.toByteArray());
269    
270        byte[] signature = sign(is, key);
271        MimeBodyPart sigPart = new MimeBodyPart();
272    
273        ByteArrayDataSource dataSource = new ByteArrayDataSource(
274                                                                 signature, "signature", "application/pgp-signature");
275    
276        javax.activation.DataHandler dh = new javax.activation.DataHandler(dataSource);
277    
278        sigPart.setDataHandler(dh);
279    
280        mpart.addBodyPart(p);
281        mpart.addBodyPart(sigPart);
282    
283        MimeBodyPart returnValue = new MimeBodyPart();
284        returnValue.setContent(mpart);
285        return returnValue;
286      }

496      /**
497       * Identifies the encryption type of the given MimePart.  Returns
498       * true if this MimePart has an encryption type (encrypted, signed,
499       * etc.) that's handled by this EncryptionUtils.
500       */
501      public String checkEncryptionType(MimePart mp) throws MessagingException {
502        String contentType = mp.getContentType();
503        if (contentType != null) {
504          ContentType ct = null;
505          try {
506            ct = new ContentType(contentType);
507          } catch (javax.mail.internet.ParseException pe) {
508            return null;
509          }
510    
511          String baseType = ct.getBaseType();
512          if (baseType.equalsIgnoreCase("multipart/encrypted")) {
513            String protocol = ct.getParameter("protocol");
514            if (protocol.equalsIgnoreCase("application/pgp-encrypted"))
515              return EncryptionManager.PGP;
516          } else if (baseType.equalsIgnoreCase("multipart/signed")) {
517            String protocol = ct.getParameter("protocol");
518            if (protocol != null && protocol.equalsIgnoreCase("application/pgp-signature")) {
519              return EncryptionManager.PGP;
520            }
521          } else if (baseType.equalsIgnoreCase("application/pgp-keys")) {
522            return EncryptionManager.PGP;
523          }
524        }
525    
526        /*
527        // Check if the message is in text, namely after -----BEGIN PGP MESSAGE-----
528        try {
529          Object obj = mp.getContent();
530          if (obj instanceof String){
531            String s = (String) obj;
532            if (s.indexOf(BEGIN_PGP_MESSAGE) == 0 ||
533               s.indexOf(BEGIN_PGP_SIGNED_MESSAGE) == 0){
534              return EncryptionManager.PGP;
535            }
536          }
537        } catch (IOException ie){
538        }
539        */
540        return null;
541      }
