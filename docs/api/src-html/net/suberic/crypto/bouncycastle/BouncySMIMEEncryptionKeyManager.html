051      /**
052       * Loads this KeyStore from the given input stream.
053       *
054       * <p>If a password is given, it is used to check the integrity of the
055       * keystore data. Otherwise, the integrity of the keystore is not checked.
056       *
057       * <p>In order to create an empty keystore, or if the keystore cannot
058       * be initialized from a stream (e.g., because it is stored on a hardware
059       * token device), you pass <code>null</code>
060       * as the <code>stream</code> argument.
061       *
062       * <p> Note that if this KeyStore has already been loaded, it is
063       * reinitialized and loaded again from the given input stream.
064       *
065       * @param stream the input stream from which the keystore is loaded, or
066       * null if an empty keystore is to be created.
067       * @param password the (optional) password used to check the integrity of
068       * the keystore.
069       *
070       * @exception IOException if there is an I/O or format problem with the
071       * keystore data
072       * @exception NoSuchAlgorithmException if the algorithm used to check
073       * the integrity of the keystore cannot be found
074       */
075      public void loadPublicKeystore(InputStream stream, char[] password)
076        throws IOException, java.security.NoSuchAlgorithmException, NoSuchProviderException, KeyStoreException, java.security.cert.CertificateException {
077    
078        if (publicKeyStore == null)
079          initKeyStores();
080    
081        publicKeyStore.load(stream, password);
082    
083      }

179      /**
180       * Returns the key associated with the given alias, using the given
181       * password to recover it.
182       *
183       * @param alias the alias name
184       * @param password the password for recovering the key
185       *
186       * @return the requested key, or null if the given alias does not exist
187       * or does not identify a <i>key entry</i>.
188       *
189       * @exception KeyStoreException if the keystore has not been initialized
190       * (loaded).
191       * @exception NoSuchAlgorithmException if the algorithm for recovering the
192       * key cannot be found
193       * @exception UnrecoverableKeyException if the key cannot be recovered
194       * (e.g., the given password is wrong).
195       */
196      public java.security.Key getPublicKey(String alias)
197        throws KeyStoreException, NoSuchAlgorithmException, UnrecoverableKeyException {
198        if (publicKeyStore != null) {
199          if (publicKeyStore.isCertificateEntry(alias) || publicKeyStore.isKeyEntry(alias)) {
200                    BouncySMIMEEncryptionKey ek = new BouncySMIMEEncryptionKey();
201                    Certificate[] chain;
202                    
203                    if(publicKeyStore.isKeyEntry(alias)){
204                            chain = publicKeyStore.getCertificateChain(alias);
205                    }else{
206                            chain = new Certificate[1];
207                            chain[0] = publicKeyStore.getCertificate(alias);
208                    }
209                    
210                    ek.setCertificateChain(chain);
211                    ek.setDisplayAlias(alias);
212                    return ek;
213          }
214          
215          return null;
216        } else
217          throw new KeyStoreException("KeyStore not initialized.");
218    
219      }

221      /**
222       * Returns the key associated with the given alias, using the given
223       * password to recover it.
224       *
225       * @param alias the alias name
226       * @param password the password for recovering the key
227       *
228       * @return the requested key, or null if the given alias does not exist
229       * or does not identify a <i>key entry</i>.
230       *
231       * @exception KeyStoreException if the keystore has not been initialized
232       * (loaded).
233       * @exception NoSuchAlgorithmException if the algorithm for recovering the
234       * key cannot be found
235       * @exception UnrecoverableKeyException if the key cannot be recovered
236       * (e.g., the given password is wrong).
237       */
238      public java.security.Key getPrivateKey(String alias, char[] password) throws KeyStoreException, NoSuchAlgorithmException, UnrecoverableKeyException {
239        if (privateKeyStore != null) {
240          if (privateKeyStore.isKeyEntry(alias)) {
241            BouncySMIMEEncryptionKey ek = new BouncySMIMEEncryptionKey();
242            Certificate[] chain = privateKeyStore.getCertificateChain(alias);
243            ek.setCertificateChain(chain);
244            PrivateKey privKey = (PrivateKey)privateKeyStore.getKey(alias, password);
245            ek.setDisplayAlias(alias);
246            ek.setKey(privKey);
247            return ek;
248          }
249          
250          return null;
251        } else
252          throw new KeyStoreException("KeyStore not initialized.");
253    
254      }

338      /**
339       * Lists all the alias names of this keystore.
340       *
341       * @param forSignature indicates whether the private keys are for signature
342       * @return set of the alias names
343       *
344       * @exception KeyStoreException if the keystore has not been initialized
345       * (loaded).
346       */
347      public Set publicKeyAliases(boolean forSignature)
348        throws KeyStoreException {
349              //Liao: consider forSignature
350        if (publicKeyStore != null) {
351          Enumeration aliasEnum = publicKeyStore.aliases();
352          HashSet returnValue = new HashSet();
353          while (aliasEnum.hasMoreElements()) {
354            String alias = (String) aliasEnum.nextElement();
355            //if (pkcsKeyStore.isCertificateEntry(alias))
356            returnValue.add(alias);
357          }
358          return returnValue;
359        } else {
360          throw new KeyStoreException("KeyStore not initialized.");
361        }
362      }

364      /**
365       * Lists all the alias names of this keystore.
366       * @param forSignature indicates whether the private keys are for signature
367       * @return set of the alias names
368       *
369       * @exception KeyStoreException if the keystore has not been initialized
370       * (loaded).
371       */
372      public Set privateKeyAliases(boolean forSignature)
373        throws KeyStoreException {
374              //Liao: Consider forSignature
375                if (privateKeyStore != null) {
376                    Enumeration aliasEnum = privateKeyStore.aliases();
377                    HashSet returnValue = new HashSet();
378                    while (aliasEnum.hasMoreElements()) {
379                                    String alias = (String) aliasEnum.nextElement();
380                                    if (privateKeyStore.isKeyEntry(alias))
381                                      returnValue.add(alias);
382                    }
383                    
384                    return returnValue;
385                  } else {
386                    throw new KeyStoreException("KeyStore not initialized.");
387                  }
