001    package net.suberic.crypto;
002    
003    import javax.mail.internet.ContentType;
004    import javax.mail.internet.MimeMultipart;
005    
006    /**
007     * A subclass of MimeMultipart which allows for a custom content type.
008     */
009    public class MultipartEncrypted extends MimeMultipart {
010    
011      ContentType cType = null;
012    
013      /**
014       * Makes a Multipart that uses the given content type, but otherwise
015       * is exactly like a multipart/mixed.
016       */
017      public MultipartEncrypted(ContentType ct) {
018        super("mixed");
019        cType = ct;
020      }
021      
022      /**
023       * Returns the content type given in the constructor.
024       */
025      public String getContentType() {
026        try {
027          ContentType newCtype = new ContentType(super.getContentType());
028          ContentType retCtype = new ContentType(cType.toString());
029          retCtype.setParameter("boundary", newCtype.getParameter("boundary"));
030          return retCtype.toString();
031        } catch (Exception e) {
032          e.printStackTrace();
033        }
034    
035        return "null";
036      }
037    
038    }
