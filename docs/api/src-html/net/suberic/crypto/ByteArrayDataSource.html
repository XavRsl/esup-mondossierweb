001    package net.suberic.crypto;
002    
003    import javax.activation.DataSource;
004    
005    /**
006     * A DataSource based off of a byte array.
007     */
008    public class ByteArrayDataSource implements DataSource {
009    
010      byte[] content;
011    
012      String name;
013    
014      String contentType;
015      
016      /**
017       * Creates a ByteArrayDataSource from the given byte array.
018       */
019      public ByteArrayDataSource(byte[] newContent, String newName, String newContentType) {
020        content = newContent;
021        name = newName;
022        contentType = newContentType;
023      }
024    
025      /**
026       * This method returns the MIME type of the data in the form of a string. 
027       * For this implementation, this is the type that is set in the constructor.
028       *
029       * @return the MIME type 
030       */
031      public java.lang.String getContentType() {
032        return contentType;
033      }
034    
035      /**
036       * Returns a new ByteArrayInputStream based off of the byte array given
037       * as content.
038       *
039       * @return a ByteArrayInputStream
040       */
041      public java.io.InputStream getInputStream() {
042        return new java.io.ByteArrayInputStream(content);
043      }
044    
045      /**
046       * Return the name of this object where the name of the object is dependant 
047       * on the nature of the underlying objects.
048       * 
049       * @return the name given in the constructor
050       */
051      public java.lang.String getName() {
052        return name;
053      }
054    
055      /**
056       * This method returns an OutputStream where the data can be written and 
057       * throws the appropriate exception if it can not do so.
058       * 
059       * Not appropriate for this implementation; throws an IOException.
060       * 
061       * @return an OutputStream
062       */
063      public java.io.OutputStream getOutputStream() throws java.io.IOException {
064        throw new java.io.IOException("Output stream not supported.");
065      }
066    }
