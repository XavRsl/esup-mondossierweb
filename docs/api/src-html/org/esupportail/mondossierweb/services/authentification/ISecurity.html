006     * interface de Security. regroupe les méthodes indispensables à la

010     */

014             * donne le type de l’utilisateur dans l’application a partir de son

017             * @return le type de l’utilisateur
018             */

021            /**
022             * @param nom_appli

024             * @param user_id 
025             * @param code_ctrl 
026             * @param type_ctrl 
027             * @return habilitation
028             */
029            Boolean habilitationApogee(final String nom_appli, final String fonction, final String user_id, final String code_ctrl, final String type_ctrl);
030    
031            /**
032             * donne La propriete ldap du contact désignant son codetu.
033             * @return La propriete ldap du contact désignant son codetu
034             */
035            String getAttributLdapCodEtu();

037            /**
038             * donne le type ldap de l’utilisateur a partir de son identifiant.
039             * @param login
040             * @return le type ldap de l’utilisateur
041             */
042            String typeLdap(final String login);
043    }
