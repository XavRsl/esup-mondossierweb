405            /**
406             * 
407             * @param vueEtudiant
408             * @return le rang dans la liste des Notes et Résultat (aux diplomes et étapes) en cache pour la vueEtudiant
409             */
410            private String getRangNotesEtResultatsEnCache(boolean vueEtudiant){
411                    int rang=0;
412                    boolean enCache=false;
413    
414                    //on parcourt le résultatVdiVet pour voir si on a ce qu'on cherche:
415                    for(CacheResultatsVdiVet crvv : cacheResultats.getResultVdiVet()){
416                            if(!enCache){
417                                    //si on a déjà les infos:
418                                    if(crvv.isVueEtudiant() == vueEtudiant){
419                                            enCache=true;
420                                    }else{
421                                            //on a pas trouvé, on incrémente le rang pour se placer sur le rang suivant
422                                            rang++;
423                                    }
424                            }
425                    }
426    
427                    //si on a pas les infos en cache:
428                    if(!enCache){
429                            return null;
430                    }
431    
432                    return ""+rang;
433    
434            }

436            /**
437             * récupère les résultat aux diplomes et etapes dans le cache (en s'indexant sur le rang)
438             * @param rang
439             */
440            private void recupererCacheResultatVdiVet(int rang){
441                    //1-on vide les listes existantes
442                    if(diplomes!=null){
443                            diplomes.clear();
444                    }
445                    if(etapes!=null){
446                            etapes.clear();
447                    }
448                    //2-on récupère les infos du cache.
449                    diplomes = new LinkedList<Diplome>(cacheResultats.getResultVdiVet().get(rang).getDiplomes());
450                    etapes = new LinkedList<Etape>(cacheResultats.getResultVdiVet().get(rang).getEtapes());
451                    
452                    
453            }

467            public void renseigneNotesEtResultats() {
468                    //On regarde si on a pas déjà les infos dans le cache:
469                    String rang = getRangNotesEtResultatsEnCache(true);
470    
471                    if(rang == null){
472                            etudiantManager.setNotesEtResultats(this);
473                            //AJOUT DES INFOS recupérées dans le cache. true car on est en vue Etudiant
474                            ajouterCacheResultatVdiVet(true);
475                    }else{
476                            //on récupére les infos du cache grace au rang :
477                            recupererCacheResultatVdiVet(new Integer(rang));
478                    }
479            }

497            /**
498             * 
499             * @param etape
500             * @param vueEtudiant
501             * @return  le rang dans la liste des Notes et Résultat (aux elp et epr) en cache pour la vueEtudiant
502             */
503            private String getRangNotesEtResultatsEnCache(Etape etape, boolean vueEtudiant){
504                    int rang=0;
505                    boolean enCache=false;
506    
507                    //on parcourt le résultatElpEpr pour voir si on a ce qu'on cherche:
508                    for(CacheResultatsElpEpr cree : cacheResultats.getResultElpEpr()){
509                            if(!enCache){
510                                    //si on a déjà les infos:
511                                    if(cree.getEtape().getAnnee().equals(etape.getAnnee())
512                                                    && cree.getEtape().getCode().equals(etape.getCode())
513                                                    && cree.getEtape().getVersion().equals(etape.getVersion())
514                                                    && cree.isVueEtudiant() == vueEtudiant){
515                                            enCache=true;
516                                    }else{
517                                            //on a pas trouvé, on incrémente le rang pour se placer sur le rang suivant
518                                            rang++;
519                                    }
520                            }
521                    }
522    
523                    //si on a pas les infos en cache:
524                    if(!enCache){
525                            return null;
526                    }
527                    
528                    return ""+rang;
529    
530            }

565            public void renseigneNotesElpEpr(int index) {
566                    
567                    //On regarde si on a pas déjà les infos dans le cache:
568                    String rang = getRangNotesEtResultatsEnCache(etapes.get(index),true);
569    
570                    if(rang == null){
571                            //On regarde si les notes globales sont en cache pour récupérer l'etape qui va avec la vue.
572                            String rangNoteGlobal = getRangNotesEtResultatsEnCache(true);
573                            if(rangNoteGlobal == null){
574                                    //on recupere les info des notes Globales et on les met dans le cache
575                                    etudiantManager.setNotesEtResultats(this);
576                                    ajouterCacheResultatVdiVet(true);
577                            }else{
578                                    recupererCacheResultatVdiVet(new Integer(rangNoteGlobal));
579                            }
580                            
581                            etudiantManager.setNotesElpEpr(this, etapes.get(index));
582                            //AJOUT DES INFOS recupérées dans le cache. true car on est en vue Etudiant
583                            ajouterCacheResultatElpEpr(etapes.get(index), true);
584                    }else{
585                            //on récupére les infos du cache grace au rang :
586                            recupererCacheResultatElpEpr(new Integer(rang));
587                    }
588            
589            }
