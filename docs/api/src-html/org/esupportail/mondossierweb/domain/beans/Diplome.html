004    package org.esupportail.mondossierweb.domain.beans;
005    
006    import java.io.Serializable;
007    import java.util.ArrayList;
008    import java.util.List;
009    
010    /**
011     * Diplôme.
012     * @author Charlie Dubois
013     */
014    public class Diplome implements Serializable{
015    
016            private static final long serialVersionUID = -725836072091894826L;
017            /**
018             * code de la composante a laquelle appartient le diplôme.
019             */
020            private String cod_cmp;
021            /**
022             * libelle de la composante a laquelle appartient le diplôme.
023             */
024            private String lib_cmp;
025            /**
026             * code du diplôme.
027             */
028            private String cod_dip;
029            /**
030             * code type diplôme établissement.
031             */
032            private String cod_tpd_etb;
033            /**
034             * version du diplôme.
035             */
036            private String cod_vrs_vdi;
037            /**
038             * libelle du diplôme.
039             */
040            private String lib_web_vdi;
041            /**
042             * année universitaire d'obtention.
043             */
044            private String annee;
045            /**
046             * le rang de l'étudiant pour son résultat au diplome.
047             */
048            private String rang;
049            /**
050             * les résultats.
051             */
052            private List<Resultat> resultats;
053            
054            /**
055             * constructeur vide.
056             *
057             */
058            public Diplome() {
059                    super();
060                    resultats = new ArrayList<Resultat>();
061            }
062            public String getCod_cmp() {
063                    return cod_cmp;
064            }
065            public void setCod_cmp(String cod_cmp) {
066                    this.cod_cmp = cod_cmp;
067            }
068            public String getCod_dip() {
069                    return cod_dip;
070            }
071            public void setCod_dip(String cod_dip) {
072                    this.cod_dip = cod_dip;
073            }
074            public String getCod_tpd_etb() {
075                    return cod_tpd_etb;
076            }
077            public void setCod_tpd_etb(String cod_tpd_etb) {
078                    this.cod_tpd_etb = cod_tpd_etb;
079            }
080            public String getCod_vrs_vdi() {
081                    return cod_vrs_vdi;
082            }
083            public void setCod_vrs_vdi(String cod_vrs_vdi) {
084                    this.cod_vrs_vdi = cod_vrs_vdi;
085            }
086            public String getLib_cmp() {
087                    return lib_cmp;
088            }
089            public void setLib_cmp(String lib_cmp) {
090                    this.lib_cmp = lib_cmp;
091            }
092            public String getLib_web_vdi() {
093                    return lib_web_vdi;
094            }
095            public void setLib_web_vdi(String lib_web_vdi) {
096                    this.lib_web_vdi = lib_web_vdi;
097            }
098            public String getAnnee() {
099                    return annee;
100            }
101            public void setAnnee(final String annee) {
102                    this.annee = annee;
103            }
104            public List<Resultat> getResultats() {
105                    return resultats;
106            }
107            public void setResultats(List<Resultat> resultats) {
108                    this.resultats = resultats;
109            }
110            public String getRang() {
111                    return rang;
112            }
113            public void setRang(String rang) {
114                    this.rang = rang;
115            }
116            
117            
118    }
