120             * va chercher et renseigne les informations concernant le calendrier
121             * des examens de 
122             * l'étudiant placé en paramètre avec le WS apo-cri-web.
123             */
124            @Override
125            public void setCalendrier(Etudiant e) {
126    
127                    try {
128                            e.getCalendrier().clear();
129                            List<ExamensEtape> listeCalendExamEtp = new ArrayList<ExamensEtape>();
130    
131                            if (e.isInscriptionsRenseignees()) {
132                                    List<Inscription> vinscr = e.getLinsciae();
133                                    int max_annee = 0;
134                                    // Recherche dernière année
135                                    for (int i = 0; i < vinscr.size(); i++) {
136                                            int cur_annee = Integer.parseInt(vinscr.get(i).getCod_anu().substring(0, 4));
137                                            if (cur_annee > max_annee) {
138                                                    max_annee = cur_annee;
139                                            }
140                                    }               
141                                    
142                                    // Pour chaque étape de la dernière année
143                                    for (int i = 0; i < vinscr.size(); i++) {
144                                            int cur_annee = Integer.parseInt(vinscr.get(i).getCod_anu().substring(0, 4));
145                                            
146                                            if (cur_annee == max_annee) {
147                                                    ExamensEtape examEtp = new ExamensEtape();
148                                                    examEtp.setEtape(vinscr.get(i).getCod_etp());
149                                                    examEtp.setVet(vinscr.get(i).getCod_vrs_vet());
150                                                    examEtp.setLibEtape(vinscr.get(i).getLib_etp());
151                                    
152                                                    // Récuperation des examens
153                                                    ParamRequeteDTO param = new ParamRequeteDTO();
154                                                    param.setChaine1(e.getCod_ind());
155                                                    param.setChaine2(vinscr.get(i).getCod_etp());
156                                                    param.setChaine3(vinscr.get(i).getCod_vrs_vet());
157                                                    List<Examen> listeCalendExam = new ArrayList<Examen>();
158                                                    listeCalendExam = service.getExamensEtape(param);
159                                                            
160                                                    // Affectation des examens
161                                                    if ((listeCalendExam!=null)&&(listeCalendExam.size()>0)) {
162                                                            examEtp.setCalendrierExamensEtape(listeCalendExam);
163                                                            e.setExisteExamEtape(true);
164                                                            
165                                                            //renseigner la page info
166                                                            examEtp.setInfos("");
167                                                            String codCin = listeCalendExam.get(0).getCodCin();
168                                                            if (codCin!=null) {
169                                                                    String pageInfos = service.getPageInfos(codCin);
170                                                                    if (pageInfos!=null) examEtp.setInfos(pageInfos);
171                                                            }
172    
173                                                            listeCalendExamEtp.add(examEtp);
174                                                    }
175                                            }
176                                    }
177                                    e.setCalendrierEtape(listeCalendExamEtp);
178                            }
179                    } catch(Exception ex) {
180                            LOG.error(ex);
181                    }
182            }

251                    String vlibEtp = null;
252                    for (int i = 0; i < vinscr.size(); i++) {

256                            }
257                    }
258                    return vlibEtp;
