008    /**
009     * méthodes indispensables devant être fournies par le bean etudiantManager.
010     * @author Charlie Dubois
011     *
012     */
013    public interface IEtudiant {
014    
015            /**
016             * renseigne le cod_ind de l'étudiant.
017             * @param e
018             */
019            void setCodInd(Etudiant e);
020            /**
021             * renseigne les attributs concernant l'état-civil.
022             * @param e
023             */
024            void setEtatCivil(Etudiant e);
025            /**
026             * renseigne l'url de la photo.
027             * @param e
028             */
029            void setPhoto(Etudiant e);
030            /**
031             * renseigne les attributs concernant les adresses.
032             * @param e
033             */
034            void setAdresses(Etudiant e);
035            /**
036             * @param e
037             * @return vrai si la transcation s'est bien passée
038             */
039            boolean setMajAdresses(Etudiant e);
040            /**
041             * renseigne les attributs concernant les inscriptions.
042             * @param e
043             */
044            void setInscriptions(Etudiant e);
045            /**
046             * renseigne les attributs concernant l'inscription pédagogique d'une étape choisie
047             * @param e
048             * @param et
049             */
050            void setInscriptionPedagogique(Etudiant e, Etape et);
051            /**
052             * renseigne les attributs concernant le calendrier des examens.
053             * @param e
054             */
055            void setCalendrier(Etudiant e);
056            /**
057             * renseigne les attributs concernant les notes et résultats obtenus.
058             * @param e
059             */
060            void setNotesEtResultats(Etudiant e);
061            /**
062             * renseigne les attributs concernant les notes et résultats obtenus 
063             * avec la vue enseignant.
064             * @param e
065             */
066            void setNotesEtResultatsEnseignant(Etudiant e);
067            /**
068             * renseigne les attributs concernant les résultats obtenus 
069             * aux éléments pédagogiques d'une étape choisie.
070             * @param e
071             * @param et
072             */
073            void setNotesElpEpr(Etudiant e, Etape et);
074            /**
075             * renseigne les attributs concernant les résultats obtenus 
076             * aux éléments pédagogiques d'une étape choisie avec la vue enseignant.
077             * @param e
078             * @param et
079             */
080            void setNotesElpEprEnseignant(Etudiant e, Etape et);
081            
082            /**
083             * renseigne les attributs concernant le calendrier de rentrée 
084             * @param etudiant
085             */
086            void setCalendrierRentree(Etudiant etudiant);
087            
088    
089            
090    
091    }
