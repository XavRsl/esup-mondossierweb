004    package org.esupportail.mondossierweb.domain.beans;
005    
006    import java.util.List;
007    
008    
009    /**
010     * représente le cache pour stocker les résultats aux Elp et Epreuves déjà récupérés pour l'étudiant dont on consulte le dossier.
011     * @author Charlie Dubois
012     */
013    public class CacheResultatsElpEpr {
014    
015            /**
016             * Faux si c'est les résultats visibles pour l'enseignant
017             * Vrai si c'est à destination de l'étudiant.
018             */
019            private boolean vueEtudiant;
020            /**
021             * l'étape concernée par ces résultats.
022             */
023            private Etape etape;
024            /**
025             * la liste des éléments pédagogique (avec résultats) d'une étape choisie.
026             */
027            private List<ElementPedagogique> elementsPedagogiques;
028            
029            public CacheResultatsElpEpr(){
030                    super();
031            }
032    
033            
034    
035            public boolean isVueEtudiant() {
036                    return vueEtudiant;
037            }
038    
039    
040    
041            public void setVueEtudiant(boolean vueEtudiant) {
042                    this.vueEtudiant = vueEtudiant;
043            }
044    
045    
046    
047            public Etape getEtape() {
048                    return etape;
049            }
050    
051            public void setEtape(Etape etape) {
052                    this.etape = etape;
053            }
054    
055            public List<ElementPedagogique> getElementsPedagogiques() {
056                    return elementsPedagogiques;
057            }
058    
059            public void setElementsPedagogiques(
060                            List<ElementPedagogique> elementsPedagogiques) {
061                    this.elementsPedagogiques = elementsPedagogiques;
062            }
063            
064            
065    }
