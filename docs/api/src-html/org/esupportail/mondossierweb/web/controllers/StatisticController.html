001    package org.esupportail.mondossierweb.web.controllers;
002    
003    import org.springframework.jmx.export.annotation.ManagedAttribute;
004    import org.springframework.jmx.export.annotation.ManagedResource;
005    
006    /**
007     * bean contenant les statistiques.
008     * @author Charlie Dubois
009     */
010    @ManagedResource(objectName = "spring:name=statisticEsupMonDossierWeb", description = "les statistiques du dossier etudiant")
011    public class StatisticController {
012    
013            /**
014             * le nombre de connexion depuis le lancement du Serveur d'application.
015             */
016            private int nbConnexion;
017            
018            /**
019             * le nombre de connexion d'étudiants depuis le lancement du Serveur d'application.
020             */
021            private int nbConnexionEtudiant;
022            
023            /**
024             * le nombre de connexion d'enseignants depuis le lancement du Serveur d'application.
025             */
026            private int nbConnexionEnseignant;
027    
028            /**
029             * le constructeur vide.
030             */
031            public StatisticController() {
032                    super();
033                    nbConnexion = 0;
034                    nbConnexionEtudiant = 0;
035                    nbConnexionEnseignant = 0;
036            }
037            
038            @ManagedAttribute
039            public int getNbConnexion() {
040                    return nbConnexion;
041            }
042    
043            public void setNbConnexion(int nbConnexion) {
044                    this.nbConnexion = nbConnexion;
045            }
046            
047            @ManagedAttribute
048            public int getNbConnexionEtudiant() {
049                    return nbConnexionEtudiant;
050            }
051    
052            public void setNbConnexionEtudiant(int nbConnexionEtudiant) {
053                    this.nbConnexionEtudiant = nbConnexionEtudiant;
054            }
055    
056            @ManagedAttribute
057            public int getNbConnexionEnseignant() {
058                    return nbConnexionEnseignant;
059            }
060    
061            public void setNbConnexionEnseignant(int nbConnexionEnseignant) {
062                    this.nbConnexionEnseignant = nbConnexionEnseignant;
063            }
064    
065            /**
066             * incremente le nombre de connexion étudiant depuis le lancement du Serveur d'application.
067             */
068            public void incNbConnexionEtudiant(){
069                    nbConnexionEtudiant++;
070                    nbConnexion++;
071            }
072            
073            /**
074             * incremente le nombre de connexion enseignant depuis le lancement du Serveur d'application.
075             */
076            public void incNbConnexionEnseignant(){
077                    nbConnexionEnseignant++;
078                    nbConnexion++;
079            }
080    }
