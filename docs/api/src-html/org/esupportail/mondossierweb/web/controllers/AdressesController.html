187             * pour que l'étudiant puisse modifier ses adresses.
188             * @return la vue avec le formulaire de modification des adresses
189             */
190            public String modifierAdresses() {
191    
192                    adan1 = etudiant.getAdresseannuelle1();
193                    adan2 = etudiant.getAdresseannuelle2();
194                    adan3 = etudiant.getAdresseannuelle3();
195                    adantel = etudiant.getNumerotelannuel();
196                    adfixe1 = etudiant.getAdressefixe1();
197                    adfixe2 = etudiant.getAdressefixe2();
198                    adfixe3 = etudiant.getAdressefixe3();
199                    adfixetel = etudiant.getNumerotelfixe();
200                    emailPerso = etudiant.getEmailPerso();
201                    telPortable = etudiant.getTelPortable();
202    
203                    adanEtranger = etudiant.getAdresseannuelleetranger();
204                    adfixeEtranger = etudiant.getAdressefixeetranger();
205    
206                    if (adanEtranger.equals("")) {
207                            boolAnEtranger = false;
208                    } else {
209                            boolAnEtranger = true;
210                    }
211    
212                    if (adfixeEtranger.equals("")) {
213                            boolFixeEtranger = false;
214                    } else {
215                            boolFixeEtranger = true;
216                    }
217    
218                    typehebergement = etudiant.getAdresseannuelleType();
219                    adfixepays = etudiant.getPaysfixe();
220                    adanpays = etudiant.getPaysannuel();
221                    adancp = etudiant.getAdresseannuellecp();
222                    adanville = etudiant.getVilleannuelle();
223                    adfixecp = etudiant.getAdressefixecp();
224                    adfixeville = etudiant.getVillefixe();
225    
226                    if (typehebergement.equals(DOMCILPARENTAL)) {
227                            adresseIdentique = true;
228                    } else {
229                            adresseIdentique = false;
230                    }
231    
232                    //on remplit la liste des villes(adresse annuelle) si l'adresse est en france
233                    if (adanpays.equals("FRANCE")) {
234                            boolAnEtranger = false;
235                            if(adancp!= null && !adancp.equals("") && Pattern.matches("[0-9]*",adancp))
236                                    cdtoan = service.getCommunes(adancp);
237    
238                    } else {
239                            boolAnEtranger = true;
240                    }
241    
242                    TypeHebergementDTO[] thvo = service.getTypesHebergement();
243                    PaysDTO[] pdto = service.getPays();
244    
245                    ltypehebergement = new ArrayList<SelectItem>();
246                    for (int i = 0; i < thvo.length; i++) {
247                            ltypehebergement.add(new SelectItem(thvo[i].getLibTypeHebergement(), thvo[i].getLibTypeHebergement()));
248                    }
249    
250                    lpays = new ArrayList<SelectItem>();
251                    for (int i = 0; i < pdto.length; i++) {
252                            boolean insere = false;
253                            int j = 0;
254                            while (!insere && j < lpays.size()) {
255                                    if (lpays.get(j).getLabel().compareTo(pdto[i].getLibPay()) > 0) {
256                                            lpays.add(j, new SelectItem(pdto[i].getLibPay(), pdto[i].getLibPay()));
257                                            insere = true;
258                                    }
259                                    j++;
260                            }
261                            if (!insere) {
262                                    lpays.add(new SelectItem(pdto[i].getLibPay(), pdto[i].getLibPay()));
263                            }
264                    }
265    
266                    lvillesan = new ArrayList<SelectItem>();
267                    if (cdtoan != null && (cdtoan.length > 0)) {
268                            for (int i = 0; i < cdtoan.length; i++) {
269                                    boolean insere = false;
270                                    int j = 0;
271                                    while (!insere && j < lvillesan.size()) {
272                                            if (lvillesan.get(j).getLabel().compareTo(cdtoan[i].getLibCommune()) > 0) {
273                                                    lvillesan.add(j, new SelectItem(cdtoan[i].getLibCommune(), cdtoan[i].getLibCommune()));
274                                                    insere = true;
275                                            }
276                                            if (lvillesan.get(j).getLabel().equals(cdtoan[i].getLibCommune())) {
277                                                    insere = true;
278                                            }
279                                            j++;
280                                    }
281                                    if (!insere) {
282                                            lvillesan.add(new SelectItem(cdtoan[i].getLibCommune(), cdtoan[i].getLibCommune()));
283                                    }
284                            }
285                    }
286    
287                    //on remplit la liste des villes(adresse fixe) si l'adresse est en france
288                    if (adfixepays.equals("FRANCE")) {
289                            boolFixeEtranger = false;
290                            if(adfixecp!= null && !adfixecp.equals("") && Pattern.matches("[0-9]*",adfixecp))
291                                    cdtofixe = service.getCommunes(adfixecp);
292                            
293                    } else  {
294                            boolFixeEtranger = true;
295                    }
296                    lvillesfixe = new ArrayList<SelectItem>();
297                    if (cdtofixe != null && (cdtofixe.length > 0)) {
298                            for (int i = 0; i < cdtofixe.length; i++) {
299                                    boolean insere = false;
300                                    int j = 0;
301                                    while (!insere && j < lvillesfixe.size()) {
302                                            if (lvillesfixe.get(j).getLabel().compareTo(cdtofixe[i].getLibCommune()) > 0) {
303                                                    lvillesfixe.add(j, new SelectItem(cdtofixe[i].getLibCommune(), cdtofixe[i].getLibCommune()));
304                                                    insere = true;
305                                            }
306                                            if (lvillesfixe.get(j).getLabel().equals(cdtofixe[i].getLibCommune())) {
307                                                    insere = true;
308                                            }
309                                            j++;
310                                    }
311                                    if (!insere) {
312                                            lvillesfixe.add(new SelectItem(cdtofixe[i].getLibCommune(), cdtofixe[i].getLibCommune()));
313                                    }
314                            }
315                    }
316    
317    
318                    return View.MODIFIER_ADRESSES;
319            }

325            public String enregistrerModif() {
326                    //1-vérification des parametres
327                    boolean erreur = false;
328                    boolean succes = false;
329                    String message = "";
330                    if (typehebergement.equals("")) {
331                            message = "Veuillez indiquer un type d'hébergement pour l'adresse annuelle";

335                    }

338                                    message = "Veuillez remplir le premier champ de l'adresse annuelle";
339                                    FacesMessage messageX = new FacesMessage(FacesMessage.SEVERITY_ERROR, message, message);
340                                    FacesContext.getCurrentInstance().addMessage(null, messageX);
341                                    erreur = true;
342                            }
343                            if ((adanville == null || adanville.equals("")) && !boolAnEtranger) {
344                                    message = "Veuillez sélectionner une ville pour l'adresse annuelle";
345                                    FacesMessage messageX = new FacesMessage(FacesMessage.SEVERITY_ERROR, message, message);
346                                    FacesContext.getCurrentInstance().addMessage(null, messageX);
347                                    erreur = true;
348                            }
349                            if (adanpays.equals("")) {
350                                    message = "Veuillez sélectionner un pays pour l'adresse annuelle";
351                                    FacesMessage messageX = new FacesMessage(FacesMessage.SEVERITY_ERROR, message, message);
352                                    FacesContext.getCurrentInstance().addMessage(null, messageX);
353                                    erreur = true;
354                            }
355                            if (!adantel.equals("") && (!Pattern.matches("[0-9[.]]*", adantel))){
356                                    message = "Veuillez indiquer un numéro de téléphone pour l'adresse annuelle";
357                                    FacesMessage messageX = new FacesMessage(FacesMessage.SEVERITY_ERROR, message, message);
358                                    FacesContext.getCurrentInstance().addMessage(null, messageX);
359                                    erreur = true;
360                            }
361                            if (adanEtranger.equals("") && boolAnEtranger) {
362                                    message = "Veuillez indiquer une ville pour l'adresse annuelle";
363                                    FacesMessage messageX = new FacesMessage(FacesMessage.SEVERITY_ERROR, message, message);
364                                    FacesContext.getCurrentInstance().addMessage(null, messageX);
365                                    erreur = true;
366                            }
367    
368                    }
369    
370                    if (adfixe1.equals("")) {
371                            message = "Veuillez remplir le premier champ de l'adresse fixe";
372                            FacesMessage messageX = new FacesMessage(FacesMessage.SEVERITY_ERROR, message, message);
373                            FacesContext.getCurrentInstance().addMessage(null, messageX);
374                            erreur = true;
375                    }
376                    if ((adfixeville == null || adfixeville.equals("")) && !boolFixeEtranger) {
377                            message = "Veuillez sélectionner une ville pour l'adresse fixe";
378                            FacesMessage messageX = new FacesMessage(FacesMessage.SEVERITY_ERROR, message, message);
379                            FacesContext.getCurrentInstance().addMessage(null, messageX);
380                            erreur = true;
381                    }
382                    if (adfixepays.equals("")) {
383                            message = "Veuillez sélectionner un pays pour l'adresse fixe";
384                            FacesMessage messageX = new FacesMessage(FacesMessage.SEVERITY_ERROR, message, message);
385                            FacesContext.getCurrentInstance().addMessage(null, messageX);
386                            erreur = true;
387                    }
388                    if (!adfixetel.equals("") && (!Pattern.matches("[0-9[.]]*", adfixetel))){
389                            message = "Veuillez indiquer un numéro de téléphone pour l'adresse fixe";
390                            FacesMessage messageX = new FacesMessage(FacesMessage.SEVERITY_ERROR, message, message);
391                            FacesContext.getCurrentInstance().addMessage(null, messageX);
392                            erreur = true;
393                    }
394                    if (adfixeEtranger.equals("") && boolFixeEtranger) {
395                            message = "Veuillez indiquer une ville pour l'adresse fixe";
396                            FacesMessage messageX = new FacesMessage(FacesMessage.SEVERITY_ERROR, message, message);
397                            FacesContext.getCurrentInstance().addMessage(null, messageX);
398                            erreur = true;
399                    }
400                    if (!emailPerso.equals("") && (!Pattern.matches("[a-zA-Z[._-][\\d]]*[@][a-zA-Z[._-][\\d]]*[.][a-zA-Z[._-][d]]*", emailPerso))) {
401                            message = "L'email personnnel est mal renseigné";
402                            FacesMessage messageX = new FacesMessage(FacesMessage.SEVERITY_ERROR, message, message);
403                            FacesContext.getCurrentInstance().addMessage(null, messageX);
404                            erreur = true;
405                    }
406                    if (!telPortable.equals("") && (!Pattern.matches("[0-9[.]]*", telPortable))) {
407                            message = "Le numéro de téléphone portable est mal renseigné";
408                            FacesMessage messageX = new FacesMessage(FacesMessage.SEVERITY_ERROR, message, message);
409                            FacesContext.getCurrentInstance().addMessage(null, messageX);
410                            erreur = true;
411                    }
412    
413                    //2-maj les attribut utiles au changement d'adresse du bean etudiant
414                    if (!erreur) {
415    
416                            etudiant.setAdressefixe1(adfixe1);
417                            etudiant.setAdressefixe2(adfixe2);
418                            etudiant.setAdressefixe3(adfixe3);

423                            etudiant.setAdressefixeetranger(adfixeEtranger);
424                            etudiant.setAdresseannuelleType(typehebergement);

426                            etudiant.setEmailPerso(emailPerso);
427                            etudiant.setTelPortable(telPortable);
428    
429                            if (!adresseIdentique) {
430                                    etudiant.setAdresseannuelle1(adan1);
431                                    etudiant.setAdresseannuelle2(adan2);
432                                    etudiant.setAdresseannuelle3(adan3);
433                                    etudiant.setNumerotelannuel(adantel);
434                                    etudiant.setVilleannuelle(adanville);
435                                    etudiant.setPaysannuel(adanpays);
436                                    etudiant.setAdresseannuelleetranger(adanEtranger);
437                                    etudiant.setAdresseannuellecp(getCodePostal(etudiant.getVilleannuelle(), cdtoan));
438                            } else {
439                                    etudiant.setAdresseannuelle1(adfixe1);
440                                    etudiant.setAdresseannuelle2(adfixe2);
441                                    etudiant.setAdresseannuelle3(adfixe3);
442                                    etudiant.setNumerotelannuel(adfixetel);
443                                    etudiant.setVilleannuelle(adfixeville);
444                                    etudiant.setPaysannuel(adfixepays);
445                                    etudiant.setAdresseannuelleetranger(adfixeEtranger);
446                                    etudiant.setAdresseannuellecp(etudiant.getAdressefixecp());             
447                            }
448    
449    
450    
451                            //3:
452                            succes = etudiant.majAdresses();
453                            if (!succes) {
454                                    message = "Un problème est survenu pendant la mise à jour. Veuillez réessayer ultérieurement";
455                                    FacesMessage messageX = new FacesMessage(FacesMessage.SEVERITY_ERROR, message, message);
456                                    FacesContext.getCurrentInstance().addMessage(null, messageX);
457                            }
458    
459                            etudiant.renseigneAdresses();
460                            return View.ADRESSES;
461                    } 
462    
463                    return View.MODIFIER_ADRESSES;
464    
465            }

472                    if (Pattern.matches("^[0-9]{2}[0-9]*", adancp)) { 
473                            cdtoan = service.getCommunes(adancp);
474                            String ville1 = "";
475                            lvillesan.clear();
476                            for (int i = 0; i < cdtoan.length; i++) {
477    
478                                    boolean insere = false;
479                                    int j = 0;
480                                    while (!insere && j < lvillesan.size()) {
481                                            if (lvillesan.get(j).getLabel().compareTo(cdtoan[i].getLibCommune()) > 0) {
482                                                    lvillesan.add(j, new SelectItem(cdtoan[i].getLibCommune(), cdtoan[i].getLibCommune()));
483                                                    insere = true;
484                                            }
485                                            if (lvillesan.get(j).getLabel().equals(cdtoan[i].getLibCommune())) {
486                                                    insere = true;
487                                            }
488                                            if (!insere) {
489                                                    j++;
490                                            }
491                                    }
492                                    if (!insere) {
493                                            lvillesan.add(new SelectItem(cdtoan[i].getLibCommune(), cdtoan[i].getLibCommune()));
494                                    }
495                                    //Pour afficher le CP de la premiere ville directement
496                                    if (j == 0 && !cdtoan[i].getLibCommune().equals(ville1)) {
497                                            adancp = cdtoan[i].getCodePostal();
498                                            ville1 = cdtoan[i].getLibCommune();
499                                    }
500                            }
501    
502                            adanville = lvillesan.get(0).getLabel();
503    
504                    } else {
505                            String message = "Le code être composé d'au moins 2 chiffres";
506                            FacesMessage messageX = new FacesMessage(FacesMessage.SEVERITY_ERROR, message, message);
507                            FacesContext.getCurrentInstance().addMessage(null, messageX);
508                    }
509                    return View.MODIFIER_ADRESSES;
510            }

596             * @param ville
597             * @return le codePostal de la ville
598             */
599            private String getCodePostal(final String ville, CommuneDTO[] cdt) {
600                    String cp = "";
601                    boolean trouve = false;
602                    int i = 0;
603                    if(cdt != null){ //Pays Etranger
604                            while (!trouve && i < cdt.length) {
605                                    if (cdt[i].getLibCommune().equals(ville)) {
606                                            cp = cdt[i].getCodePostal();
607                                            trouve = true;
