133            public Map<String, Boolean> getCertScolAutorisations() {
134                    if(!etudiant.getCod_etu().equals(codetuMapAutorisationsCert)){
135                            //on reinitialise pour le cas d'un user Enseignant qui peut consulter plusieurs dossiers à la suite
136                            autorisationCertificats=null;
137                    }
138                    if (autorisationCertificats == null){
139                            Config config = (Config)  BeanUtils.getBean("config");
140                            DaoServiceIBatisImpl daoService = (DaoServiceIBatisImpl) BeanUtils.getBean("daoService");
141                            if (anneeEnCours == null){
142                                    anneeEnCours = daoService.getAnneeUniversitaireEnCours();
143                            }
144                            autorisationCertificats = new HashMap<String, Boolean>();
145                            Etudiant etu = getEtatcivilcontroller().getEtudiant();
146    
147                            for (Inscription ins : etu.getLinsciae()) {
148                                    boolean mapUpdated = false;
149                                    // autoriser ou non la generation de certificats de scolarite.
150                                    if (!config.isCertificatScolaritePDF()) {
151                                            autorisationCertificats.put(ins.getCodEtpCodAnuConcat(), false);
152                                            mapUpdated=true;
153                                    }
154                                    // autoriser ou non les personnels à imprimer les certificats.
155                                    if (!mapUpdated && !config.isCertScolAutorisePersonnel() && getSessionController().isEnseignant()) {
156                                            autorisationCertificats.put(ins.getCodEtpCodAnuConcat(), false);
157                                            mapUpdated=true;
158                                    }
159                                    // autorise l'édition de certificat de scolarité uniquement pour l'année en cours.
160                                    if (!mapUpdated && !config.isCertificatScolariteTouteAnnee() && !ins.getCod_anu().equals(anneeEnCours)) {
161                                            autorisationCertificats.put(ins.getCodEtpCodAnuConcat(), false);
162                                            mapUpdated=true;
163                                    }
164                                    if (!mapUpdated && !config.getCertScolTypDiplomeDesactive().isEmpty()) {
165                                            // interdit les certificats pour certains types de diplomes
166                                            String codeTypeDiplome = daoService.getCodeTypeDiplome(ins.getCod_dip());
167    
168                                            if (config.getCertScolTypDiplomeDesactive().contains(codeTypeDiplome)) {
169                                                    autorisationCertificats.put(ins.getCodEtpCodAnuConcat(), false);
170                                                    mapUpdated=true;
171                                            }
172                                    }
173                                    //interdit l'edition de certificat pour les étudiants si l'inscription n'est pas payée
174                                    if (!mapUpdated && !ins.isEstEnRegle() && !getSessionController().isEnseignant()){
175                                            autorisationCertificats.put(ins.getCodEtpCodAnuConcat(), false);
176                                            mapUpdated=true;
177                                    }
178                                    if(!mapUpdated)
179                                            autorisationCertificats.put(ins.getCodEtpCodAnuConcat(), true);
180                            }
181    
182                    }
183                    codetuMapAutorisationsCert = etudiant.getCod_etu();
184                    return autorisationCertificats;
185            }

187            public Map<String, Boolean> getIpAutorisations() {
188                    if(!etudiant.getCod_etu().equals(codetuMapAutorisationsIP)){
189                            //on reinitialise pour le cas d'un user Enseignant qui peut consulter plusieurs dossiers à la suite
190                            autorisationIP=null;
191                    }
192                    if (autorisationIP == null){
193                            if (anneeEnCours == null){
194                                    DaoServiceIBatisImpl daoService = (DaoServiceIBatisImpl) BeanUtils.getBean("daoService");
195                                    anneeEnCours = daoService.getAnneeUniversitaireEnCours();
196                            }
197                            autorisationIP = new HashMap<String, Boolean>();
198                            for (Inscription ins : etudiant.getLinsciae()) {
199                                    // autoriser ou non la visualisation du détail de l'IP.
200                                    if (ins.getCod_anu().equals(anneeEnCours)) {
201                                            autorisationIP.put(ins.getCodEtpCodAnuConcat(), true);
202    
203                                    }else{
204                                            autorisationIP.put(ins.getCodEtpCodAnuConcat(), false);
205                                    }
206                            }
207                    }
208                    codetuMapAutorisationsIP = etudiant.getCod_etu();
209                    return autorisationIP;
210            }

212            public String preciserEtapeInscription() {
213                    FacesContext context = FacesContext.getCurrentInstance();
214                    String row = (String) context.getExternalContext().getRequestParameterMap().get("row");
215    
216                    if(row != null && !row.equals("")){
217                            selectedRow = new Integer(row);
218    
219                            //etudiant = etatcivilcontroller.getEtudiant();
220    
221                            Etape selectedEtape = new Etape(etudiant.getLinsciae().get(selectedRow).getCod_etp(),
222                                            etudiant.getLinsciae().get(selectedRow).getCod_vrs_vet(),
223                                            etudiant.getLinsciae().get(selectedRow).getCod_anu());
224                            selectedEtape.setLibelle(etudiant.getLinsciae().get(selectedRow).getLib_etp());
225    
226                            etudiant.getEtudiantManager().setInscriptionPedagogique(etudiant, selectedEtape);
227    
228    
229                            return View.INSCRIPTION_PEDAGOGIQUE;
230                    }
231                    return null;
232            }

238            public String exportPdf() {
239    
240                    FacesContext facesContext = FacesContext.getCurrentInstance();
241    
242                    ExternalContext externalContext = facesContext.getExternalContext();
243    
244                    etudiant = etatcivilcontroller.getEtudiant();
245    
246                    String row = (String) facesContext.getExternalContext().getRequestParameterMap().get("row");
247                    int selectedRow=0;
248                    if(row != null && !row.equals("")){
249                            selectedRow = new Integer(row);
250                    }else{
251                            return null;
252                    }
253                    inscription = etudiant.getLinsciae().get(selectedRow);
254    
255                    // verifie les autorisations
256                    if (!getCertScolAutorisations().get(inscription.getCodEtpCodAnuConcat())) {
257                            return null;
258                    }
259    
260                    Document document = configureDocument();
261                    try {
262    
263                            ByteArrayOutputStream baosPDF = new ByteArrayOutputStream(OUTPUTSTREAM_SIZE);
264                            PdfWriter docWriter = null;
265                            docWriter = PdfWriter.getInstance(document, baosPDF);
266                            docWriter.setEncryption(null, null, PdfWriter.AllowPrinting, PdfWriter.ENCRYPTION_AES_128);
267                            docWriter.setStrictImageSequence(true);
268                            creerPdfCertificatScolarite(document);
269                            docWriter.close();
270    
271                            byte[] bytes = baosPDF.toByteArray();
272                            String nomFichier = getString("PDF.TITLE.CERTIFICAT")+"_" + inscription.getCod_etp() + "_" + inscription.getCod_anu().replace('/', '-') + "_" + etatcivilcontroller.getEtudiant().getNom().replace('.', ' ').replace(' ', '_') + ".pdf";
273    
274                            long id = DownloadUtils.setDownload(bytes, nomFichier, "application/force-download", "attachment");
275    
276                            String url = DownloadUtils.getDownloadUrl(id);
277    
278                            externalContext.redirect(url);
279    
280                            bytes = null;
281                    } catch (DownloadException e) {
282                            LOG.error(e);
283                    } catch (DocumentException e) {
284                            LOG.error(e);
285                    } catch (IOException e) {
286                            LOG.error(e);
287                    }
288    
289                    facesContext.responseComplete();
290    
291                    return null;
292    
293            }

295            /**
296             * 
297             * @param document pdf
298             */
299            public void creerPdfCertificatScolarite(final Document document) {
300    
301                    //configuration des fonts
302                    Font normal = FontFactory.getFont(FontFactory.HELVETICA, 10, Font.NORMAL);
303                    Font normalBig = FontFactory.getFont(FontFactory.HELVETICA, 12, Font.BOLD);
304                    Font header = FontFactory.getFont(FontFactory.HELVETICA, 14, Font.BOLD);
305    
306                    //date
307                    Date d = new Date();
308                    DateFormat dateFormat = new SimpleDateFormat("dd/MM/yyyy");
309                    String date = dateFormat.format(d);
310    
311                    document.open();
312                    try {
313                            Config config = (Config)  BeanUtils.getBean("config");
314                            DaoServiceIBatisImpl daoService = (DaoServiceIBatisImpl) BeanUtils.getBean("daoService");
315                    
316                            Signataire signataire = daoService.getSignataire(config.getCertScolCodeSignataire());
317    
318                            //ajout image test
319                            if (config.isCertScolUtiliseLogo() && config.getLogoUniversitePdf() != null && !config.getLogoUniversitePdf().equals("")){
320                                    Image imageLogo = Image.getInstance(config.getLogoUniversitePdf());
321                                    /*
322                                    imageLogo.scalePercent(72f/(float)config.getImagesDPI()*100f);
323                                    imageLogo.setAbsolutePosition((MARGIN_LEFT / 2.54f) * 72f, document.getPageSize().height() - ((MARGIN_TOP / 2.54f) * 72f));
324                                     */
325                                    float scaleRatio = 40 / imageLogo.height(); 
326                                    float newWidth=scaleRatio * imageLogo.width();
327                                    imageLogo.scaleAbsolute(newWidth, 40);
328                                    imageLogo.setAbsolutePosition(100, 750);
329                                    document.add(imageLogo);
330                            }
331    
332                            Paragraph pTitre = new Paragraph("\n\n"+getString("PDF.TITLE.CERTIFICAT").toUpperCase(), header);
333                            pTitre.setAlignment(Element.ALIGN_CENTER);
334                            document.add(pTitre);
335    
336                            Paragraph pCertifie = new Paragraph("\n\n\n\n" + signataire.getQua_sig() + " "+getString("PDF.CERTIFICAT.CERTIFIE")+"\n\n", normal);
337                            pCertifie.setAlignment(Element.ALIGN_LEFT);
338                            document.add(pCertifie);
339    
340                            if (etudiant.getNom() != null) {
341                                    Paragraph pNom = new Paragraph(etudiant.getNom(), normalBig);
342                                    pNom.indentationLeft();
343                                    document.add(pNom);
344                            }
345                            if (etudiant.getCod_nne() != null) {
346                                    Paragraph pNNE = new Paragraph("\n"+getString("PDF.CERTIFICAT.ID")+" : " + etudiant.getCod_nne().toLowerCase(), normal);
347                                    pNNE.indentationLeft();
348                                    document.add(pNNE);
349                            }
350                            if (etudiant.getCod_etu() != null) {
351                                    Paragraph p01 = new Paragraph(getString("PDF.CERTIFICAT.NUMETUDIANT")+" : " + etudiant.getCod_etu(), normal);
352                                    p01.setAlignment(Element.ALIGN_LEFT);
353                                    document.add(p01);
354                            }
355                            if (etudiant.getDatenaissance() != null) {
356                                    Paragraph pDateNaissance = new Paragraph(getString("PDF.CERTIFICAT.NAISSANCE1")+" " + etudiant.getDatenaissance(), normal);
357                                    pDateNaissance.setAlignment(Element.ALIGN_LEFT);
358                                    document.add(pDateNaissance);
359                            }
360                            if ((etudiant.getLieunaissance() != null) && (etudiant.getDepartementnaissance() != null)) {
361                                    Paragraph pLieuNaissance = new Paragraph(getString("PDF.CERTIFICAT.NAISSANCE2")+" " + etudiant.getLieunaissance() + " (" + etudiant.getDepartementnaissance() + ")", normal);
362                                    pLieuNaissance.setAlignment(Element.ALIGN_LEFT);
363                                    document.add(pLieuNaissance);
364                            }
365    
366                            Paragraph pEstInscrit = new Paragraph("\n"+getString("PDF.CERTIFICAT.INSCRIT")+" " + inscription.getCod_anu() + "\n ", normal);
367                            pEstInscrit.setAlignment(Element.ALIGN_LEFT);
368                            document.add(pEstInscrit);
369    
370                            float[] widths = {1.5f, 7.5f};
371                            PdfPTable table = new PdfPTable(widths);
372                            table.setWidthPercentage(100f);
373                            table.addCell(makeCell(getString("PDF.DIPLOME")+" :", normal));
374                            table.addCell(makeCell(inscription.getLib_dip(), normal));
375                            table.addCell(makeCell(getString("PDF.YEAR")+" :", normal));
376                            table.addCell(makeCell(inscription.getLib_etp(), normal));
377                            table.addCell(makeCell(getString("PDF.COMPOSANTE")+" :", normal));
378                            table.addCell(makeCell(inscription.getLib_comp(), normal));
379                            document.add(table);
380    
381                            document.add(new Paragraph(" "));
382    
383                            float[] widthsSignataire = {2f, 1.3f};
384                            PdfPTable tableSignataire = new PdfPTable(widthsSignataire);
385                            tableSignataire.setWidthPercentage(100f);
386                            tableSignataire.addCell(makeCellSignataire("", normal));
387                            tableSignataire.addCell(makeCellSignataire(getString("PDF.CERTIFICAT.FAIT1")+" " + config.getCertScolLieuEdition() + getString("PDF.CERTIFICAT.FAIT2")+" " + date, normal));
388                            tableSignataire.addCell(makeCellSignataire("", normal));
389                            tableSignataire.addCell(makeCellSignataire(signataire.getNom_sig(), normal));
390                            //ajout signature
391                            if (signataire.getImg_sig_std() != null){
392                                    tableSignataire.addCell(makeCellSignataire("", normal));
393                                    LOG.debug(signataire.getImg_sig_std().toString());
394                                    Image imageSignature = Image.getInstance(signataire.getImg_sig_std());
395    
396                                    float scaleRatio = 40 / imageSignature.height(); 
397                                    float newWidth=scaleRatio * imageSignature.width();
398                                    imageSignature.scaleAbsolute(newWidth, 40);
399                                    imageSignature.setAbsolutePosition(100, 750);
400    
401                                    PdfPCell cellSignature = new PdfPCell();
402                                    cellSignature.setBorder(0);
403                                    cellSignature.setImage(imageSignature);
404                                    //cellSignature.setFixedHeight(72f/(float)config.getImagesDPI() * imageSignature.height());
405                                    cellSignature.setHorizontalAlignment(Element.ALIGN_CENTER);
406                                    tableSignataire.addCell(cellSignature);
407                            }
408                            document.add(tableSignataire);
409    
410                    } catch (BadElementException e) {
411                            LOG.error(e);
412                    } catch (MalformedURLException e) {
413                            LOG.error(e);
414                    } catch (IOException e) {
415                            LOG.error(e);
416                    } catch (DocumentException e) {
417                            LOG.error(e);
418                    }
419                    // step 6: fermeture du document.
420                    document.close();
421    
422    
423    
424    
425            }

427            private PdfPCell makeCell(String str, Font font) {
428                    PdfPCell cell = new PdfPCell();
429                    cell.setBorder(0);
430                    cell.setPhrase(new Phrase(str, font));
431                    return cell;
432            }
433    
434            private PdfPCell makeCellSignataire(String str, Font font) {
435                    PdfPCell cell = makeCell(str, font);
436                    cell.setHorizontalAlignment(Element.ALIGN_CENTER);
437                    return cell;
438            }

440            /**
441             * configure le document pdf.
442             * @param width
443             * @param height
444             * @param margin
445             * @return doc
446             */
447            private Document configureDocument() {
448    
449                    Document document = new Document();
450    
451                    document.setPageSize(PageSize.A4);
452                    float marginTop = (MARGIN_TOP / 2.54f) * 72f;
453                    float marginRight = (MARGIN_RIGHT / 2.54f) * 72f;
454                    float marginBottom = (MARGIN_BOTTOM / 2.54f) * 72f;
455                    float marginLeft = (MARGIN_LEFT / 2.54f) * 72f;
456                    document.setMargins(marginLeft, marginRight, marginTop, marginBottom);
457    
458                    return document;
459            }
