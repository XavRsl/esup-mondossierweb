005    package org.esupportail.mondossierweb.web.controllers;
006    
007    import org.esupportail.mondossierweb.domain.beans.User;
008    import org.esupportail.commons.utils.Assert;
009    
010    
011    
012    /**
013     * An abstract class inherited by all the beans for them to get:
014     * - the context of the application (sessionController).
015     * - the domain service (domainService).
016     * - the application service (applicationService).
017     * - the i18n service (i18nService).
018     */
019    public abstract class AbstractContextAwareController extends AbstractDomainAwareBean {
020    
021            /**
022             * 
023             */
024            private static final long serialVersionUID = 8225406557666684184L;
025            /**
026             * The SessionController.
027             */
028            private SessionController sessionController;
029            
030    
031    
032            /**
033             * Constructor.
034             */
035            protected AbstractContextAwareController() {
036                    super();
037            }
038    
039            /**
040             * @see org.esupportail.mondossierweb.web.controllers.AbstractDomainAwareBean#afterPropertiesSetInternal()
041             */
042            @Override
043            public void afterPropertiesSetInternal() {
044                    Assert.notNull(this.sessionController, "property sessionController of class " 
045                                    + this.getClass().getName() + " can not be null");
046            }
047    
048            /**
049             * @param sessionController the sessionController to set
050             */
051            public void setSessionController(final SessionController sessionController) {
052                    this.sessionController = sessionController;
053            }
054    
055            /**
056             * @return the sessionController
057             */
058            public SessionController getSessionController() {
059                    return sessionController;
060            }
061    
062            /**
063             * @see org.esupportail.mondossierweb.web.controllers.AbstractDomainAwareBean#getCurrentUser()
064             */
065            @Override
066            protected User getCurrentUser() {
067                    return sessionController.getCurrentUser();
068            }
069    
070    
071    
072    }
