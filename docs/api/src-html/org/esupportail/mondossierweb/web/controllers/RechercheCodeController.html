110            /**
111             * valide le formulaire et transmet au controller qui liste les inscrits.
112             * @return la vue suivante
113             */
114            public String chercher() {
115                    try {
116                            rechercheParCode = true;
117                            boolean erreur = false;
118                            String message = "";
119                            libGpe = "";
120                            if (!(annee != null && !annee.equals(""))) {
121                                    message = "Veuillez entrer une année";
122                                    FacesMessage messageX = new FacesMessage(FacesMessage.SEVERITY_ERROR, message, message);
123                                    FacesContext.getCurrentInstance().addMessage(null, messageX);
124                                    erreur = true;
125                            } else {
126                                    annee = annee.trim();
127                                    if (!Pattern.matches("^[0-9]{4}", annee)) {
128                                            message = "L'année doit être composée de 4 chiffres";
129                                            FacesMessage messageX = new FacesMessage(FacesMessage.SEVERITY_ERROR, message, message);
130                                            FacesContext.getCurrentInstance().addMessage(null, messageX);
131                                            erreur = true;
132                                    }
133                            }
134    
135                            if (!(code != null && !code.equals(""))) {
136                                    message = "Veuillez entrer un code";
137                                    FacesMessage messageX = new FacesMessage(FacesMessage.SEVERITY_ERROR, message, message);
138                                    FacesContext.getCurrentInstance().addMessage(null, messageX);
139                                    erreur = true;
140                            }else{
141                                    code=code.trim();
142                            }
143                            if (type.equals("Etape")  &&    !(version != null && !version.equals(""))) {
144                                    message = "Veuillez entrer une version";
145                                    FacesMessage messageX = new FacesMessage(FacesMessage.SEVERITY_ERROR, message, message);
146                                    FacesContext.getCurrentInstance().addMessage(null, messageX);
147                                    erreur = true;
148                            }else{
149                                    if (type.equals("Etape")){
150                                            version = version.trim();
151                                    }
152                            }
153                            if (type.equals("Etape")  &&    !(Pattern.matches("^[0-9]*", version))) {
154                                    message = "Version mal renseignée";
155                                    FacesMessage messageX = new FacesMessage(FacesMessage.SEVERITY_ERROR, message, message);
156                                    FacesContext.getCurrentInstance().addMessage(null, messageX);
157                                    erreur = true;
158                            }
159    
160                            if (!erreur) {          
161    
162                                    if (type.equals("Etape")) {
163                                            //Recherche d'une Etape
164                                            etape = false;
165                                            Etape e = new Etape(code, version, annee);
166                                            int check = service.checkEtape(e);
167                                            if (check > 0) {
168                                                    renseigneController();
169                                                    return listeInscritsController.enter();
170                                            } 
171                                            message = "Etape inexistante";
172                                            FacesMessage messageX = new FacesMessage(FacesMessage.SEVERITY_ERROR, message, message);
173                                            FacesContext.getCurrentInstance().addMessage(null, messageX);
174    
175                                    } else {
176                                            if (type.equals("Groupe")) {
177                                                    //Recherche des groupes d'un ELP
178                                                    int check = service.checkElementPedagogique(code);
179                                                    if (check > 0) {
180    
181                                                            //on cherche les groupes
182                                                            if(listeElpDeCollection != null && listeElpDeCollection.size() > 0){
183                                                                    listeElpDeCollection.clear();
184                                                            }
185                                                            try{
186                                                                    listeElpDeCollection = service.recupererGroupes(annee,code);
187    
188                                                                    return View.AFFICHE_RECHERCHE_GROUPES;
189                                                            }catch(WebBaseException e){
190                                                                    message = "Aucun groupe récupéré pour cet ELP";
191                                                                    FacesMessage messageX = new FacesMessage(FacesMessage.SEVERITY_ERROR, message, message);
192                                                                    FacesContext.getCurrentInstance().addMessage(null, messageX);
193    
194                                                            }
195                                                    } else{
196                                                            message = "Elément pédagogique inexistant";
197                                                            FacesMessage messageX = new FacesMessage(FacesMessage.SEVERITY_ERROR, message, message);
198                                                            FacesContext.getCurrentInstance().addMessage(null, messageX);
199                                                    }
200                                            }else{
201                                                    //Recherche Element pédagogique
202                                                    int check = service.checkElementPedagogique(code);
203                                                    if (check > 0) {
204                                                            renseigneController();
205                                                            return listeInscritsController.enter();
206                                                    } 
207                                                    message = "Elément pédagogique inexistant";
208                                                    FacesMessage messageX = new FacesMessage(FacesMessage.SEVERITY_ERROR, message, message);
209                                                    FacesContext.getCurrentInstance().addMessage(null, messageX);
210                                            }
211                                    }
212                            }       
213    
214                            return View.CHERCHE_CODE_ARBO;
215                    } catch(Exception e) {
216                            LOG.error(e);
217                            return View.EXCEPTION_ERROR;
218                    }
219            }

221            public String enterViaArborescencePourGroupe(String anne, String cod){
222                    annee = anne;
223                    code = cod;
224                    type = "Groupe";
225                    rechercheParCode = false;
226                    
227                    int check = service.checkElementPedagogique(code);
228                    if (check > 0) {
229    
230                            //on cherche les groupes
231                            if(listeElpDeCollection != null && listeElpDeCollection.size() > 0){
232                                    listeElpDeCollection.clear();
233                            }
234                            try{
235                                    listeElpDeCollection = service.recupererGroupes(annee,code);
236    
237                                    
238                            }catch(WebBaseException e){
239                                    //message = "Aucun groupe récupéré pour cet ELP";
240                                    //FacesMessage messageX = new FacesMessage(FacesMessage.SEVERITY_ERROR, message, message);
241                                    //FacesContext.getCurrentInstance().addMessage(null, messageX);
242    
243                            }
244                    }
245                    return View.AFFICHE_RECHERCHE_GROUPES;
246            }

248            /**
249             * @return la vue du formulaire
250             */
251            public String enter() {
252                    try {
253                            annee = service.getAnneeEnCours();
254                    } catch(Exception e) {
255                            LOG.error(e);
256                            LOG.error("iBATIS : probleme de récupération de l'année en cours dans la base");
257                            return View.EXCEPTION_ERROR;
258                    }
259                    type = "Etape";
260                    code = "";
261                    version = "";
262                    libGpe = "";
263                    session1 = false;
264                    session2 = false;
265                    rechercheParCode = true;
266                    etape = false;
267                    return View.CHERCHE_CODE_ARBO;
268            }

282            /**
283             * 
284             * @return
285             */
286            public String validGroupe(){
287    
288                    //recupère les infos du groupe sélectionné dans les attributs a passer a listeInscritsController
289                    FacesContext context = FacesContext.getCurrentInstance();
290                    code = (String) context.getExternalContext().getRequestParameterMap().get("code");
291                    
292                    libGpe = service.getLibelleGroupe(code);
293    
294                    renseigneController();
295                    return listeInscritsController.enter();
296            }
297            /**
298             * renseigne les attributs utiles du controller listeInscritsController.
299             */
300            private void renseigneController() {

302                    listeInscritsController.setCode(code);
303                    listeInscritsController.setVersion(version);
304                    listeInscritsController.setEtape(etape);
305                    listeInscritsController.setSession1(session1);
306                    listeInscritsController.setSession2(session2);
307                    listeInscritsController.setType(type);
308                    listeInscritsController.setLibelle(libGpe);
309                    listeInscritsController.setRecherchecode(true);
310            }
