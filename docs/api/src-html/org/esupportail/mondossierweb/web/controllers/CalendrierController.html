118            /**
119             * 
120             * @return le fichier pdf.
121             */
122            public String exportPdf() {
123                    FacesContext facesContext = FacesContext.getCurrentInstance();
124    
125                    ExternalContext externalContext = facesContext.getExternalContext();
126    
127    
128                    
129                            Document document = configureDocument( MARGE_PDF);
130    
131                            try {
132    
133                                    ByteArrayOutputStream baosPDF = new ByteArrayOutputStream(OUTPUTSTREAM_SIZE);
134                                    PdfWriter docWriter = null;
135                                    docWriter = PdfWriter.getInstance(document, baosPDF);
136                                    docWriter.setStrictImageSequence(true);
137                                    String titre = getString("PDF.TITLE.CALENDRIER")+" " + etudiant.getNom().replace('.', ' ').replace(' ', '_') + ".pdf";
138    
139    
140                                    creerPdfCalendrier(document);
141    
142                                    docWriter.close();
143                                    byte[] bytes = baosPDF.toByteArray();
144    
145                                    String typeDownload = "application/force-download";
146                                    
147                                    long id = DownloadUtils.setDownload(bytes, titre, typeDownload, "attachment");
148                                    
149                                    String url = DownloadUtils.getDownloadUrl(id);
150                                    
151                                    externalContext.redirect(url);
152                                    
153                                    bytes = null;
154                                    
155                            } catch (DownloadException e) {
156                                    LOG.error(e);
157                            } catch (DocumentException e) {
158                                    LOG.error(e);
159                            } catch (IOException e) {
160                                    LOG.error(e);
161                            }
162                            facesContext.responseComplete();
163            
164                    return null;
165    
166            }

171            /**
172             * 
173             * @param document pdf
174             */
175            public void creerPdfCalendrier(final Document document) {
176    
177    
178    
179                    //configuration des fonts
180                    Font normal = FontFactory.getFont(FontFactory.TIMES_ROMAN, 10, Font.NORMAL);
181                    Font normalbig = FontFactory.getFont(FontFactory.TIMES_ROMAN, 11, Font.BOLD);
182                    Font legerita = FontFactory.getFont(FontFactory.TIMES_ROMAN, 9, Font.ITALIC);
183                    Font headerbig = FontFactory.getFont(FontFactory.TIMES_ROMAN, 16, Font.BOLD);
184                    Font header = FontFactory.getFont(FontFactory.TIMES_ROMAN, 12, Font.BOLD);
185    
186                    //pieds de pages:
187                    Date d = new Date();
188                    DateFormat dateFormat = new SimpleDateFormat("dd-MM-yyyy");
189                    String date = dateFormat.format(d);
190                    //alignement des libellés du pied de page:
191                    String partie1 = getString("PDF.TITLE.CALENDRIER");
192                    String partie2 = getString("PDF.EDITION.DATE")+" : " + date;
193                    if (partie1.length() < ECARTEMENT_PIED_PAGE_PDF) {
194                            int diff = ECARTEMENT_PIED_PAGE_PDF - partie1.length();
195                            for (int i = 0; i < diff; i++) {
196                                    partie1 = partie1 + " ";
197    
198                            }
199                    } 
200                    if (partie2.length() < ECARTEMENT_PIED_PAGE_PDF) {
201                            int diff = ECARTEMENT_PIED_PAGE_PDF - partie2.length();
202                            for (int i = 0; i < diff; i++) {
203                                    partie2 = " " + partie2;
204                            }
205                    }
206    
207                    //création du pied de page:
208                    Phrase phra = new Phrase(partie1 + " -"+getString("PDF.PAGE"), legerita);
209                    Phrase phra2 = new Phrase("- "+partie2, legerita);
210                    HeaderFooter hf = new HeaderFooter(phra, phra2);
211                    hf.setAlignment(HeaderFooter.ALIGN_CENTER);
212                    document.setFooter(hf);  
213    
214    
215                    //ouverte du document.
216                    document.open();
217                    try {
218                            //ajout image test
219                            Config config = (Config)  BeanUtils.getBean("config");
220                            if (config.getLogoUniversitePdf() != null && !config.getLogoUniversitePdf().equals("")){
221                                    Image image1 = Image.getInstance(config.getLogoUniversitePdf());
222                                    float scaleRatio = 40 / image1.height();
223                                    float newWidth=scaleRatio * image1.width();
224                                    image1.scaleAbsolute(newWidth, 40);
225                                    image1.setAbsolutePosition(800 - newWidth, 528);
226                                    document.add(image1);
227                            }
228    
229    
230    
231                            //nouveau paragraphe
232                            Paragraph p = new Paragraph(getString("PDF.TITLE.CALENDRIER").toUpperCase()+"\n\n", headerbig);
233                            p.indentationLeft();
234                            document.add(p);
235    
236                            if (etudiant.getNom() != null) {
237                                    Paragraph p0 = new Paragraph(etudiant.getNom(), normal);
238                                    p0.indentationLeft();
239                                    document.add(p0);
240                            }
241                            if (etudiant.getCod_etu() != null) {
242                                    Paragraph p01 = new Paragraph(getString("PDF.FOLDER")+" : " + etudiant.getCod_etu(), normal);
243                                    p01.indentationLeft();
244                                    document.add(p01);
245                            }
246                            if (etudiant.getCod_nne() != null) {
247                                    Paragraph p02 = new Paragraph(getString("PDF.NNE")+" : " + etudiant.getCod_nne(), normal);
248                                    p02.indentationLeft();
249                                    document.add(p02);
250                            }
251                            if (etudiant.getEmail() != null) {
252                                    Paragraph p03 = new Paragraph(getString("PDF.MAIL") + " : " + etudiant.getEmail(), normal);
253                                    p03.indentationLeft();
254                                    document.add(p03);
255                            }
256    
257                            Paragraph p03 = new Paragraph(getString("PDF.EDITION.DATE") + " : " + date, normal);
258                            p03.indentationLeft();
259                            document.add(p03);
260                            document.add(new Paragraph("\n"));
261    
262    
263    
264    
265                            //Partie Calendrier
266                            PdfPTable table = new PdfPTable(1);
267                            table.setWidthPercentage(98);
268                            PdfPCell cell = new PdfPCell(new Paragraph(getString("PDF.SUBTITLE.CALENDRIER").toUpperCase()+" ", header));
269                            cell.setBorder(Rectangle.NO_BORDER);
270                            cell.setBackgroundColor(new Color(153, 153, 255));
271                            table.addCell(cell);
272    
273                            if(etudiant.getCalendrier()!=null && etudiant.getCalendrier().size()>0){
274    
275                                    PdfPTable table2 = new PdfPTable(7);
276    
277                                    table2.setWidthPercentage(98);
278                                    int [] tabWidth = {15,10,10,40,30,10,60};
279                                    table2.setWidths(tabWidth);
280    
281                                    Paragraph p1 = new Paragraph(getString("PDF.DATE"),normalbig);
282                                    Paragraph p2 = new Paragraph(getString("PDF.HEURE"),normalbig);
283                                    Paragraph p3 = new Paragraph(getString("PDF.DUREE"),normalbig);
284                                    Paragraph p4 = new Paragraph(getString("PDF.BATIMENT"),normalbig);
285                                    Paragraph p5 = new Paragraph(getString("PDF.SALLE"),normalbig);
286                                    Paragraph p6 = new Paragraph(getString("PDF.PLACE"),normalbig);
287                                    Paragraph p7 = new Paragraph(getString("PDF.EXAMEN"),normalbig);
288    
289                                    PdfPCell ct1 = new PdfPCell(p1);
290                                    PdfPCell ct2 = new PdfPCell(p2);
291                                    PdfPCell ct3 = new PdfPCell(p3);
292                                    PdfPCell ct4 = new PdfPCell(p4);
293                                    PdfPCell ct5 = new PdfPCell(p5);
294                                    PdfPCell ct6 = new PdfPCell(p6);
295                                    PdfPCell ct7 = new PdfPCell(p7);
296    
297                                    ct1.setBorder(Rectangle.BOTTOM); ct1.setBorderColorBottom(Color.black);
298                                    ct2.setBorder(Rectangle.BOTTOM); ct2.setBorderColorBottom(Color.black);
299                                    ct3.setBorder(Rectangle.BOTTOM); ct2.setBorderColorBottom(Color.black);
300                                    ct4.setBorder(Rectangle.BOTTOM); ct1.setBorderColorBottom(Color.black);
301                                    ct5.setBorder(Rectangle.BOTTOM); ct2.setBorderColorBottom(Color.black);
302                                    ct6.setBorder(Rectangle.BOTTOM); ct2.setBorderColorBottom(Color.black);
303                                    ct7.setBorder(Rectangle.BOTTOM); ct2.setBorderColorBottom(Color.black);
304    
305                                    table2.addCell(ct1);
306                                    table2.addCell(ct2);
307                                    table2.addCell(ct3);
308                                    table2.addCell(ct4);
309                                    table2.addCell(ct5);
310                                    table2.addCell(ct6);
311                                    table2.addCell(ct7);
312    
313    
314    
315    
316                                    for (int i = 0; i < etudiant.getCalendrier().size(); i++) {
317                                            Paragraph pa = new Paragraph(etudiant.getCalendrier().get(i).getDate(), normal);
318                                            PdfPCell celltext = new PdfPCell(pa);
319                                            celltext.setBorder(Rectangle.NO_BORDER);
320    
321                                            Paragraph pa2 = new Paragraph(etudiant.getCalendrier().get(i).getHeure(), normal);
322                                            PdfPCell celltext2 = new PdfPCell(pa2);
323                                            celltext2.setBorder(Rectangle.NO_BORDER);
324    
325                                            Paragraph pa3 = new Paragraph(etudiant.getCalendrier().get(i).getDuree(), normal);
326                                            PdfPCell celltext3 = new PdfPCell(pa3);
327                                            celltext3.setBorder(Rectangle.NO_BORDER);
328    
329                                            Paragraph pa4 = new Paragraph(etudiant.getCalendrier().get(i).getBatiment(), normal);
330                                            PdfPCell celltext4 = new PdfPCell(pa4);
331                                            celltext4.setBorder(Rectangle.NO_BORDER);
332    
333                                            Paragraph pa5 = new Paragraph(etudiant.getCalendrier().get(i).getSalle(), normal);
334                                            PdfPCell celltext5 = new PdfPCell(pa5);
335                                            celltext5.setBorder(Rectangle.NO_BORDER);
336    
337                                            Paragraph pa6 = new Paragraph(etudiant.getCalendrier().get(i).getPlace(), normal);
338                                            PdfPCell celltext6 = new PdfPCell(pa6);
339                                            celltext6.setBorder(Rectangle.NO_BORDER);
340    
341                                            Paragraph pa7 = new Paragraph(etudiant.getCalendrier().get(i).getEpreuve(), normal);
342                                            PdfPCell celltext7 = new PdfPCell(pa7);
343                                            celltext7.setBorder(Rectangle.NO_BORDER);
344    
345                                            table2.addCell(celltext);
346                                            table2.addCell(celltext2);
347                                            table2.addCell(celltext3);
348                                            table2.addCell(celltext4);
349                                            table2.addCell(celltext5);
350                                            table2.addCell(celltext6);
351                                            table2.addCell(celltext7);
352    
353                                            /*PdfPCell celltext4 = new PdfPCell(table3);
354                                    celltext4.setBorder(Rectangle.NO_BORDER);
355                                    table2.addCell(celltext4);*/
356    
357                                    }
358                                    document.add(table);
359                                    document.add(table2);
360                                    document.add(new Paragraph("\n"));
361                            }else{
362    
363    
364                                    document.add(table);
365                                    Paragraph pinfo = new Paragraph("      "+getString("PDF.SUBTITLE.NOCALENDRIER"), normalbig);
366                                    pinfo.indentationLeft();
367                                    document.add(pinfo);
368                                    document.add(new Paragraph("\n"));
369                            }
370    
371    
372    
373    
374    
375    
376                    } catch (BadElementException e) {
377                            LOG.error(e);
378                    } catch (MalformedURLException e) {
379                            LOG.error(e);
380                    } catch (IOException e) {
381                            LOG.error(e);
382                    } catch (DocumentException e) {
383                            LOG.error(e);
384                    }
385                    // step 6: fermeture du document.
386                    document.close();
387    
388    
389            }

398            /**
399             * configure le document pdf.
400             * @param width
401             * @param height
402             * @param margin
403             * @return doc
404             */
405            private Document configureDocument(final float margin) {
406    
407                    Document document = new Document();
408    
409                    document.setPageSize(PageSize.A4.rotate());
410                    float marginPage = (margin / 2.54f) * 72f;
411                    document.setMargins(marginPage, marginPage, marginPage, marginPage);
412    
413                    return document;
414            }
