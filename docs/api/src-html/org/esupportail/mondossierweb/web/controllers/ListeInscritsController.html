249                    while(controllerEnAction){
250                            //on attend que l'action en cours (methode 'enter') soit terminée.
251                            //Si la méthode 'enter' est déjà en cours d'exécution, c'est que l'utilisateur a du cliquer 2 fois de suite sur 
252                            //la page web, appelant deux fois la méthode successivement.
253                            //on empeche les méthodes de s'exécuter en même temps pour ne pas avoir de ConcurrentModificationException lors de setLoginInscrits().
254                            try {
255                                    Thread.sleep(100);

257                                    LOG.error(e);
258                                    LOG.info("erreur pendant thread.sleep enter()");
259                            }
260                    }
261                    controllerEnAction = true;
262    
263                    //Si un pdf est en cours de génération on l'arrete.
264                    while (download) {
265                            stopGeneration=true;
266                            try {
267                                    Thread.sleep(100);

269                                    LOG.error(e);
270                                    LOG.info("erreur pendant thread.sleep2  enter()");
271                            }
272    
273                    }
274                    download = false;
275                    stopGeneration=false;
276    
277                    photosValides = false;
278                    listeInscrits.clear();
279                    listeInscritsTrombi.clear();
280                    recherchecodegroupe = false;
281    
282                    if (type.equals("Etape")) {
283                            //On part d'une Etape pour établir une liste d'étudiant
284                            ObjetRecherche objr = new ObjetRecherche();
285                            objr.setCode(code);
286                            objr.setVersion(version);
287                            libelle = service.getLibelleEtape(objr);
288                            Etape e = new Etape(code, version, annee);
289                            if (!session1 && !session2) {
290                                    listeInscrits = ( ArrayList<Inscrit>) service.getInscritsEtape(e);
291                            } 
292                            if (session1 && !session2) {
293                                    listeInscrits = (ArrayList<Inscrit>) service.getInscritsEtapeJuin(e);     
294                            } 
295                            if (!session1 && session2) {
296                                    listeInscrits = (ArrayList<Inscrit>) service.getInscritsEtapeSep(e);
297                            } 
298                            if (session1 && session2) {
299                                    listeInscrits = (ArrayList<Inscrit>) service.getInscritsEtapeJuinSep(e);
300                            }
301                    } else {
302                            if (type.equals("Groupe")) {
303                                    //On part d'un Groupe pour établir une liste d'étudiant
304                                    ObjetRecherche r = new ObjetRecherche();
305                                    r.setAnneeencours(annee);
306                                    r.setCode(code);
307                                    listeInscrits = (ArrayList<Inscrit>) service.getInscritsGroupe(r);
308                                    recherchecodegroupe=true;
309                            }else{
310                                    //On part d'un ELP pour établir une liste d'étudiant
311                                    libelle = service.getLibelleElementPedagogique(code);
312                                    ElementPedagogique e = new ElementPedagogique(code, annee);
313                                    if (!session1 && !session2 && !etape) {
314                                            listeInscrits = (ArrayList<Inscrit>) service.getInscritsElementPedagogique(e);
315                                    }
316                                    if (session1 && !session2 && !etape) {
317                                            listeInscrits = (ArrayList<Inscrit>) service.getInscritsElementPedagogiqueJuin(e);
318                                    }
319                                    if (!session1 && session2 && !etape) {
320                                            listeInscrits = (ArrayList<Inscrit>) service.getInscritsElementPedagogiqueSep(e);
321                                    }
322                                    if (!session1 && !session2 && etape) {
323                                            listeInscrits = (ArrayList<Inscrit>) service.getInscritsElementPedagogiqueEtape(e);
324                                    }
325                                    if (session1 && session2 && !etape) {
326                                            listeInscrits = (ArrayList<Inscrit>) service.getInscritsElementPedagogiqueJuinSep(e);
327                                    }
328                                    if (session1 && !session2 && etape) {
329                                            listeInscrits = (ArrayList<Inscrit>) service.getInscritsElementPedagogiqueJuinEtape(e);
330                                    }
331                                    if (!session1 && session2 && etape) {
332                                            listeInscrits = (ArrayList<Inscrit>) service.getInscritsElementPedagogiqueSepEtape(e);
333                                    }
334                                    if (session1 && session2 && etape) {
335                                            listeInscrits = (ArrayList<Inscrit>) service.getInscritsElementPedagogiqueJuinSepEtape(e);

337                            }

340                    setLoginInscrits();
341                    setMailInscrits();
342    
343                    //on vérifie que les photo sont récupérées pour savoir si on peut afficher le lien vers le trombinoscope:

435            public String naviguationTrombinoscope() {
436                    FacesContext context = FacesContext.getCurrentInstance();
437                    String action = (String) context.getExternalContext().getRequestParameterMap().get("action");
438                    //on récupère la page en cours au moment du clic pour gérer le cas d'un retour arrière du navigateur
439                    String paramPage = (String) context.getExternalContext().getRequestParameterMap().get("pageEnCours");
440                    
441                    if(paramPage != null && !paramPage.equals("")){
442                            pageVisualiseeTrombinoscope = new Integer(paramPage);
443                    }
444                    
445                    affichageTrombiDecoupe = true;
446    
447                    //Si un pdf est en cours de génération on l'arrete.
448                    while (download) {
449                            stopGeneration=true;
450                            try {
451                                    Thread.sleep(100);
452                            } catch (InterruptedException e) {
453                                    LOG.error(e);
454                                    LOG.info("erreur pendant thread.sleep navigationTrombinoscope(");
455                            }
456                    }
457                    download = false;
458                    stopGeneration=false;
459    
460    
461                    if (action.equals("suivant")) {
462                            //cas ou on fait 'suivant':
463                            for (int i = (nbInscritsPageTrombi * pageVisualiseeTrombinoscope); i < (nbInscritsPageTrombi + (nbInscritsPageTrombi * pageVisualiseeTrombinoscope)); i++) {
464                                    if (i < listeInscrits.size()) {
465                                            Inscrit ins = listeInscrits.get(i);
466                                            ins.setUrlphoto(photo.getUrlPhoto(ins.getCod_ind(), ins.getCod_etu()));
467                                    }
468                            }
469                            pageVisualiseeTrombinoscope++;
470                            //prepare la 1er page:
471                            setListeTrombi(pageVisualiseeTrombinoscope);
472    
473                            affichageTrombiDecoupePremierePage = false;
474                    } else {
475                            //cas ou on fait 'précédent':
476                            pageVisualiseeTrombinoscope--;
477                            for (int i = (nbInscritsPageTrombi * pageVisualiseeTrombinoscope); i < (nbInscritsPageTrombi + (nbInscritsPageTrombi * pageVisualiseeTrombinoscope)); i++) {
478                                    if (i < listeInscrits.size()) {
479                                            Inscrit ins = listeInscrits.get(i);
480                                            ins.setUrlphoto(photo.getUrlPhoto(ins.getCod_ind(), ins.getCod_etu()));
481                                    }
482                            }
483    
484                            //prepare la 1er page:
485                            setListeTrombi(pageVisualiseeTrombinoscope);
486    
487                            affichageTrombiDecoupeDernierePage = false;
488                            if (pageVisualiseeTrombinoscope == 1) {
489                                    affichageTrombiDecoupePremierePage = true;
490                            }
491    
492                    }
493                    nbPageTotalVisualiseeTrombinoscope = (listeInscrits.size() / nbInscritsPageTrombi) + 1;
494                    return View.INSCRITS_TROMBINOSCOPE;
495            }
496            /**
497             * créer la liste de liste de NB_INSCRITS_LIGNE_TROMBI inscrits pour la page placée en parametre 
498             * pour l'affichage du trombinoscope.
499             * 
500             */
501            public void setListeTrombi(final int page) {
502                    listeInscritsTrombi.clear();
503                    int nblignes = (nbInscritsPageTrombi / NB_INSCRITS_LIGNE_TROMBI) + 1;
504                    int compteur = nbInscritsPageTrombi * (page - 1);
505                    for (int i = 0; i < nblignes; i++) {
506                            ArrayList<Inscrit> linscrits = new ArrayList<Inscrit>();
507                            for (int k = 0; k < NB_INSCRITS_LIGNE_TROMBI; k++) {
508                                    if (compteur < listeInscrits.size() && compteur < (nbInscritsPageTrombi * page) ) {
509                                            linscrits.add(listeInscrits.get(compteur));
510                                            compteur++;
511                                    } else {
512                                            if (compteur >= listeInscrits.size()) {
513                                                    //on va afficher la derniere page
514                                                    affichageTrombiDecoupeDernierePage = true;
515                                            }
516                                    }
517                            }
518                            listeInscritsTrombi.add(linscrits);
519                    }
520            }

524             * pour l'affichage du trombinoscope.
525             *
526             */
527            public void setListeTrombi() {
528                    listeInscritsTrombi = new ArrayList<ArrayList<Inscrit>>();
529                    int nblignes = (listeInscrits.size() / NB_INSCRITS_LIGNE_TROMBI) + 1;
530                    int compteur = 0;
531                    for (int i = 0; i < nblignes; i++) {
532                            ArrayList<Inscrit> linscrits = new ArrayList<Inscrit>();
533                            for (int k = 0; k < NB_INSCRITS_LIGNE_TROMBI; k++) {
534                                    if (compteur < listeInscrits.size()) {
535                                            linscrits.add(listeInscrits.get(compteur));
536                                            compteur++;
537                                    }
538                            }
539                            listeInscritsTrombi.add(linscrits);
540                    }
541            }

543            /**
544             * exporte la liste des inscrits en format excel.
545             * @return le fichier excel dans la session en passant par la DownloadServlet
546             */
547            public String exportExcel() {
548                    FacesContext facesContext = FacesContext.getCurrentInstance();
549    
550                    ExternalContext externalContext = facesContext.getExternalContext();
551    
552                    //Si un pdf est en cours de génération on l'arrete.
553                    while (download) {
554                            stopGeneration=true;
555                            try {
556                                    Thread.sleep(100);
557                            } catch (InterruptedException e) {
558                                    LOG.error(e);
559                                    LOG.info("erreur pendant thread.sleep exportExcel(");
560                            }
561    
562                    }
563                    download = false;
564                    stopGeneration=false;
565    
566    
567                    try {
568                            ByteArrayOutputStream baos = new ByteArrayOutputStream(OUTPUTSTREAM_SIZE);
569                            HSSFWorkbook wb = creerExcel();
570                            wb.write(baos);
571    
572                            byte[] bytes = baos.toByteArray();
573    
574                            String typeDownload = "application/force-download";
575    
576    
577                            long id = DownloadUtils.setDownload(bytes, libelle.replace('.', ' ').replace(' ', '_') + ".xls", typeDownload, "attachment");
578    
579                            String url = DownloadUtils.getDownloadUrl(id);
580    
581                            externalContext.redirect(url);
582    
583                    } catch (DownloadException e) {
584                            LOG.error(e);
585                    } catch (IOException e) {
586                            LOG.error(e);
587                    }
588                    facesContext.responseComplete();
589    
590    
591                    return null;
592            }

594            /**
595             * créer le fichier excel à partir de la liste des inscrits.
596             * @return le fichier excel de la liste des inscrits.
597             */
598            @SuppressWarnings("deprecation")
599            public HSSFWorkbook creerExcel() {
600                    //      creation du fichier excel
601                    HSSFWorkbook wb = new HSSFWorkbook();
602                    HSSFSheet sheet = wb.createSheet("page1");
603    
604                    //formatage de la taille des colonne
605                    sheet.setColumnWidth((short) 0, (short) (4000));
606                    sheet.setColumnWidth((short) 1, (short) (6000));

1147                                            for (int i = compteur; i < NB_INSCRITS_LIGNE_TROMBI_PDF; i++) {
1148                                                    PdfPCell cell = new PdfPCell();
1149                                                    cell.cloneNonPositionParameters(border);
1150                                                    table.addCell(cell);
1151                                            }
1152    
1153                                    }
1154    
1155                                    document.add(table);
1156                            }
1157                    } catch (BadElementException e) {
1158                            LOG.error(e);
1159                    } catch (MalformedURLException e) {
1160                            LOG.error(e);
1161                    } catch (IOException e) {
1162                            LOG.error(e);
1163                    } catch (DocumentException e) {
1164                            LOG.error(e);
1165                    }
1166    
1167                    // step 6: fermeture du document.
1168                    document.close();
1169                    //on remet les urls pour affichage dans le navigateur.
1170                    for (ArrayList<Inscrit> li : listeInscritsTrombi) {
1171                            for (Inscrit inscrit : li) {
1172                                    inscrit.setUrlphoto(photo.getUrlPhoto(inscrit.getCod_ind(), inscrit.getCod_etu()));
1173                            }
1174                    }
1175            }
