136                    if (etudiant.getDiplomes().size() < 1 && etudiant.getEtapes().size() < 1) {

192             * @return la page des notes et r�sultats dans la nouvelle vue.
193             */
194            public String changerVue() {
195                    if(etudiant.getElementsPedagogiques()!=null)
196                            etudiant.getElementsPedagogiques().clear();
197    
198                    //on change la vue:
199                    if (vueEtudiant) {
200                            vueEtudiant = false;
201                    } else {
202                            vueEtudiant = true;
203                    }
204                    //si on est sur la page du detail des notes:
205                    if (page.equals("detailEtape")) {
206                            if (vueEtudiant) {
207                                    etudiant.renseigneNotesElpEpr(selectedRow);
208                            } else {
209                                    etudiant.renseigneNotesElpEprVueEnseignant(selectedRow);
210                            }
211                            return View.NOTES_ETAPE;
212                    } 
213    
214                    //on est dans la page des notes aux dipl�mes:
215                    if (vueEtudiant) {
216                            vueEtudiantNotesDiplomes = true;
217                            etudiant.renseigneNotesEtResultats();
218                    } else {
219                            vueEtudiantNotesDiplomes = false;
220                            etudiant.renseigneNotesEtResultatsVueEnseignant();
221                    }
222    
223    
224                    return View.NOTES;
225            }

253             * 
254             * @return le fichier pdf.

256            public String exportPdf() {
257    
258    
259                    FacesContext facesContext = FacesContext.getCurrentInstance();
260    
261                    ExternalContext externalContext = facesContext.getExternalContext();
262    
263    
264    
265                    Document document = configureDocument( MARGE_PDF);
266    
267                    try {
268    
269                            ByteArrayOutputStream baosPDF = new ByteArrayOutputStream(OUTPUTSTREAM_SIZE);
270                            PdfWriter docWriter = PdfWriter.getInstance(document, baosPDF);
271                            docWriter.setStrictImageSequence(true);
272                            
273                            // ajout du filigrane
274                            Config config = (Config)  BeanUtils.getBean("config");
275                            if(config.isInsertionFiligranePdfNotes()){
276                                    docWriter.setPageEvent(new Watermark());
277                            }
278                            String titre = "";
279                            if (page.equals("detailEtape")) {
280                                    creerPdfDetail(document);
281                                    titre = getString("PDF.TITLE.DETAIL") + " " + etudiant.getNom().replace('.', ' ').replace(' ', '_') + ".pdf";
282                            } else {
283                                    creerPdfGlobal(document);
284                                    titre = getString("PDF.TITLE.NOTES") + " " + etudiant.getNom().replace('.', ' ').replace(' ', '_') + ".pdf";
285                            }
286                            docWriter.close();
287                            byte[] bytes = baosPDF.toByteArray();
288    
289                            String typeDownload = "application/force-download";
290    
291                            long id = DownloadUtils.setDownload(bytes, titre, typeDownload, "attachment");
292    
293                            String url = DownloadUtils.getDownloadUrl(id);
294    
295                            externalContext.redirect(url);
296    
297                            bytes = null;
298    
299                    } catch (DownloadException e) {
300                            LOG.error(e);
301                    } catch (DocumentException e) {
302                            LOG.error(e);
303                    } catch (IOException e) {
304                            LOG.error(e);
305                    }
306                    facesContext.responseComplete();
307    
308    
309                    return null;
310    
311            }
