154            /**
155             * @return the current user, or null if guest.
156             */
157            @Override
158            public User getCurrentUser() {
159                    
160                    if (ContextUtils.getRequestAttribute(CURRENT_USER_ATTRIBUTE) == null) {
161                            String currentUserId = "";
162                            if( authenticationService.getAuthInfo() != null){
163                                    currentUserId = authenticationService.getAuthInfo().getId();
164                            }
165                    
166                            if (currentUserId == null || currentUserId.equals("")) {
167                                    return null;
168                            }
169                            User user = getDomainService().getUser(currentUserId);
170                            ContextUtils.setRequestAttribute(CURRENT_USER_ATTRIBUTE, user);
171                    }
172                    return (User) ContextUtils.getRequestAttribute(CURRENT_USER_ATTRIBUTE);
173            }

225            /**
226             * JSF callback.
227             * @return a String.
228             * @throws IOException
229             */
230            public String logout() throws IOException {
231                    if (isPortlet()) {
232                            throw new UnsupportedOperationException("logout() should not be called in portlet mode.");
233                    }
234                    FacesContext facesContext = FacesContext.getCurrentInstance();
235                    ExternalContext externalContext = facesContext.getExternalContext();
236                    HttpServletRequest request = (HttpServletRequest) externalContext.getRequest();
237                    String logoutUrl = externalContext.getInitParameter(LOGOUT_URL_PARAM);
238                    if (logoutUrl == null) {
239                            throw new ConfigException("context parameter '" + LOGOUT_URL_PARAM + "' not found");
240                    }
241                    String returnUrl = request.getRequestURL().toString().replaceFirst("/styleshe.*", "/stylesheets/login.faces");
242    
243                    String forwardUrl = String.format(logoutUrl, StringUtils.utf8UrlEncode(returnUrl));
244                    // note: the session beans will be kept event when invalidating
245                    // the session so they have to be reset (by the exception controller).
246                    // We invalidate the session however for the other attributes.
247                    request.getSession().invalidate();
248                    request.getSession(true);
249                    // calling this method will reset all the beans of the application
250                    exceptionController.restart();
251                    externalContext.redirect(forwardUrl);
252                    facesContext.responseComplete();
253                    return null;
254                    //return "navigationLogin";
255            }

309            /**
310             * Détermine le type de l'utilisateur.
311             */
312            public void typeUser() {
313    
314                    typeuser = "";
315                    etudiant.reset();
316                    User user = getCurrentUser();
317                    if (user == null) {
318                            //utitisateur non loggué        
319                            typeuser = "nonloggue";
320                    } else {
321                            //on test si l'utilisateur fait parti des login à exclure
322                            boolean exclure=false;
323                            if(config.getListeLoginExclus() != null && config.getListeLoginExclus().size()>0){
324                                    for(String login : config.getListeLoginExclus()){
325                                            if (user.getId() != null && login!=null && user.getId().equals(login)){
326                                                    //on exclu l'utilisateur
327                                                    exclure = true;
328                                            }
329                                    }
330                            }
331    
332                            if(!exclure){
333                                    //on test si l'utilisateur est la personne a substituer à un étudiant:
334                                    if (user.getId() != null && user.getId().equals(config.getIdLoginTest()) && config.getIdEtudiantTest() != null && !config.getIdEtudiantTest().equals("")) {
335                                            setIduser(config.getIdEtudiantTest());
336                                    } else {
337                                            setIduser(user.getId());
338                                    }
339                                    typeuser = security.getTypeUser(getIduser());
340    
341                                    if (typeuser.equals(STUDENT_USER)) { 
342                                            etudiant.setCod_etu(service.getCodEtuFromLogin(getIduser()));
343                                            if(etudiant.getCod_etu() == null || etudiant.getCod_etu().equals("error")){
344                                                    //L'utilisateur est 'student' dans le ldap mais on a pas réussi a récupérer ses identifiants.
345                                                    //le message ERROR_LDAP est celui qui convient le mieux
346                                                    typeuser = ERROR_LDAP;
347                                            }
348                                    }
349                            }else{
350                                    typeuser=LOGIN_EXCLU;
351                            }
352                    }
353    
354            }

359            /**
360             * @return la page d'accueil selon le type de l'utilisateur
361             */
362            public String accueil() {
363                    //détermine le type de l'utilisateur stocké dans typeuser
364                    typeUser();
365    
366                    //on loggue les connexions si demandé.
367                    if(config.isLogConnexion()){
368                            if (typeuser.equals(STUDENT_USER)) { 
369                                    LOG.info("etudiant "+ iduser +" connecte depuis "+getRemoteAdresse());
370                            }else{
371                                    if(typeuser.equals(LOGIN_EXCLU)){
372                                            LOG.info("utilisateur "+ iduser +" EXCLU depuis "+getRemoteAdresse());
373                                    }else{
374                                            if(typeuser.equals(TEACHER_USER)){
375                                                    LOG.info("enseignant "+ iduser +" connecte depuis "+getRemoteAdresse());
376                                            }
377                                    }
378                            }
379                    }
380    
381                    if      (!config.isActivation()) {
382                            return View.MAINTENANCE;
383                    } else {
384                            if (typeuser.equals(ERROR_LDAP)) {
385                                    return View.ERROR_LDAP;
386                            }
387                            
388                            if(typeuser.equals(LOGIN_EXCLU)){
389                                    return View.ERROR;
390                            }
391                            
392                            if (typeuser.equals(ERROR_BDD)) {
393                                    return View.ERROR_BDD;
394                            }
395    
396                            if (typeuser.equals(STUDENT_USER)) {
397                                    if      (config.isPartieEtudiant()) {
398                                            statisticController.incNbConnexionEtudiant();
399                                            return View.INDEX_ETU;
400                                    } 
401                                    return View.ERROR;
402                            } else if (typeuser.equals(TEACHER_USER)) {
403                                    if      (config.isPartieEnseignant()) {
404                                            statisticController.incNbConnexionEnseignant();
405                                            return View.INDEX_ENS;
406                                    } 
407                                    return View.ERROR;
408                            } else if (typeuser.equals(UNAUTHORIZED_USER)) {
409                                    return View.ERROR;
410                            } else {
411                                    if (!isPortlet()) {
412                                            return View.LOGIN;
413                                            //return View.ERROR_CONNEXION;
414                                    } 
415                                    return View.ERROR;
416                            }
417                    }
418            }

435            /**
436             * @return l'adresse ip de l'utilisateur
437             */
438            public String getRemoteAdresse() {
439                    String ra = "";
440    
441                    try {
442    
443                            String address = HttpUtils.getClient().toString();
444                            StringTokenizer s = new StringTokenizer(address, "/");
445                            while (s.hasMoreTokens()) {
446                                    ra = s.nextToken();
447                            }
448                    } catch (UnknownHostException e) {
449                            LOG.error(e);
450                    }
451    
452    
453    
454                    return ra;
455            }
