004    package org.esupportail.mondossierweb.web.converters;
005    
006    import org.esupportail.mondossierweb.domain.beans.Config;
007    import org.springframework.orm.ibatis.support.SqlMapClientDaoSupport;
008    
009    /**
010     * la classe qui permet d'obtenir l'e-mail d'un étudiant connaissant son login.
011     * @author Charlie Dubois
012     */
013    public class EmailConverterUnivLorraineImpl extends SqlMapClientDaoSupport implements EmailConverterInterface{
014            /**
015             * la bean qui contient la String de complétion de l'e-mail.
016             */
017            private Config config;
018            
019            
020            /**
021             * le constructeur.
022             */
023            public EmailConverterUnivLorraineImpl() {
024                    super();
025            }
026            
027            /**
028             * @param login
029             * @return l'adresse mail.
030             */
031            public String getMail(final String login) {
032                    //Gestion du cas ou le login est null ou vide
033                    if (login != null && !login.equals("") && config.getExtMail() != null && !config.getExtMail().equals("")) {
034                            String mail="";
035                            //aller chercher le mail dans annu_mel_login
036                            mail =  (String) getSqlMapClientTemplate().queryForObject("Mellogin.getMail", login);
037                            if(mail == null || mail.equals("null") || mail.equals("error"))
038                                    mail = "";
039                            return mail;
040                    }
041                    return "";      
042            }
043            
044            
045            /**
046             * @return config
047             */
048            public Config getConfig() {
049                    return config;
050            }
051            
052            /**
053             * @param config
054             */
055            public void setConfig(final Config config) {
056                    this.config = config;
057            }
058    }
