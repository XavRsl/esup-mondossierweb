007    import org.esupportail.commons.services.logging.Logger;
008    import org.esupportail.commons.services.logging.LoggerImpl;
009    import org.springframework.orm.ibatis.support.SqlMapClientDaoSupport;
010    
011    
012    /**
013     * Classe qui sait comment récupérer le Login Etudiant depuis le cod_etu
014     * @author Charlie Dubois
015     *
016     */
017    public class DaoLoginCodeEtudiantImplBasic extends SqlMapClientDaoSupport implements IDaoLoginCodeEtudiant {
018    
019            /**
020             * Un logger.
021             */
022            private static final Logger LOG = new LoggerImpl(DaoLoginCodeEtudiantImplBasic.class);
023    
024    
025            /**
026             * constructeur vide.
027             */
028            public DaoLoginCodeEtudiantImplBasic() {
029                    super();
030    
031            }
032            /**
033             * @see org.esupportail.mondossierweb.dao.IDaoLoginCodeEtudiant#getLoginFromCodInd(java.lang.String)
034             */
035            public String getLoginFromCodEtu(final String codetu) {
036                    //via ibatis et une requete SQL:
037                    try {
038                            
039                            String login =  (String) getSqlMapClientTemplate().queryForObject("Mellogin.getLogin", codetu);
040    
041                            if(login != null && !login.equals("") && !login.equals("error")){
042                                    return login;
043                            }
044                            
045                    } catch (Exception e) {
046                            LOG.error(e);
047                    }
048                    
049                    LOG.error("Erreur lors de la récupération du login a partir du codetu "+codetu);
050                    
051                    return null;
052    
053            }
054    
055    }
