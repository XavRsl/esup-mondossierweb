145            /**
146             * @see org.esupportail.mondossierweb.dao.IDaoService#getPay(java.lang.String)
147             */
148            public String[] getPay(final String pays) {
149                    PaysDTO[] pdto = monProxyGeo.recupererPays(null, "O");
150                    int i = 0;
151                    while (i < pdto.length) {
152                            if (pdto[i].getLibPay().equals(pays)) {
153                                    String[] pay = new String[2];
154                                    pay[0] = pdto[i].getLibPay();
155                                    pay[1] = pdto[i].getCodePay();
156                                    return pay;
157                            }
158                            i++;
159                    }
160                    return null;
161            }

174             * @see org.esupportail.mondossierweb.dao.IDaoService#getCommune(java.lang.String)
175             */
176            public String[] getCommune(final String codp) {
177                    //return monProxyGeo.recupererCommune(null, null, null);
178                    CommuneDTO[]cdto = monProxyGeo.recupererCommune(codp, "O", "T");
179                    CommuneDTO c = cdto[0];
180                    String[] ville = new String[3];
181                    ville[0] = c.getLibCommune();
182                    ville[1] = c.getCodeCommune();
183                    ville[2] = c.getCodePostal();
184    
185                    return ville;
186    
187            }

189            /**
190             * méthode utile pour les villes ayant le même code postal.
191             */
192            public String[] getCommune(final String codp, final String nom) {
193                    CommuneDTO[]cdto = monProxyGeo.recupererCommune(codp, "O", "T");
194                    String[] ville = new String[3];
195                    for (int i = 0; i < cdto.length; i++) {
196                            CommuneDTO c = cdto[i];
197                            if (c.getLibCommune().equals(nom)){
198                                    ville[0] = c.getLibCommune();
199                                    ville[1] = c.getCodeCommune();
200                                    ville[2] = c.getCodePostal();
201                            }
202                    }
203    
204                    return ville;
205    
206            }

576            public List<ElpDeCollection> recupererGroupes(String annee, String codElp) {
577                    //appel WS Offre de foramtion 'recupererGroupe'
578                    List<ElpDeCollection> listeElp = new LinkedList<ElpDeCollection>();
579    
580                    RecupererGroupeDTO recupererGroupeDTO = monProxyOffreDeFormation.recupererGroupe(annee, null, null, null, codElp, null);
581    
582                    if (recupererGroupeDTO != null){
583    
584                            //On parcourt les ELP
585                            for(ElementPedagogiDTO elp : recupererGroupeDTO.getListElementPedagogi()){
586                                    ElpDeCollection el = new ElpDeCollection(elp.getCodElp(), elp.getLibElp());
587                                    
588                                    List<CollectionDeGroupes> listeCollection = new LinkedList<CollectionDeGroupes>();
589    
590                                    //On parcourt les collections de l'ELP
591                                    for( CollectionDTO2 cd2: elp.getListCollection()){
592                                            CollectionDeGroupes collection = new CollectionDeGroupes(cd2.getCodExtCol());
593    
594                                            List<Groupe> listegroupe = new LinkedList<Groupe>();
595                                            
596                                            //On parcourt les groupes de la collection
597                                            for(GroupeDTO gd2 : cd2.getListGroupe()){
598                                                    //On récupère les infos sur le groupe
599                                                    Groupe groupe = new Groupe(gd2.getCodExtGpe());
600                                                    groupe.setLibGroupe(gd2.getLibGpe());
601                                                    
602                                                    //on récupère le codeGpe a la main car il est pas retourne par le WS.
603                                                    ParamRequeteDTO param = new ParamRequeteDTO();
604                                                    param.setChaine1(groupe.getCodGroupe());
605                                                    param.setChaine2(collection.getCodCollection());
606                                                    param.setChaine3(elp.getCodElp());
607                                                    groupe.setCleGroupe(getCleGroupe(param));
608                                                    
609                                                    
610                                                    if(gd2.getCapaciteGpe() != null){
611                                                            if(gd2.getCapaciteGpe().getCapMaxGpe() != null){
612                                                                    groupe.setCapMaxGpe(gd2.getCapaciteGpe().getCapMaxGpe());
613                                                            }else{
614                                                                    groupe.setCapMaxGpe(0);
615                                                            }
616                                                            if(gd2.getCapaciteGpe().getCapIntGpe()!=null){
617                                                                    groupe.setCapIntGpe(gd2.getCapaciteGpe().getCapIntGpe());
618                                                            }else{
619                                                                    groupe.setCapIntGpe(0);
620                                                            }
621                                                    }else {
622                                                            groupe.setCapMaxGpe(0);
623                                                            groupe.setCapIntGpe(0);
624                                                    }
625                                                    //On ajoute le groupe à la liste de la collection
626                                                    listegroupe.add(groupe);
627                                            }
628                                            //on insere la liste créé dans la collection
629                                            collection.setListeGroupes(listegroupe);
630                                            //On ajoute la collection a la liste
631                                            listeCollection.add(collection);
632                                    }
633                                    //On insere la liste créé dans l'ELP
634                                    el.setListeCollection(listeCollection);
635                                    //On ajoute l'ELP a la liste
636                                    listeElp.add(el);
637                            }
638    
639                    }
640    
641                    return listeElp;
642            }
