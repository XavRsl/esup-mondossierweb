004    package org.esupportail.commons.utils; 
005    
006    import java.util.Arrays;
007    import java.util.Collection;
008    import java.util.List;
009    import java.util.Map;
010    
011    import org.esupportail.commons.exceptions.ConfigException;
012    import org.springframework.util.CollectionUtils;
013    import org.springframework.util.ObjectUtils;
014    import org.springframework.util.StringUtils;
015    
016    
017    /**
018     * An Assert class that throws ConfigException.
019     */
020    public abstract class Assert {
021    
022            /**
023             * Constructor.
024             */
025            private Assert() {
026                    throw new UnsupportedOperationException();
027            }
028    
029            /**
030             * Assert a boolean expression, throwing <code>IllegalArgumentException</code>
031             * if the test result is <code>false</code>.
032             * <pre class="code">Assert.isTrue(i &gt; 0, "The value must be greater than zero");</pre>
033             * @param expression a boolean expression
034             * @param message the exception message to use if the assertion fails
035             * @throws IllegalArgumentException if expression is <code>false</code>
036             */
037            public static void isTrue(final boolean expression, final String message) {
038                    if (!expression) {
039                            throw new ConfigException(message);
040                    }
041            }
042    
043            /**
044             * Assert that an object is <code>null</code> .
045             * <pre class="code">Assert.isNull(value, "The value must be null");</pre>
046             * @param object the object to check
047             * @param message the exception message to use if the assertion fails
048             * @throws IllegalArgumentException if the object is not <code>null</code>
049             */
050            public static void isNull(final Object object, final String message) {
051                    if (object != null) {
052                            throw new ConfigException(message);
053                    }
054            }
055    
056            /**
057             * Assert that an object is not <code>null</code> .
058             * <pre class="code">Assert.notNull(clazz, "The class must not be null");</pre>
059             * @param object the object to check
060             * @param message the exception message to use if the assertion fails
061             * @throws IllegalArgumentException if the object is <code>null</code>
062             */
063            public static void notNull(final Object object, final String message) {
064                    if (object == null) {
065                            throw new ConfigException(message);
066                    }
067            }
068    
069            /**
070             * Assert that a string is not empty; that is, it must not be <code>null</code> and not empty.
071             * <pre class="code">Assert.hasLength(name, "Name must not be empty");</pre>
072             * @param text the string to check
073             * @param message the exception message to use if the assertion fails
074             * @see StringUtils#hasLength
075             */
076            public static void hasLength(final String text, final String message) {
077                    if (!StringUtils.hasLength(text)) {
078                            throw new ConfigException(message);
079                    }
080            }
081    
082            /**
083             * Assert that a string has valid text content; that is, it must not be <code>null</code>
084             * and must contain at least one non-whitespace character.
085             * <pre class="code">Assert.hasText(name, "Name must not be empty");</pre>
086             * @param text the string to check
087             * @param message the exception message to use if the assertion fails
088             * @see StringUtils#hasText
089             */
090            public static void hasText(final String text, final String message) {
091                    if (!StringUtils.hasText(text)) {
092                            throw new ConfigException(message);
093                    }
094            }
095    
096            /**
097             * Assert that the given text does not contain the given substring.
098             * <pre class="code">Assert.doesNotContain(name, "rod", "Name must not contain 'rod'");</pre>
099             * @param textToSearch the text to search
100             * @param substring the substring to find within the text
101             * @param message the exception message to use if the assertion fails
102             */
103            public static void doesNotContain(final String textToSearch, final String substring, final String message) {
104                    if (StringUtils.hasLength(textToSearch) && StringUtils.hasLength(substring) 
105                                    && textToSearch.indexOf(substring) != -1) {
106                            throw new ConfigException(message);
107                    }
108            }
109    
110            /**
111             * Assert that an array has elements; that is, it must not be
112             * <code>null</code> and must have at least one element.
113             * <pre class="code">Assert.notEmpty(array, "The array must have elements");</pre>
114             * @param array the array to check
115             * @param message the exception message to use if the assertion fails
116             * @throws IllegalArgumentException if the object array is <code>null</code> or has no elements
117             */
118            public static void notEmpty(final Object[] array, final String message) {
119                    if (ObjectUtils.isEmpty(array)) {
120                            throw new ConfigException(message);
121                    }
122            }
123    
124            /**
125             * Assert that a collection has elements; that is, it must not be
126             * <code>null</code> and must have at least one element.
127             * <pre class="code">Assert.notEmpty(collection, "Collection must have elements");</pre>
128             * @param collection the collection to check
129             * @param message the exception message to use if the assertion fails
130             */
131            @SuppressWarnings("unchecked")
132            public static void notEmpty(final Collection collection, final String message) {
133                    if (CollectionUtils.isEmpty(collection)) {
134                            throw new ConfigException(message);
135                    }
136            }
137    
138            /**
139             * Assert that a Map has entries; that is, it must not be <code>null</code>
140             * and must have at least one entry.
141             * <pre class="code">Assert.notEmpty(map, "Map must have entries");</pre>
142             * @param map the map to check
143             * @param message the exception message to use if the assertion fails
144             */
145            @SuppressWarnings("unchecked")
146            public static void notEmpty(final Map map, final String message) {
147                    if (CollectionUtils.isEmpty(map)) {
148                            throw new ConfigException(message);
149                    }
150            }
151    
152            /**
153             * Assert that the provided object is an instance of the provided class.
154             * <pre class="code">Assert.instanceOf(Foo.class, foo);</pre>
155             * @param clazz the required class
156             * @param obj the object to check
157             * @param message a message which will be prepended to the message produced by
158             * the function itself, and which may be used to provide context. It should
159             * normally end in a ": " or ". " so that the function generate message looks
160             * ok when prepended to it.
161             * @see Class#isInstance
162             */
163            @SuppressWarnings("unchecked")
164            public static void isInstanceOf(final Class clazz, final Object obj, final String message) {
165                    Assert.notNull(clazz, "The clazz to perform the instanceof assertion cannot be null");
166                    if (obj == null) {
167                            Assert.isTrue(clazz.isInstance(obj), message 
168                                            + "Object of class '" + "[null]" 
169                                            + "' must be an instance of '" + clazz.getName() + "'");
170                    } else {
171                            Assert.isTrue(clazz.isInstance(obj), message 
172                                            + "Object of class '" + obj.getClass().getName() 
173                                            + "' must be an instance of '" + clazz.getName() + "'");
174                    }
175            }
176    
177            /**
178             * Assert that <code>superType.isAssignableFrom(subType)</code> is <code>true</code>.
179             * <pre class="code">Assert.isAssignable(Number.class, myClass);</pre>
180             * @param superType the super type to check
181             * @param subType the sub type to check
182             * @param message a message which will be prepended to the message produced by
183             * the function itself, and which may be used to provide context. It should
184             * normally end in a ": " or ". " so that the function generate message looks
185             * ok when prepended to it.
186             */
187            @SuppressWarnings("unchecked")
188            public static void isAssignable(final Class superType, final Class subType, final String message) {
189                    Assert.notNull(superType, "superType cannot be null");
190                    Assert.notNull(subType, "subType cannot be null");
191                    Assert.isTrue(superType.isAssignableFrom(subType), message + "Type [" + subType.getName()
192                                                    + "] is not assignable to type [" + superType.getName() + "]");
193            }
194    
195    
196            /**
197             * Assert a boolean expression, throwing <code>IllegalStateException</code>
198             * if the test result is <code>false</code>. Call isTrue if you wish to
199             * throw IllegalArgumentException on an assertion failure.
200             * <pre class="code">Assert.state(id == null, "The id property must not already be initialized");</pre>
201             * @param expression a boolean expression
202             * @param message the exception message to use if the assertion fails
203             */
204            public static void state(final boolean expression, final String message) {
205                    if (!expression) {
206                            throw new ConfigException(message);
207                    }
208            }
209    
210            /**
211             * Assert that a list of strings contains a value.
212             * @param values
213             * @param listName 
214             * @param value 
215             */
216            public static void contains(
217                            final List<String> values,
218                            final String listName,
219                            final String value) {
220                    if (values == null) {
221                            throw new ConfigException("list " + listName + " should not be null");
222                    }
223                    if (!values.contains(value)) {
224                            String msg = "allowed values for " + listName + " are ";
225                            String separator = "";
226                            for (String string : values) {
227                                    msg += separator + "'" + string + "'";
228                                    separator = ", ";
229                            }
230                            throw new ConfigException(msg);
231                    }
232            }
233    
234            /**
235             * Assert that an array of strings contains a value.
236             * @param values
237             * @param listName 
238             * @param value 
239             */
240            public static void contains(
241                            final String [] values,
242                            final String listName,
243                            final String value) {
244                    contains(Arrays.asList(values), listName, value);
245            }
246    
247            /**
248             * Assert that a list of strings contains a value.
249             * @param values
250             * @param listName 
251             * @param value 
252             */
253            public static void contains(
254                            final List<Integer> values,
255                            final String listName,
256                            final Integer value) {
257                    if (values == null) {
258                            throw new ConfigException("list " + listName + " should not be null");
259                    }
260                    if (!values.contains(value)) {
261                            String msg = "allowed values for " + listName + " are ";
262                            String separator = "";
263                            for (Integer integer : values) {
264                                    separator = ", ";
265                                    msg += separator + "'" + integer + "'";
266                            }
267                            throw new ConfigException(msg);
268                    }
269            }
270    
271            /**
272             * Assert that an array of strings contains a value.
273             * @param values
274             * @param listName 
275             * @param value 
276             */
277            public static void contains(
278                            final Integer [] values,
279                            final String listName,
280                            final Integer value) {
281                    contains(Arrays.asList(values), listName, value);
282            }
