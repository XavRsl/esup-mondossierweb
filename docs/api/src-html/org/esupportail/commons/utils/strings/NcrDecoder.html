354            /**
355             * Decode a String from HTML to UTF-8.
356             * @param str
357             * @return the decoded string.
358             */
359            public static String decode(final String str) {
360                    if (str == null) {
361                            return null;
362                    }
363                    StringBuffer ostr = new StringBuffer();
364                    int i1 = 0;
365                    int i2 = 0;
366                    while (i2 < str.length()) {
367                            i1 = str.indexOf("&", i2);
368                            if (i1 == -1 ) {
369                                    ostr.append(str.substring(i2, str.length()));
370                                    break;
371                            }
372                            ostr.append(str.substring(i2, i1));
373                            i2 = str.indexOf(";", i1);
374                            if (i2 == -1 ) {
375                                    ostr.append(str.substring(i1, str.length()));
376                                    break;
377                            }
378                            String tok = str.substring(i1 + 1, i2);
379                            if (tok.charAt(0) == '#') {
380                                    tok = tok.substring(1);
381                                    try {
382                                            int radix = TEN;
383                                            if (tok.trim().charAt(0) == 'x') {
384                                                    radix = SIXTEEN;
385                                                    tok = tok.substring(1, tok.length());
386                                            }
387                                            ostr.append((char) Integer.parseInt(tok, radix));
388                                    } catch (NumberFormatException exp) {
389                                            ostr.append('?');
390                                    }
391                            } else {
392                                    String decodedEntity = getEntities().get(tok);
393                                    if (decodedEntity != null) {
394                                            ostr.append(decodedEntity);
395                                    } else {
396                                            ostr.append("???&" + tok + ";???");
397                                    }
398                            }
399                            i2++;
400                    }
401                    return ostr.toString();
402            }
