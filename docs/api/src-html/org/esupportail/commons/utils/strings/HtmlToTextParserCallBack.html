004    package org.esupportail.commons.utils.strings; 
005    
006    import java.io.IOException;
007    import java.io.Reader;
008    import java.io.StringReader;
009    
010    import javax.swing.text.html.HTMLEditorKit;
011    import javax.swing.text.html.parser.ParserDelegator;
012    
013    /**
014     * A class to convert HTML to text using Swing.
015     */
016    public final class HtmlToTextParserCallBack extends HTMLEditorKit.ParserCallback {   
017            
018            /**
019             * A StringBuffer to store the result of the parsing.
020             */
021            private StringBuffer textResult = new StringBuffer();
022    
023            /**
024             * Private constructor.
025             */
026            private HtmlToTextParserCallBack() {
027                    super();
028            }
029    
030            /**
031             * @see javax.swing.text.html.HTMLEditorKit.ParserCallback#handleText(char[], int)
032             */
033            @Override
034            public void handleText(
035                            final char[] data, 
036                            @SuppressWarnings("unused") final int unusedPos) {      
037                    textResult.append(data);
038                    }    
039            
040            /**
041             * Parse HTML and return text.
042             * @param htmlString an HTML string.
043             * @return a String.
044             * @throws IOException
045             */
046            static String convert(final String htmlString) throws IOException {
047                    Reader r = new StringReader(htmlString);      
048                    ParserDelegator parser = new ParserDelegator();      
049                    HtmlToTextParserCallBack callback = new HtmlToTextParserCallBack();
050                    parser.parse(r, callback, false);
051                    return callback.getTextResult().toString();
052            }
053    
054            /**
055             * @return Returns the textResult.
056             */
057            StringBuffer getTextResult() {
058                    return textResult;
059            }
060    }
