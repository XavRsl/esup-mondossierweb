073            /**
074             * @return The request headers.
075             */
076            @SuppressWarnings("unchecked")
077            public static Map<String, List<String>> getRequestHeaders() {
078                    HttpServletRequest request = getHttpServletRequest();
079                    if (request == null) {
080                            return null;
081                    }
082                    Enumeration<String> headerNames = request.getHeaderNames();
083                    if (headerNames == null) {
084                            return null;
085                    }
086                    Map<String, List<String>>  result = new HashMap<String, List<String>>();
087                    while (headerNames.hasMoreElements()) {
088                            String headerName = headerNames.nextElement();
089                            List<String> values = new ArrayList<String>();
090                            result.put(headerName, values);
091                            Enumeration<String> headers = request.getHeaders(headerName); 
092                            while (headers.hasMoreElements()) {
093                                    String header = headers.nextElement(); 
094                                    values.add(header);
095                            }
096                    }
097                    return result;
098            }

124             * @return the cookies as a set of readable strings.
125             */
126            public static Set<String> getCookiesStrings() {
127                    Set<String> sortedStrings = new TreeSet<String>();
128                    Cookie [] cookies = getCookies();
129                    if (cookies != null) {
130                            for (Cookie cookie : cookies) {
131                                    sortedStrings.add(cookie.getName() + " = [" + cookie.getValue() + "]");
132                            }
133                    }
134                    return sortedStrings;
135            }

137            /**
138             * @return The headers of the request, as a set of strings.
139             */
140            @SuppressWarnings("unchecked")
141            public static Set<String> getRequestHeadersStrings() {
142                    Set<String> sortedStrings = new TreeSet<String>();
143                    HttpServletRequest request = getHttpServletRequest();
144                    if (request == null) {
145                            return null;
146                    }
147                    Enumeration<String> names = request.getHeaderNames();
148                    while (names.hasMoreElements()) {
149                            String name = names.nextElement();
150                            Enumeration<String> headers = request.getHeaders(name);
151                            while (headers.hasMoreElements()) {
152                                    String header = headers.nextElement();
153                                    sortedStrings.add(name + ": " + header);
154                            }
155                    }
156                    return sortedStrings;
157            }

176                    Set<String> sortedStrings = new TreeSet<String>();
177                    HttpServletRequest request = getHttpServletRequest();
178                    if (request == null) {
179                            return null;
180                    }
181                    Enumeration<String> names = request.getParameterNames();
182                    while (names.hasMoreElements()) {
183                            String name = names.nextElement();
184                            String [] values = request.getParameterValues(name);
185                            for (String value : values) {
186                                    sortedStrings.add(name + " = [" + value + "]");
187                            }
188                    }
189                    return sortedStrings;
190            }

207            /**
208             * @return The client as a resolved InetAddress.
209             * @throws UnknownHostException 
210             */
211            public static InetAddress getClient() throws UnknownHostException {
212                    String rawClientString = getRawClientString();
213                    if (rawClientString == null) {
214                            throw new UnknownHostException("could not get client address!");
215                    }
216                    if (LOG.isDebugEnabled()) {
217                            LOG.debug("resolving '" + rawClientString + "'");
218                    }
219                    return getRealInetAddress(rawClientString);
220            }

222            /**
223             * @return The client as a printable string.
224             */
225            public static String getClientString() {
226                    String rawClientString = getRawClientString(); 
227                    if (rawClientString == null) {
228                            return null;
229                    }
230                    try {
231                            InetAddress client = getRealInetAddress(rawClientString);
232                            return new StringBuffer(client.getHostAddress())
233                            .append(" (").append(client.getHostName()).append(")").toString();
234                    } catch (UnknownHostException e) {
235                            LOG.error("UnknownHostException caught: " + e);
236                            return rawClientString;
237                    }
238            }

270            /**
271             * @param name 
272             * @return the real IP address, not localhost.
273             * @throws UnknownHostException 
274             */
275            @SuppressWarnings("unchecked")
276            private static InetAddress getRealInetAddress(final String name) 
277            throws UnknownHostException {
278                    if (!name.equals("localhost") && !name.equals("127.0.0.1") && !name.equals("0:0:0:0:0:0:0:1")) {
279                            InetAddress host = InetAddress.getByName(name);
280                            host.getAddress();
281                            return host;
282                    }
283                    try {
284                            Enumeration interfaces = NetworkInterface.getNetworkInterfaces();
285                            if (interfaces == null) {
286                                    InetAddress host = InetAddress.getByName(name);
287                                    return host;
288                            }       
289                            while (interfaces.hasMoreElements()) {
290                                    NetworkInterface card = (NetworkInterface) interfaces.nextElement();
291                                    Enumeration addresses = card.getInetAddresses();
292                                    if (addresses == null) {
293                                            continue;
294                                    }
295                                    while (addresses.hasMoreElements()) {
296                                            InetAddress address = (InetAddress) addresses.nextElement();
297                                            address.getAddress();
298                                            String addressName = address.getHostName();
299                                            if (!addressName.equals("localhost") 
300                                                            && !addressName.equals("127.0.0.1") 
301                                                            && !addressName.equals("0:0:0:0:0:0:0:1")) {
302                                                    return address;
303                                            }
304                                    }
305                            }
306                    } catch (SocketException e) {
307                            // fall back to default
308                    }
309                    return InetAddress.getByName(name);
310            }

402            /**
403             * @return The current HttpServletResponse, null if not found.
404             */
405            public static HttpServletResponse getHttpServletResponse() {
406                    FacesContext facesContext = FacesContext.getCurrentInstance();
407                    if (facesContext == null) {
408                            if (LOG.isDebugEnabled()) {
409                                    LOG.debug("FacesContext.getCurrentInstance() returns null, " 
410                                                    + "can not get the current HTTP servlet response.");
411                            }
412                            return null;
413                    }
414                    ExternalContext externalContext = facesContext.getExternalContext();
415                    if (externalContext == null) {
416                            if (LOG.isDebugEnabled()) {
417                                    LOG.debug("facesContext.getExternalContext() returns null, " 
418                                                    + "can not get the current HTTP servlet response.");
419                            }
420                            return  null;
421                    }
422                    Object responseObject = externalContext.getResponse();
423                    if (responseObject == null) {
424                            if (LOG.isDebugEnabled()) {
425                                    LOG.debug("externalContext.getRequest() returns null, " 
426                                                    + "can not get the current HTTP servlet response.");
427                            }
428                            return null;
429                    }
430                    if (!(responseObject instanceof HttpServletResponse)) {
431                            if (LOG.isDebugEnabled()) {
432                                    LOG.debug("requestObject is not a HttpServletResponse, " 
433                                                    + "can not get the current HTTP servlet response.");
434                            }
435                            return null;
436                    }
437                    return (HttpServletResponse) responseObject;
438            }
