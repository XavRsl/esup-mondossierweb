052            /**
053             * @param requestWrapper
054             * @return The HttpServletRequest instance that corresponds to a ServletRequestWrapper, or null if not possible.
055             */
056            private static HttpServletRequest getHttpServletRequestFromServletRequestWrapper(
057                            final ServletRequestWrapper requestWrapper) {
058                    ServletRequest servletRequest = requestWrapper.getRequest();
059                    if (servletRequest == null) {
060                            if (LOG.isDebugEnabled()) {
061                                    LOG.debug("servletRequest is null");
062                            }
063                            return null;
064                    }
065                    if (LOG.isDebugEnabled()) {
066                            LOG.debug("retrieved a ServletRequest instance from portletRequest");
067                    }
068                    if (!(servletRequest instanceof HttpServletRequest)) {
069                            if (LOG.isDebugEnabled()) {
070                                    LOG.debug("servletRequest ('" + servletRequest.getClass().getName() 
071                                                    + "') is not a HttpServletRequest");
072                            }
073                            return null;
074                    }
075                    if (LOG.isDebugEnabled()) {
076                            LOG.debug("servletRequest ('" + servletRequest.getClass().getName() 
077                                            + "') casted to HttpServletRequest");
078                    }
079                    return (HttpServletRequest) servletRequest;
080            }

087                            final MultipartPortletRequestWrapper requestWrapper) {
088                    ActionRequest actionRequest = MultipartUtils.getActionRequest(requestWrapper);
089                    return getHttpServletRequestFromPortletRequest(actionRequest);

092            /**
093             * @param portletRequest 
094             * @return The HttpServletRequest instance that corresponds to a PortletRequest, or null if not possible.
095             */
096            static HttpServletRequest getHttpServletRequestFromPortletRequest(final Object oPortletRequest) {
097                    final PortletRequest portletRequest = (PortletRequest) oPortletRequest;
098                    if (portletRequest == null) {
099                            if (LOG.isDebugEnabled()) {
100                                    LOG.debug("portletRequest is null");
101                            }
102                            return null;
103                    }
104                    if (portletRequest instanceof ServletRequestWrapper) {
105                            return getHttpServletRequestFromServletRequestWrapper(
106                                            (ServletRequestWrapper) portletRequest);
107                    }
108                    if (portletRequest instanceof MultipartPortletRequestWrapper) {
109                            return getHttpServletRequestFromMultipartPortletRequestWrapper(
110                                            (MultipartPortletRequestWrapper) portletRequest);
111                    }
112                    if (LOG.isDebugEnabled()) {
113                            LOG.debug("portletRequest ('" + portletRequest.getClass().getName() 
114                                            + "') is nor a ServletRequestWrapper neither a MultipartPortletRequestWrapper");
115                    }
116                    
117                    return null;
118            }

120            /**
121             * Bind the portlet request and context to the current thread (to use Sping-scoped beans).
122             * @param request
123             * @param context 
124             * @return the request attributes (that should be released by unbindRequest()).
125             */
126            public static PortletRequestAttributes bindRequestAndContext(
127                            final PortletRequest request,
128                            final PortletContext context) {
129                    LocaleContextHolder.setLocale(request.getLocale());
130                    PortletRequestAttributes requestAttributes = new PortletRequestAttributes(request);
131                    RequestContextHolder.setRequestAttributes(requestAttributes);
132                    BeanUtils.initBeanFactory(context);
133                    return requestAttributes;
134            }

136            /**
137             * Unbind the request from the current thread.
138             * @param requestAttributes
139             */
140            public static void unbindRequest(
141                            final PortletRequestAttributes requestAttributes) {
142                    if (requestAttributes != null) {
143                            requestAttributes.requestCompleted();
144                    }
145                    RequestContextHolder.setRequestAttributes(null);
146                    LocaleContextHolder.setLocale(null);
147            }

178            /**
179             * @return The request attributes.
180             * @throws NoRequestBoundException 
181             */
182            static RequestAttributes getContextAttributes() throws NoRequestBoundException {
183                    RequestAttributes requestAttributes = RequestContextHolder.getRequestAttributes();
184                    if (requestAttributes == null) {
185                            throw new NoRequestBoundException();
186                    }
187                    if (!(requestAttributes instanceof ServletRequestAttributes) 
188                                    && !(requestAttributes instanceof PortletRequestAttributes)) {
189                            throw new IllegalArgumentException(
190                                            "requestAttributes of unknown class [" + requestAttributes.getClass() + "]");
191                    }
192                    return requestAttributes;
193            }

295            /**
296             * @param attributes
297             * @return The request attributes, as a set of strings.
298             */
299            private static Set<String> getAttributesStrings(
300                            final Map<String, Object> attributes) {
301                    Set<String> sortedStrings = new TreeSet<String>();
302                    Set<String> keys = attributes.keySet();
303                    for (String key : keys) {
304                            Object obj = attributes.get(key);
305                            String objToString = null;
306                            if (obj != null) {
307                                    try {
308                                            if (obj instanceof CallResultMap) {
309                                                    objToString = 
310                                                            obj.getClass() + "#" + obj.hashCode() 
311                                                            + "[size=" + ((CallResultMap) obj).size() + "]";
312                                            } else {
313                                                    objToString = obj.toString();
314                                            }
315                                    } catch (Throwable t) {
316                                            LOG.error(t);
317                                    }
318                            }
319                            if (objToString != null) {
320                                    sortedStrings.add(key + " = [" + objToString + "]");
321                            } else {
322                                    sortedStrings.add(key + " = null");
323                            }
324                    }
325                    return sortedStrings;
326            }

328            /**
329             * @param request
330             * @return All the attributes of the current request.
331             */
332            @SuppressWarnings("unchecked")
333            private static Map<String, Object> getRequestAttributes(final HttpServletRequest request) {
334                    Map<String, Object> attributes = new Hashtable<String, Object>();
335                    if (request != null) {
336                            Enumeration<String> names = request.getAttributeNames();
337                            while (names.hasMoreElements()) {
338                                    String name = names.nextElement();
339                                    Object obj = request.getAttribute(name);
340                                    if (obj != null) {
341                                            attributes.put(name, obj);
342                                    }
343                            }
344                    } else {
345                            LOG.warn("no request, can not get request attributes");
346                    }
347                    return attributes;
348            }
