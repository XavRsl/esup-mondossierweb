006    /**
007     * <p>Encodes and decodes to and from Base64 notation.</p>
008     * <p>Homepage: <a href="http://iharder.net/base64">http://iharder.net/base64</a>.</p>
009     *
010     * <p>The <tt>options</tt> parameter, which appears in a few places, is used to pass 
011     * several pieces of information to the encoder. In the "higher level" methods such as 
012     * encodeBytes( bytes, options ) the options parameter can be used to indicate such 
013     * things as first gzipping the bytes before encoding them, not inserting linefeeds 
014     * (though that breaks strict Base64 compatibility), and encoding using the URL-safe 
015     * and Ordered dialects.</p>
016     *
017     * <p>The constants defined in Base64 can be OR-ed together to combine options, so you 
018     * might make a call like this:</p>
019     *
020     * <code>String encoded = Base64.encodeBytes( mybytes, Base64.GZIP | Base64.DONT_BREAK_LINES );</code>
021     *
022     * <p>to compress the data before encoding it and then making the output have no newline characters.</p>
023     *
024     *
025     * <p>
026     * Change Log:
027     * </p>
028     * <ul>
029     *  <li>v2.2.2 - Fixed encodeFileToFile and decodeFileToFile to use the
030     *   Base64.InputStream class to encode and decode on the fly which uses
031     *   less memory than encoding/decoding an entire file into memory before writing.</li>
032     *  <li>v2.2.1 - Fixed bug using URL_SAFE and ORDERED encodings. Fixed bug
033     *   when using very small files (~< 40 bytes).</li>

040     *   URL and file name friendly format as described in Section 4 of RFC3548.

350            /**
351             * Returns one of the _SOMETHING_ALPHABET byte arrays depending on
352             * the options specified.
353             * It's possible, though silly, to specify ORDERED and URLSAFE
354             * in which case one of them will be picked, though there is
355             * no guarantee as to which one will be picked.
356             */
357            private final static byte[] getAlphabet( int options )
358            {
359                    if( (options & URL_SAFE) == URL_SAFE ) return _URL_SAFE_ALPHABET;
360                    else if( (options & ORDERED) == ORDERED ) return _ORDERED_ALPHABET;
361                    else return _STANDARD_ALPHABET;
362                    
363            }       // end getAlphabet
364

367             * Returns one of the _SOMETHING_DECODABET byte arrays depending on
368             * the options specified.
369             * It's possible, though silly, to specify ORDERED and URL_SAFE
370             * in which case one of them will be picked, though there is
371             * no guarantee as to which one will be picked.
372             */
373            private final static byte[] getDecodabet( int options )
374            {
375                    if( (options & URL_SAFE) == URL_SAFE ) return _URL_SAFE_DECODABET;
376                    else if( (options & ORDERED) == ORDERED ) return _ORDERED_DECODABET;
377                    else return _STANDARD_DECODABET;

387        /**
388         * Encodes or decodes two files from the command line;
389         * <strong>feel free to delete this method (in fact you probably should)
390         * if you're embedding this code into a larger program.</strong>
391         */
392        public final static void main( String[] args )
393        {
394            if( args.length < 3 ){
395                usage("Not enough arguments.");
396            }   // end if: args.length < 3
397            else {
398                String flag = args[0];
399                String infile = args[1];
400                String outfile = args[2];
401                if( flag.equals( "-e" ) ){
402                    Base64.encodeFileToFile( infile, outfile );
403                }   // end if: encode
404                else if( flag.equals( "-d" ) ) {
405                    Base64.decodeFileToFile( infile, outfile );
406                }   // end else if: decode    
407                else {
408                    usage( "Unknown flag: " + flag );
409                }   // end else    
410            }   // end else
411        }   // end main

450        /**
451         * <p>Encodes up to three bytes of the array <var>source</var>
452         * and writes the resulting four Base64 bytes to <var>destination</var>.
453         * The source and destination arrays can be manipulated
454         * anywhere along their length by specifying 
455         * <var>srcOffset</var> and <var>destOffset</var>.
456         * This method does not check to make sure your arrays
457         * are large enough to accomodate <var>srcOffset</var> + 3 for
458         * the <var>source</var> array or <var>destOffset</var> + 4 for
459         * the <var>destination</var> array.
460         * The actual number of significant bytes in your array is
461         * given by <var>numSigBytes</var>.</p>
462             * <p>This is the lowest level of the encoding methods with
463             * all possible parameters.</p>
464         *
465         * @param source the array to convert
466         * @param srcOffset the index where conversion begins
467         * @param numSigBytes the number of significant bytes in your array
468         * @param destination the array to hold the conversion
469         * @param destOffset the index where output will be put
470         * @return the <var>destination</var> array
471         * @since 1.3
472         */
473        private static byte[] encode3to4( 
474         byte[] source, int srcOffset, int numSigBytes,
475         byte[] destination, int destOffset, int options )
476        {
477                    byte[] ALPHABET = getAlphabet( options ); 
478            
479            //           1         2         3  
480            // 01234567890123456789012345678901 Bit position
481            // --------000000001111111122222222 Array position from threeBytes
482            // --------|    ||    ||    ||    | Six bit groups to index ALPHABET
483            //          >>18  >>12  >> 6  >> 0  Right shift necessary
484            //                0x3f  0x3f  0x3f  Additional AND
485            
486            // Create buffer with zero-padding if there are only one or two
487            // significant bytes passed in the array.
488            // We have to shift left 24 in order to flush out the 1's that appear
489            // when Java treats a value as negative that is cast from a byte to an int.
490            int inBuff =   ( numSigBytes > 0 ? ((source[ srcOffset     ] << 24) >>>  8) : 0 )
491                         | ( numSigBytes > 1 ? ((source[ srcOffset + 1 ] << 24) >>> 16) : 0 )
492                         | ( numSigBytes > 2 ? ((source[ srcOffset + 2 ] << 24) >>> 24) : 0 );
493    
494            switch( numSigBytes )
495            {
496                case 3:
497                    destination[ destOffset     ] = ALPHABET[ (inBuff >>> 18)        ];
498                    destination[ destOffset + 1 ] = ALPHABET[ (inBuff >>> 12) & 0x3f ];
499                    destination[ destOffset + 2 ] = ALPHABET[ (inBuff >>>  6) & 0x3f ];
500                    destination[ destOffset + 3 ] = ALPHABET[ (inBuff       ) & 0x3f ];
501                    return destination;
502                    
503                case 2:
504                    destination[ destOffset     ] = ALPHABET[ (inBuff >>> 18)        ];
505                    destination[ destOffset + 1 ] = ALPHABET[ (inBuff >>> 12) & 0x3f ];
506                    destination[ destOffset + 2 ] = ALPHABET[ (inBuff >>>  6) & 0x3f ];
507                    destination[ destOffset + 3 ] = EQUALS_SIGN;
508                    return destination;
509                    
510                case 1:
511                    destination[ destOffset     ] = ALPHABET[ (inBuff >>> 18)        ];
512                    destination[ destOffset + 1 ] = ALPHABET[ (inBuff >>> 12) & 0x3f ];
513                    destination[ destOffset + 2 ] = EQUALS_SIGN;
514                    destination[ destOffset + 3 ] = EQUALS_SIGN;
515                    return destination;
516                    
517                default:
518                    return destination;
519            }   // end switch
520        }   // end encode3to4

700        {
701            // Isolate options
702            int dontBreakLines = ( options & DONT_BREAK_LINES );
703            int gzip           = ( options & GZIP   );
704            
705            // Compress?
706            if( gzip == GZIP )
707            {
708                java.io.ByteArrayOutputStream  baos  = null;
709                java.util.zip.GZIPOutputStream gzos  = null;
710                Base64.OutputStream            b64os = null;
711                
712        
713                try
714                {
715                    // GZip -> Base64 -> ByteArray
716                    baos = new java.io.ByteArrayOutputStream();
717                    b64os = new Base64.OutputStream( baos, ENCODE | options );
718                    gzos  = new java.util.zip.GZIPOutputStream( b64os ); 
719                
720                    gzos.write( source, off, len );
721                    gzos.close();
722                }   // end try
723                catch( java.io.IOException e )
724                {
725                    e.printStackTrace();
726                    return null;
727                }   // end catch
728                finally
729                {
730                    try{ gzos.close();  } catch( Exception e ){}
731                    try{ b64os.close(); } catch( Exception e ){}
732                    try{ baos.close();  } catch( Exception e ){}
733                }   // end finally
734    
735                // Return value according to relevant encoding.
736                try
737                {
738                    return new String( baos.toByteArray(), PREFERRED_ENCODING );
739                }   // end try
740                catch (java.io.UnsupportedEncodingException uue)
741                {
742                    return new String( baos.toByteArray() );
743                }   // end catch
744            }   // end if: compress
745            
746            // Else, don't compress. Better not to use streams at all then.
747            else
748            {
749                // Convert option to boolean in way that code likes it.
750                boolean breakLines = dontBreakLines == 0;
751                
752                int    len43   = len * 4 / 3;
753                byte[] outBuff = new byte[   ( len43 )                      // Main 4:3

777                }   // end if: some padding needed
778    
779                
780                // Return value according to relevant encoding.
781                try

784                }   // end try
785                catch (java.io.UnsupportedEncodingException uue)
786                {
787                    return new String( outBuff, 0, e );
788                }   // end catch
789                
790            }   // end else: don't compress

1409            /**

1411             * to/from Base64 and returns the next byte.
1412             *
1413             * @return next byte
1414             * @since 1.3
1415             */
1416            public int read() throws java.io.IOException 
1417            { 
1418                // Do we need to get data?
1419                if( position < 0 )
1420                {
1421                    if( encode )
1422                    {
1423                        byte[] b3 = new byte[3];
1424                        int numBinaryBytes = 0;
1425                        for( int i = 0; i < 3; i++ )
1426                        {
1427                            try
1428                            { 
1429                                int b = in.read();
1430                                
1431                                // If end of stream, b is -1.
1432                                if( b >= 0 )
1433                                {
1434                                    b3[i] = (byte)b;
1435                                    numBinaryBytes++;
1436                                }   // end if: not end of stream
1437                                
1438                            }   // end try: read
1439                            catch( java.io.IOException e )
1440                            {   
1441                                // Only a problem if we got no data at all.
1442                                if( i == 0 )
1443                                    throw e;
1444                                
1445                            }   // end catch
1446                        }   // end for: each needed input byte
1447                        
1448                        if( numBinaryBytes > 0 )
1449                        {
1450                            encode3to4( b3, 0, numBinaryBytes, buffer, 0, options );
1451                            position = 0;
1452                            numSigBytes = 4;
1453                        }   // end if: got data
1454                        else
1455                        {
1456                            return -1;
1457                        }   // end else
1458                    }   // end if: encoding
1459                    
1460                    // Else decoding
1461                    else
1462                    {
1463                        byte[] b4 = new byte[4];
1464                        int i = 0;
1465                        for( i = 0; i < 4; i++ )
1466                        {
1467                            // Read four "meaningful" bytes:
1468                            int b = 0;
1469                            do{ b = in.read(); }
1470                            while( b >= 0 && decodabet[ b & 0x7f ] <= WHITE_SPACE_ENC );
1471                            
1472                            if( b < 0 )
1473                                break; // Reads a -1 if end of stream
1474                            
1475                            b4[i] = (byte)b;
1476                        }   // end for: each needed input byte
1477                        
1478                        if( i == 4 )
1479                        {
1480                            numSigBytes = decode4to3( b4, 0, buffer, 0, options );
1481                            position = 0;
1482                        }   // end if: got four characters
1483                        else if( i == 0 ){
1484                            return -1;
1485                        }   // end else if: also padded correctly
1486                        else
1487                        {
1488                            // Must have broken out from above.
1489                            throw new java.io.IOException( "Improperly padded Base64 input." );
1490                        }   // end 
1491                        
1492                    }   // end else: decode
1493                }   // end else: get data
1494                
1495                // Got data?
1496                if( position >= 0 )
1497                {
1498                    // End of relevant data?
1499                    if( /*!encode &&*/ position >= numSigBytes )
1500                        return -1;
1501                    
1502                    if( encode && breakLines && lineLength >= MAX_LINE_LENGTH )
1503                    {
1504                        lineLength = 0;
1505                        return '\n';
1506                    }   // end if
1507                    else
1508                    {
1509                        lineLength++;   // This isn't important when decoding
1510                                        // but throwing an extra "if" seems
1511                                        // just as wasteful.
1512                        
1513                        int b = buffer[ position++ ];
1514    
1515                        if( position >= bufferLength )
1516                            position = -1;
1517    
1518                        return b & 0xFF; // This is how you "cast" a byte that's
1519                                         // intended to be unsigned.
1520                    }   // end else
1521                }   // end if: position >= 0
1522                
1523                // Else error
1524                else
1525                {   
1526                    // When JDK1.4 is more accepted, use an assertion here.
1527                    throw new java.io.IOException( "Error in Base64 code reading stream." );
1528                }   // end else
1529            }   // end read
