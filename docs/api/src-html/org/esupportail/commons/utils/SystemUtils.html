004    package org.esupportail.commons.utils; 
005    
006    import java.net.InetAddress;
007    import java.net.UnknownHostException;
008    import java.util.Enumeration;
009    import java.util.Properties;
010    import java.util.Set;
011    import java.util.TreeSet;
012    
013    import org.esupportail.commons.services.logging.Logger;
014    import org.esupportail.commons.services.logging.LoggerImpl;
015    
016    /**
017     * A class that provides facilities with HTTP requests.
018     */
019    public class SystemUtils {
020    
021            /**
022             * A logger.
023             */
024            private static final Logger LOG = new LoggerImpl(SystemUtils.class);
025            
026            /**
027             * Magic number.
028             */
029            private static final long MB = 1024 * 1024;
030    
031            /**
032             * Private constructor.
033             */
034            private SystemUtils() {
035                    throw new UnsupportedOperationException();
036            }
037    
038            /**
039             * @return The system properties, as a set of strings.
040             */
041            @SuppressWarnings("unchecked")
042            public static Set<String> getSystemPropertiesStrings() {
043                    Properties properties = System.getProperties();
044                    Set<String> sortedPropertiesStrings = new TreeSet<String>();   
045                    Enumeration keys = properties.keys();
046                    while (keys.hasMoreElements()) {
047                            String key = (String) keys.nextElement();
048                            String value; 
049                            if (key.contains("password")) {
050                                    value = "******";
051                            } else {
052                                    value = properties.getProperty(key);
053                            }
054                            sortedPropertiesStrings.add(key + " = [" + value + "]");
055                    }
056                    return sortedPropertiesStrings;
057            }
058    
059            /**
060             * @return the name of the server currently running.
061             */
062            public static String getServer() {
063                    try {
064                            return InetAddress.getLocalHost().getHostName();
065                    } catch (UnknownHostException e) {
066                            LOG.error("could not get the address of the server");
067                    }
068                    return null;
069            }
070            
071            /**
072             * @return the free memory (Mb).
073             */
074            public static long getFreeMemory() {
075                    return Runtime.getRuntime().freeMemory() / MB;
076            }
077    
078            /**
079             * @return the total memory (Mb).
080             */
081            public static long getTotalMemory() {
082                    return Runtime.getRuntime().totalMemory() / MB;
083            }
084    
085            /**
086             * @return the max memory (Mb).
087             */
088            public static long getMaxMemory() {
089                    return Runtime.getRuntime().maxMemory() / MB;
090            }
091    
092    }
