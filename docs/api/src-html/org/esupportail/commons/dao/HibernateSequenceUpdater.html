005    package org.esupportail.commons.dao;
006    
007    import org.hibernate.HibernateException;
008    import org.hibernate.Session;
009    import org.springframework.orm.hibernate3.HibernateCallback;
010    
011    /**
012     * This class is used to update a Hibernate sequence.
013     */
014    public class HibernateSequenceUpdater implements HibernateCallback {
015    
016            /** The name of the sequence. */
017            private String sequenceName;
018    
019            /** The value to set. */
020            private Long sequenceId;
021    
022            /**
023             * Constructor.
024             * @param sequenceName
025             * @param sequenceId
026             */
027            public HibernateSequenceUpdater(final String sequenceName, final Long sequenceId) {
028                    this.sequenceId = sequenceId;
029                    this.sequenceName = sequenceName;
030            }
031    
032            /**
033             * Hibernate callback.
034             * @param session
035             * @return null.
036             * @throws HibernateException
037             */
038            public Object doInHibernate(final Session session) throws HibernateException {
039                    if (sequenceId != null) {
040                            session.createSQLQuery(
041                        "SELECT pg_catalog.setval('" + sequenceName + "', " + sequenceId + ", true)"
042                        ).list();
043                    }
044                    return null;
045            }
046    }
