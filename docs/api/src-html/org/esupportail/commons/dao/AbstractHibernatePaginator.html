004    package org.esupportail.commons.dao; 
005    
006    import java.util.List;
007    
008    import org.esupportail.commons.utils.Assert;
009    import org.esupportail.commons.web.beans.AbstractPaginator;
010    import org.springframework.beans.factory.InitializingBean;
011    
012    /** 
013     * A Hibernate paginator, i.e. that has a DaoService attribute to perform HQL queries.
014     * @param <E> the class of the results
015     */ 
016    @SuppressWarnings("serial")
017    public abstract class AbstractHibernatePaginator<E> extends AbstractPaginator<E> implements InitializingBean {
018            
019            /**
020             * The daoService.
021             */
022            private HibernateDaoService daoService;
023            
024            /** 
025             * Constructor.
026             */ 
027            public AbstractHibernatePaginator() { 
028                    super();
029            } 
030            
031            /** 
032             * Constructor.
033             * @param daoService 
034             * @param pageSizeValues 
035             * @param pageSize the number of results to display on the page 
036             * @deprecated use setPageSizeValues() and setDefaultPageSize() 
037             */ 
038            @Deprecated
039            public AbstractHibernatePaginator(
040                            final HibernateDaoService daoService,
041                            final List<Integer> pageSizeValues,
042                            final int pageSize) { 
043                    super(pageSizeValues, pageSize);
044                    this.daoService = daoService;
045            } 
046            
047            /**
048             * @see org.springframework.beans.factory.InitializingBean#afterPropertiesSet()
049             */
050            public void afterPropertiesSet() {
051                    Assert.notNull(this.daoService, 
052                                    "property daoService of class " + this.getClass().getName() + " can not be null");
053            }
054    
055            /**
056             * @return the daoService
057             */
058            protected HibernateDaoService getDaoService() {
059                    return daoService;
060            }
061    
062            /**
063             * @param daoService the daoService to set
064             */
065            public void setDaoService(final HibernateDaoService daoService) {
066                    this.daoService = daoService;
067            }
068    
069    }
