004    package org.esupportail.commons.web.servlet;
005    
006    import javax.servlet.ServletConfig;
007    import javax.servlet.ServletException;
008    import javax.servlet.ServletOutputStream;
009    import javax.servlet.ServletRequest;
010    import javax.servlet.ServletResponse;
011    import javax.servlet.http.HttpServlet;
012    import javax.servlet.http.HttpServletResponse;
013    
014    import org.esupportail.commons.services.application.ApplicationService;
015    import org.esupportail.commons.services.application.Version;
016    import org.esupportail.commons.services.application.VersionException;
017    import org.esupportail.commons.services.exceptionHandling.ExceptionUtils;
018    import org.esupportail.commons.utils.BeanUtils;
019    
020    /**
021     * The servlet to show the version number.
022     */
023    public class VersionServlet extends HttpServlet {
024    
025            /**
026             * The id for serialization.
027             */
028            private static final long serialVersionUID = -4582010858813700546L;
029            
030            /**
031             * The result.
032             */
033            private String result;
034            
035            /**
036             * Constructor.
037             */
038            public VersionServlet() {
039                    super();
040            }
041            
042            /**
043             * @see javax.servlet.GenericServlet#init(javax.servlet.ServletConfig)
044             */
045            @Override
046            public void init(final ServletConfig config) throws ServletException {
047                    super.init(config);
048                    try {
049                            ApplicationService applicationService = 
050                                    (ApplicationService) BeanUtils.getBean("applicationService");
051                            Version version = applicationService.getVersion();
052                            result = version.toString();
053                    } catch (Throwable t) {
054                            result = ExceptionUtils.getShortPrintableStackTrace(t);
055                    }
056            }
057    
058            /**
059             * @throws ServletException 
060             * @see javax.servlet.http.HttpServlet#service(javax.servlet.ServletRequest, javax.servlet.ServletResponse)
061             */
062            @Override
063            public void service(
064                            @SuppressWarnings("unused")
065                            final ServletRequest servletRequest, 
066                            final ServletResponse servletResponse) 
067            throws ServletException {
068                    try {
069                            HttpServletResponse response = (HttpServletResponse) servletResponse;
070                            ServletOutputStream out = response.getOutputStream();
071                            out.write(result.getBytes());
072                    } catch (Throwable t) {
073                            Exception ve = new VersionException(t);
074                            ExceptionUtils.catchException(ve);
075                            throw new ServletException(ve);
076                    }
077            }
078    
079    }
