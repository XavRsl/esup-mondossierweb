153        /**
154         * @return the lyfeCycle id.
155         */
156        private String getLifecycleId() {
157            String lifecycleId = getServletConfig().getInitParameter(javax.faces.webapp.FacesServlet.LIFECYCLE_ID_ATTR);
158            if (lifecycleId == null) {
159                    lifecycleId = LifecycleFactory.DEFAULT_LIFECYCLE;
160            }
161            return lifecycleId;
162        }

164        /**
165         * @see javax.servlet.Servlet#init(javax.servlet.ServletConfig)
166         */
167        @Override
168            public void init(final ServletConfig servletConfig) throws ServletException {
169            try {
170                    super.init(servletConfig);
171                    facesContextFactory = 
172                            (FacesContextFactory) FactoryFinder.getFactory(FactoryFinder.FACES_CONTEXT_FACTORY);
173                    LifecycleFactory lifecycleFactory = 
174                            (LifecycleFactory) FactoryFinder.getFactory(FactoryFinder.LIFECYCLE_FACTORY);
175                    lifecycle = lifecycleFactory.getLifecycle(getLifecycleId());
176                    defaultView = servletConfig.getInitParameter(DEFAULT_VIEW_PARAM);
177                    if (!StringUtils.hasText(defaultView)) {
178                            defaultView = DEFAULT_DEFAULT_VIEW;
179                            logger.info("property " + DEFAULT_VIEW_PARAM
180                                            + " is not set, using default value [" + defaultView + "]");
181                    }
182                    if (servletConfig.getInitParameter(USE_REDIRECTOR_PARAM) != null) {
183                            useRedirector = Boolean.valueOf(servletConfig.getInitParameter(USE_REDIRECTOR_PARAM));
184                    }
185                    if (useRedirector) {
186                            redirectorName = servletConfig.getInitParameter(REDIRECTOR_NAME_PARAM);
187                            if (!StringUtils.hasText(redirectorName)) {
188                                    redirectorName = DEFAULT_REDIRECTOR_NAME;
189                                    logger.info("property " + REDIRECTOR_NAME_PARAM 
190                                                    + " is not set, using default value [" + redirectorName + "]");
191                            }
192                    }
193            } catch (Throwable t) {
194                    ExceptionUtils.catchException(t);
195                    if (t instanceof ServletException) {
196                            throw (ServletException) t;
197                    }
198                    if (t.getMessage() != null) {
199                            throw new ServletException(t.getMessage(), t);
200                    }
201                    throw new ServletException(t);
202            }
203        }

216            /**
217             * Wrap exceptions thrown when already catching an exception.
218             * @param t 
219             * @throws ServletException 
220             * @throws IOException 
221             */
222            private void handleExceptionHandlingException(
223                            final Throwable t) 
224            throws ServletException, IOException {
225                    logger.error(
226                                    "An exception was thrown while already catching a previous exception:", 
227                                    t); 
228            if (t instanceof IOException) {
229                throw (IOException) t;
230            } else if (t instanceof ServletException) {
231                throw (ServletException) t;
232            } else if (t.getMessage() != null) {
233                throw new ServletException(t.getMessage(), t);
234            } else {
235                throw new ServletException(t);
236            }
237            }
238    
239            /**
240             * Catch an exception.
241             * @param throwable
242             * @param request 
243             * @param response 
244             * @throws ServletException 
245             * @throws IOException 
246             */
247            protected void catchException(
248                            final Throwable throwable, 
249                            final HttpServletRequest request, 
250                            final HttpServletResponse response) throws ServletException, IOException {
251                    ExceptionUtils.markExceptionCaught(); 
252                    ExceptionService exceptionService = null;
253                    try {
254                            exceptionService = ExceptionUtils.catchException(throwable);
255                    } catch (Throwable t) {
256                            handleExceptionHandlingException(t);
257                            // never reached, prevent from warnings
258                            return;
259                    }
260                    ExceptionUtils.markExceptionCaught(exceptionService); 
261                    try {
262                            String exceptionUrl = request.getContextPath() 
263                            + exceptionService.getExceptionView().replace(".jsp", ".faces");
264                            response.sendRedirect(exceptionUrl);
265                            logger.info("redirected the browser to [" + exceptionUrl + "]");
266                            return;
267                    } catch (Throwable t) {
268                            handleExceptionHandlingException(t);
269                    }
270            }

272            /**
273             * @param facesContext 
274             * @return the redirector
275             */
276            private DeepLinkingRedirector getRedirector(
277                            final FacesContext facesContext) {
278            VariableResolver vr = facesContext.getApplication().getVariableResolver();
279            DeepLinkingRedirector deepLinkingRedirector = 
280                    (DeepLinkingRedirector) vr.resolveVariable(facesContext, redirectorName);
281            if (deepLinkingRedirector == null) {
282                    throw new ConfigException("bean [" + redirectorName + "] not found!");
283            }
284            return deepLinkingRedirector;
285            }

287            /**
288             * @param request 
289             * @return the parameters of the current request.
290             */
291            private Map<String, String> getParams(
292                            final HttpServletRequest request) {
293                    if (request == null) {
294                            return null;
295                    }
296                    Map<String, String>  params = AbstractUrlGenerator.decodeArgToParams(
297                                    request.getParameter(ServletUrlGeneratorImpl.ARGS_PARAM));
298                    return params;
299            }

301        /**
302         * @see javax.servlet.http.HttpServlet#service(
303         * javax.servlet.http.HttpServletRequest, javax.servlet.http.HttpServletResponse)
304         */
305        @Override
306            public void service(final HttpServletRequest request,
307                            final HttpServletResponse response) 
308        throws IOException, ServletException {
309    
310            long startTime = System.currentTimeMillis();
311            FacesContext facesContext = null;
312            ServletRequestAttributes previousRequestAttributes = null;
313            boolean openDatabases = true;
314            try {
315                    previousRequestAttributes = ContextUtils.bindRequestAndContext(request, getServletContext());
316                String pathInfo = request.getPathInfo();
317    
318                // if it is a prefix mapping ...
319                if (pathInfo != null && (pathInfo.startsWith("/WEB-INF") || pathInfo .startsWith("/META-INF"))) {
320                    StringBuffer buffer = new StringBuffer();
321                    buffer.append(" Someone is trying to access a secure resource : " + pathInfo);
322                    buffer.append("\n remote address is " + request.getRemoteAddr());
323                    buffer.append("\n remote host is " + request.getRemoteHost());
324                    buffer.append("\n remote user is " + request.getRemoteUser());
325                    buffer.append("\n request URI is " + request.getRequestURI());
326                    logger.warn(buffer.toString());
327                    response.sendError(HttpServletResponse.SC_NOT_FOUND);
328                    return;
329                }
330                boolean beanError = false;
331                boolean exceptionAlreadyCaught = ExceptionUtils.exceptionAlreadyCaught();
332                if (exceptionAlreadyCaught) {
333                    ExceptionService exceptionService = ExceptionUtils.getMarkedExceptionService();
334                    if (exceptionService != null) {
335                            Throwable throwable = exceptionService.getThrowable();
336                            if (throwable != null) {
337                                    if (ExceptionUtils.hasCause(
338                                                    throwable, 
339                                                    new Class [] {
340                                                                    DatabaseException.class, 
341                                                                    VersionException.class, 
342                                                                    BeansException.class, })) { 
343                                            openDatabases = false;
344                                    }
345                                    if (ExceptionUtils.hasCause(
346                                                    throwable, 
347                                                    BeansException.class)) { 
348                                            beanError = true;
349                                    }
350                            }
351                    }
352                }
353                if (openDatabases) {
354                        DatabaseUtils.open();
355                        DatabaseUtils.begin();
356                    checkVersion();
357                            }
358                facesContext = facesContextFactory.getFacesContext(
359                            getServletConfig().getServletContext(), request, response, lifecycle);
360                boolean firstCall;
361                            DeepLinkingRedirector redirector = null;
362                if (beanError) {
363                    firstCall = false;
364                } else if (useRedirector)  {
365                                    redirector = getRedirector(facesContext);
366                                    firstCall = redirector.firstCall();
367                } else {
368                    firstCall = false;
369                }
370                    Map<String, String> params = getParams(request);
371                    if (!exceptionAlreadyCaught && (params != null || firstCall)) {
372                            String view = null;
373                            if (redirector != null) {
374                                    view = redirector.redirect(params);
375                            }
376                            if (view == null) {
377                                    view = defaultView;
378                            }
379                    Application application = facesContext.getApplication();
380                    ViewHandler viewHandler = application.getViewHandler();
381                    UIViewRoot viewRoot = viewHandler.createView(facesContext, view);
382                    viewRoot.setViewId(view);
383                    facesContext.setViewRoot(viewRoot);
384                    } else {
385                            lifecycle.execute(facesContext);
386                    }
387                            if (openDatabases) {
388                                    DatabaseUtils.commit();
389                            }
390                            lifecycle.render(facesContext);
391                    } catch (Throwable t) {
392                            catchException(t, request, response);
393            } finally {
394                    if (openDatabases) {
395                        try {
396                                            DatabaseUtils.close();
397                                    } catch (DatabaseException e) {
398                                            ExceptionUtils.catchException(e);
399                                    }
400                    }
401                    MonitorUtils.log(startTime, "");
402                    ContextUtils.unbindRequest(previousRequestAttributes);
403                    if (facesContext != null) {
404                            facesContext.release();
405                    }
406            }
407        }
