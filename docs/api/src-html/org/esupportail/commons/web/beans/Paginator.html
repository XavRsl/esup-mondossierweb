004    package org.esupportail.commons.web.beans; 
005    
006    import java.io.Serializable;
007    import java.sql.Timestamp;
008    import java.util.List;
009    
010    import javax.faces.model.SelectItem;
011    
012    /** 
013     * The interface of paginators, used to display displaying results 
014     * from a large result set over a number of pages without retrieving 
015     * all the items from the database. 
016     * 
017     * Adapted from http://blog.hibernate.org/cgi-bin/blosxom.cgi/2004/08/14#fn.html and
018     * http://blog.hibernate.org/cgi-bin/pollxn.cgi?storypath=/Gavin%20King/pagination.html
019     * @param <E> the class of the results
020     */ 
021    public interface Paginator<E> extends Serializable { 
022            
023            /**
024             * @return the page size.
025             */
026            int getPageSize();
027    
028            /**
029             * @return the current page number.
030             */
031            int getCurrentPage();
032            
033            /**
034             * Force the paginator to reload.
035             */
036            void forceReload();
037    
038            /**
039             * Go to the next page.
040             */
041            void gotoNextPage();
042    
043            /**
044             * Go to the previous page.
045             */
046            void gotoPreviousPage();
047    
048            /**
049             * Go to the first page.
050             */
051            void gotoFirstPage();
052    
053            /**
054             * Go to the last page.
055             */
056            void gotoLastPage();
057    
058            /**
059             * @return true if the page is the first one.
060             */
061            boolean isFirstPage();
062            
063            /**
064             * @return true if the page is the last one.
065             */
066            boolean isLastPage();
067            
068            /**
069             * @return the number of the previous page.
070             */
071            int getPreviousPage();
072    
073            /**
074             * @return the number of the next page.
075             */
076            int getNextPage();
077    
078            /**
079             * @return the number of the first page (always 0).
080             */
081            int getFirstPageNumber();
082    
083            /**
084             * @return the number of the last page.
085             */
086            int getLastPageNumber();
087    
088            /**
089             * @return the number of the first result.
090             */
091            int getFirstVisibleNumber();
092            
093            /**
094             * @return the number of the last result.
095             */
096            int getLastVisibleNumber();
097            
098            /**
099             * @return the list of the results.
100             */
101            List<E> getVisibleItems();
102            
103            /**
104             * @return the number of visible items.
105             */
106            int getVisibleItemsCount();
107            
108            /**
109             * @return the total number of items.
110             */
111            int getTotalItemsCount();
112            
113            /**
114             * @return The numbers of the near pages.
115             */
116            List<Integer> getNearPages();
117            
118            /**
119             * @return the time the data was loaded.
120             */
121            Timestamp getLoadTime();
122            
123            /**
124             * Set the page size.
125             * @param pageSize
126             */
127            void setPageSize(int pageSize);
128    
129            /**
130             * Set the current page.
131             * @param currentPage 
132             */
133            void setCurrentPage(final int currentPage);
134    
135            /**
136             * @return the default page size.
137             */
138            int getDefaultPageSize();
139            
140            /**
141             * @return a list of the first pages. (up to MAX_NEAR_PAGES )
142             */
143            List<Integer> getFirstPagesNumber();
144            
145            
146            /**
147             * @return a list of the last pages. (down to MAX_NEAR_PAGES )
148             */
149            List<Integer> getLastPagesNumber();
150            
151            /**
152             * @return a list with the currentPage +/- MAX_NEAR_PAGES.
153             */
154            List<Integer> getMiddlePagesNumber();
155            
156            
157            /**
158             * @return the pageSizeItems.
159             */
160            List<SelectItem> getPageSizeItems();
161    
162    }
