004    package org.esupportail.commons.web.beans; 
005    
006    import org.apache.myfaces.custom.tree2.TreeNode;
007    
008    /** 
009     * The model of a tree able to set the node identifiers.
010     */ 
011    public class TreeModelBase extends org.apache.myfaces.custom.tree2.TreeModelBase {
012            
013            /**
014             * The serialization id.
015             */
016            private static final long serialVersionUID = 6839340913523565161L;
017    
018            /**
019             * The root node.
020             */
021            private TreeNode rootNode;
022    
023            /**
024             * Bean constructor.
025             * @param root
026             */
027            public TreeModelBase(final TreeNode root) {
028                    super(root);
029                    rootNode = root;
030                    setNodeIdentifiers();
031            }
032            
033            /**
034             * Set the identifiers of the nodes of a hierarchy.
035             * @param node
036             * @param id
037             */
038            private void setChildrenIdentifiers(final TreeNode node, final String id) {
039                    for (int i = 0; i < node.getChildCount(); i++) {
040                            TreeNode child = (TreeNode) (node.getChildren().get(i));
041                            String childId = id + SEPARATOR + i;
042                            child.setIdentifier(childId);
043                            setChildrenIdentifiers(child, childId);
044                    }
045            }
046            
047            /**
048             * Set the identifiers of all the nodes of the tree.
049             */
050            protected void setNodeIdentifiers() {
051                    rootNode.setIdentifier("0");
052                    setChildrenIdentifiers(rootNode, "0");
053            }
054            
055    }
