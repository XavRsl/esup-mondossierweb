004    package org.esupportail.commons.web.beans;
005    
006    import java.util.HashSet;
007    import java.util.Set;
008    
009    import org.apache.myfaces.custom.tree2.TreeStateBase;
010    
011    /**
012     * 
013     * @author Benjamin
014     *
015     */
016    public class TransientTreeStateBase extends TreeStateBase {
017            
018            /**
019             * The serialization id.
020             */
021            private static final long serialVersionUID = -8930136920743543157L;
022    
023            /**
024             * A set to store the expanded nodes.
025             */
026        private Set<String> expandedNodes = new HashSet<String>();
027    
028        /**
029         * The id of the active node.
030         */
031        private String activeNodeId;
032    
033        /**
034             * Bean constructor.
035             */
036            public TransientTreeStateBase() {
037                    super();
038            }
039    
040            /**
041             * @see org.apache.myfaces.custom.tree2.TreeStateBase#isNodeExpanded(java.lang.String)
042             */
043            @Override
044            public boolean isNodeExpanded(final String nodeId) {
045            return expandedNodes.contains(nodeId);
046            }
047    
048            /**
049             * @see org.apache.myfaces.custom.tree2.TreeStateBase#toggleExpanded(java.lang.String)
050             */
051            @Override
052            public void toggleExpanded(final String nodeId) {
053            activeNodeId = nodeId;
054            if (expandedNodes.contains(nodeId)) {
055                expandedNodes.remove(nodeId);
056            } else {
057                    Set<String> newExpandedNodes = new HashSet<String>();
058                    for (String id : expandedNodes) {
059                                    if (nodeId.startsWith(id) || id.startsWith(nodeId)) {
060                                            newExpandedNodes.add(id);
061                                    }
062                            }
063                newExpandedNodes.add(nodeId);
064                expandedNodes = newExpandedNodes;
065            }
066            }
067    
068            /**
069             * @see org.apache.myfaces.custom.tree2.TreeStateBase#isTransient()
070             */
071            @Override
072            public boolean isTransient() {
073                    return true;
074            }
075    
076            /**
077             * @see org.apache.myfaces.custom.tree2.TreeStateBase#setTransient(boolean)
078             */
079            @Override
080            public void setTransient(
081                            @SuppressWarnings("unused")
082                            final boolean trans) {
083                    // must always be transient
084            }
085    
086            /**
087             * @see org.apache.myfaces.custom.tree2.TreeStateBase#expandPath(java.lang.String[])
088             */
089            @Override
090            public void expandPath(final String[] nodePath) {
091            for (int i = 0; i < nodePath.length; i++) {
092                String nodeId = nodePath[i];
093                expandedNodes.add(nodeId);
094            }
095            activeNodeId = nodePath[nodePath.length - 1];
096            }
097    
098            /**
099             * @see org.apache.myfaces.custom.tree2.TreeStateBase#collapsePath(java.lang.String[])
100             */
101            @Override
102            public void collapsePath(final String[] nodePath) {
103            for (int i = 0; i < nodePath.length; i++) {
104                String nodeId = nodePath[i];
105                expandedNodes.remove(nodeId);
106            }
107            activeNodeId = "0";
108            }
109    
110            /**
111             * @return the activeNodeId
112             */
113            public String getActiveNodeId() {
114                    return activeNodeId;
115            }
116    
117    }
