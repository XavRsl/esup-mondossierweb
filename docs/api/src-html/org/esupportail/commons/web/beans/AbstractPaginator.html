019    /** 
020     * An abstract paginator, independant from the underlying database implementation.
021     * @param <E> the class of the visibleItems
022     */ 
023    @SuppressWarnings("serial")
024    public abstract class AbstractPaginator<E> implements Paginator<E>, Resettable {
025            
026            /**
027         * The default values for pageSizeItems.
028         */
029        protected static final Integer [] DEFAULT_PAGE_SIZE_ITEM_VALUES = {5, 10, 15, 20};
030        
031            /**
032         * The default value of the default page size.
033         */
034        protected static final int DEFAULT_DEFAULT_PAGE_SIZE = 10;
035        
036            /**
037             * The maximum number of near pages to print.
038             */
039            protected static final int MAX_NEAR_PAGES = 12;
040    
041            /**
042             * The current page number, normally set by constructor.
043             */
044            private int currentPage;
045            
046            /**
047             * The page size, i.e. the number of visibleItems to display per page, 
048             * normally set by the constructor.
049             */
050            private int pageSize;
051            
052            /**
053             * The visible items, normally set by the constructor.
054             */
055            private List<E> visibleItems;
056            
057            /**
058             * The total number of items, normally set by the constructor.
059             */
060            private int totalItemsCount;
061            
062        /**
063         * The page size items.
064         */
065        private List<SelectItem> pageSizeItems;
066        
067        /**
068         * The default page size.
069         */
070        private int defaultPageSize;
071        
072            /**
073             * The time when the data was loaded.
074             */
075            private Timestamp loadTime;
076            
077            /**
078             * The maximum number of near pages to print.
079             */
080            private int maxNearPages;
081            
082            /**
083             * True when loading the items.
084             */
085            private boolean loadRunning;
086            
087            /**
088             * Constructor.
089             */
090            protected AbstractPaginator() {
091                    super();
092                    setPageSizeValues(Arrays.asList(getDefaultPageSizeItemValues()));
093                    setDefaultPageSize(getDefaultDefaultPageSize());
094                    reset();
095            }
096            
097            /**
098             * Constructor.
099             * @param pageSizeValues 
100             * @param pageSize 
101             * @deprecated use setPageSizeValues() and setDefaultPageSize() 
102             */
103            @Deprecated
104            protected AbstractPaginator(
105                            final List<Integer> pageSizeValues,
106                            final int pageSize) {
107                    this();
108                    if (pageSizeValues != null && !pageSizeValues.isEmpty()) {
109                            setPageSizeValues(pageSizeValues);
110                    }
111                    if (pageSize > 0) {
112                            setDefaultPageSize(pageSize);
113                            setPageSize(pageSize);
114                    }
115            }
116            
117            /**
118             * @see org.esupportail.commons.web.controllers.Resettable#reset()
119             */
120            public void reset() {
121                    pageSize = defaultPageSize;
122                    loadRunning = false;
123                    currentPage = 0;
124                    visibleItems = null;
125                    totalItemsCount = 0;
126                    loadTime = null;
127            }
128    
129            /**
130             * @return the loaded marker name.
131             */
132            protected String getRequestLoadedMarker() {
133                    return getClass().getName() + "." + hashCode() + ".loaded";
134            }
135    
136            /**
137             * @return true if the paginator has already been loaded for this request.
138             */
139            protected boolean isLoadedForThisRequest() {
140                    Boolean marker = (Boolean) ContextUtils.getRequestAttribute(getRequestLoadedMarker()); 
141                    return marker != null && marker;
142            }
143    
144            /**
145             * Set the paginator as loaded for this request.
146             */
147            protected void setLoadedForThisRequest() {
148                    ContextUtils.setRequestAttribute(getRequestLoadedMarker(), Boolean.TRUE);
149            }
150    
151            /**
152             * @see org.esupportail.commons.web.beans.Paginator#forceReload()
153             */
154            public void forceReload() {
155                    ContextUtils.setRequestAttribute(getRequestLoadedMarker(), Boolean.FALSE);
156            }
157    
158            /**
159             * Load the items.
160             */
161            protected void loadItems() {
162                    if (!loadRunning) {
163                            loadRunning = true;
164                            if (!isLoadedForThisRequest()) {
165                                    loadItemsInternal();
166                                    setLoadedForThisRequest();
167                            }
168                            loadRunning = false;
169                    }
170            }
171    
172            /**
173             * Load the items (internal).
174             */
175            protected abstract void loadItemsInternal();
176    
177            /**
178             * Set the page size values.
179             * @param pageSizeValues 
180             */
181            public void setPageSizeValues(
182                            final List<Integer> pageSizeValues) {
183                    pageSizeItems = new ArrayList<SelectItem>();
184                    for (Integer pageSizeValue : pageSizeValues) {
185                            pageSizeItems.add(new SelectItem(pageSizeValue));
186                    }
187            }
188            
189            /**
190             * @return the default page size values.
191             */
192            protected Integer [] getDefaultPageSizeItemValues() {
193                    return DEFAULT_PAGE_SIZE_ITEM_VALUES;
194            }
195    
196            /**
197             * @return the default default page size.
198             */
199            protected int getDefaultDefaultPageSize() {
200                    return DEFAULT_DEFAULT_PAGE_SIZE;
201            }
202    
203        /**
204             * @see org.esupportail.commons.web.beans.Paginator#gotoFirstPage()
205             */
206            public void gotoFirstPage() {
207                    setCurrentPage(0);
208            }
209    
210            /**
211             * @see org.esupportail.commons.web.beans.Paginator#gotoLastPage()
212             */
213            public void gotoLastPage() {
214                    setCurrentPage(getLastPageNumber());
215            }
216    
217            /**
218             * @see org.esupportail.commons.web.beans.Paginator#gotoNextPage()
219             */
220            public void gotoNextPage() {
221                    setCurrentPage(currentPage + 1);
222            }
223    
224            /**
225             * @see org.esupportail.commons.web.beans.Paginator#gotoPreviousPage()
226             */
227            public void gotoPreviousPage() {
228                    setCurrentPage(currentPage - 1);
229            }
230    
231            /**
232             * @see java.lang.Object#toString()
233             */
234            @Override
235            public String toString() {
236                    return getClass().getSimpleName() + "#" + hashCode() + "[pageSize=[" + pageSize
237                    + "], currentPage=" + currentPage + "]";
238            }
239            
240            /**
241             * @return the page size.
242             */
243            protected int getPageSizeInternal() {
244                    return pageSize;
245            }
246    
247            /**
248             * @see org.esupportail.commons.web.beans.Paginator#getPageSize()
249             */
250            public final int getPageSize() {
251                    return getPageSizeInternal();
252            }
253    
254            /**
255             * @return the current page.
256             */
257            protected int getCurrentPageInternal() {
258                    return currentPage;
259            }
260    
261            /**
262             * @see org.esupportail.commons.web.beans.Paginator#getCurrentPage()
263             */
264            public final int getCurrentPage() {
265                    loadItems();
266                    return getCurrentPageInternal();
267            }
268    
269            /**
270             * @return true if the first page.
271             */
272            protected boolean isFirstPageInternal() {
273                    return currentPage <= 0; 
274            }
275    
276            /**
277             * @see org.esupportail.commons.web.beans.Paginator#isFirstPage()
278             */
279            public final boolean isFirstPage() {
280                    loadItems();
281                    return isFirstPageInternal(); 
282            }
283    
284            /**
285             * @return true if the last page.
286             */
287            protected boolean isLastPageInternal() {
288                    return currentPage >= getLastPageNumberInternal(); 
289            } 
290    
291            /**
292             * @see org.esupportail.commons.web.beans.Paginator#isLastPage()
293             */
294            public final boolean isLastPage() {
295                    loadItems();
296                    return isLastPageInternal(); 
297            } 
298    
299            /**
300             * @return the previous page.
301             */
302            protected int getPreviousPageInternal() {
303                    return currentPage - 1;
304            }
305    
306            /**
307             * @see org.esupportail.commons.web.beans.Paginator#getPreviousPage()
308             */
309            public final int getPreviousPage() {
310                    loadItems();
311                    return getPreviousPageInternal();
312            }
313    
314            /**
315             * @return the next page.
316             */
317            public int getNextPageInternal() {
318                    return currentPage + 1;
319            }
320    
321            /**
322             * @see org.esupportail.commons.web.beans.Paginator#getNextPage()
323             */
324            public final int getNextPage() {
325                    loadItems();
326                    return getNextPageInternal();
327            }
328    
329            /**
330             * @return the first page number.
331             */
332            protected int getFirstPageNumberInternal() {
333                    return 0; 
334            }
335    
336            /**
337             * @see org.esupportail.commons.web.beans.Paginator#getFirstPageNumber()
338             */
339            public final int getFirstPageNumber() {
340                    loadItems();
341                    return getFirstPageNumberInternal(); 
342            }
343    
344            /**
345             * @return the last page number.
346             */
347            protected int getLastPageNumberInternal() {
348                    return (getTotalItemsCountInternal() - 1) / getPageSize(); 
349            }
350    
351            /**
352             * @see org.esupportail.commons.web.beans.Paginator#getLastPageNumber()
353             */
354            public final int getLastPageNumber() {
355                    loadItems();
356                    return getLastPageNumberInternal(); 
357            }
358    
359            /**
360             * @return the first visible item number.
361             */
362            protected int getFirstVisibleNumberInternal() {
363                    return currentPage * getPageSize(); 
364            }
365    
366            /**
367             * @see org.esupportail.commons.web.beans.Paginator#getFirstVisibleNumber()
368             */
369            public final int getFirstVisibleNumber() {
370                    loadItems();
371                    return getFirstVisibleNumberInternal(); 
372            }
373    
374            /**
375             * @return the last visible item number.
376             */
377            protected int getLastVisibleNumberInternal() {
378                    return Math.min(getFirstVisibleNumberInternal() + getPageSize() - 1, getTotalItemsCountInternal() - 1);
379            }
380    
381            /**
382             * @see org.esupportail.commons.web.beans.Paginator#getLastVisibleNumber()
383             */
384            public final int getLastVisibleNumber() {
385                    loadItems();
386                    return getLastVisibleNumberInternal();
387            }
388    
389            /**
390             * Set the current page (internal, do not force reload).
391             * @param currentPage 
392             */
393            protected void setCurrentPageInternal(final int currentPage) {
394                    this.currentPage = currentPage;
395            }
396    
397            /**
398             * @see org.esupportail.commons.web.beans.Paginator#setCurrentPage(int)
399             */
400            public final void setCurrentPage(final int currentPage) {
401                    forceReload();
402                    if (currentPage < 0) {
403                            setCurrentPageInternal(0);
404                    } else {
405                            setCurrentPageInternal(currentPage);
406                    }
407            }

409            /**
410             * @param visibleItems the visibleItems to set
411             */
412            protected void setVisibleItems(final List<E> visibleItems) {
413                    this.visibleItems = visibleItems;
414            }

416            /**
417             * @return the visible items.
418             */
419            protected List<E> getVisibleItemsInternal() {
420                    return visibleItems;
421            }
422    
423            /**
424             * @see org.esupportail.commons.web.beans.Paginator#getVisibleItems()
425             */
426            public final List<E> getVisibleItems() {
427                    loadItems();
428                    return getVisibleItemsInternal();
429            }
430    
431            /**
432             * @return the near pages.
433             */
434            public List<Integer> getNearPagesInternal() {
435                    List<Integer> result = new ArrayList<Integer>();
436                    int first = Math.max(getFirstPageNumber(), currentPage - MAX_NEAR_PAGES / 2);
437                    int last = Math.min(getLastPageNumber(), currentPage + MAX_NEAR_PAGES / 2);
438                    for (int i = first; i <= last; i++) {
439                            result.add(i);
440                    }
441                    return result;
442            }

445             * @see org.esupportail.commons.web.beans.Paginator#getNearPages()
446             */
447            public final List<Integer> getNearPages() {
448                    loadItems();
449                    return getNearPagesInternal();
450            }

540                    return DEFAULT_DEFAULT_PAGE_SIZE;
541            }
542            
543            /** 
544             * @return the first pages number.
545             */
546            protected List<Integer> getFirstPagesNumberInternal() {
547                    List<Integer> result = new ArrayList<Integer>();
548                    int intervalleFirst = getFirstPageNumber() + getMaxNearPages();
549                    if (getFirstPageNumber() <= getCurrentPage() && intervalleFirst > getCurrentPage() 
550                                    && intervalleFirst < getLastPageNumber()) {
551                            for (Integer i = 0; i < intervalleFirst; ++i) {
552                                    result.add(i);
553                            }
554                    } 
555                    return result;
556            }
557            
558            /** 
559             * @see org.esupportail.commons.web.beans.Paginator#getFirstPagesNumber()
560             */
561            public final List<Integer> getFirstPagesNumber() {
562                    loadItems();
563                    return getFirstPagesNumberInternal();
564            }
565            
566            /** 
567             * @return the last pages number.
568             */
569            protected List<Integer> getLastPagesNumberInternal() {
570                    List<Integer> result = new ArrayList<Integer>();
571                    int intervalleLast = getLastPageNumber() - getMaxNearPages();
572                    if (intervalleLast < getCurrentPage() && getLastPageNumber() >= getCurrentPage()
573                                    && intervalleLast > getFirstPageNumber()) {
574                            for (Integer i = getLastPageNumber(); i > intervalleLast; --i) {
575                                    result.add(i);
576                            }
577                    }
578                    Collections.sort(result);
579                    return result;
580            }
581            
582            /** 
583             * @see org.esupportail.commons.web.beans.Paginator#getLastPagesNumber()
584             */
585            public final List<Integer> getLastPagesNumber() {
586                    loadItems();
587                    return getLastPagesNumberInternal();
588            }
589            
590            /** 
591             * @return the middle pages number
592             * @see org.esupportail.commons.web.beans.Paginator#getMiddlePagesNumber()
593             */
594            protected List<Integer> getMiddlePagesNumberInternal() {
595                    List<Integer> result = new ArrayList<Integer>();
596                    int intervalleDbt = getCurrentPage() - getMaxNearPages() + 1;
597                    int intervalleFin = getCurrentPage() + getMaxNearPages() - 1;
598                    if (intervalleDbt > getFirstPageNumber() && intervalleFin < getLastPageNumber()) {
599                            for (Integer i = intervalleDbt; i <= intervalleFin; ++i) {
600                                    result.add(i);
601                            }
602                    }
603                    return result;
604            }
605

608             */
609            public final List<Integer> getMiddlePagesNumber() {
610                    loadItems();

612            }
