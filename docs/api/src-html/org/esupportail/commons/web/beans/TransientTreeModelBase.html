004    package org.esupportail.commons.web.beans;
005    
006    import org.apache.myfaces.custom.tree2.TreeNode;
007    
008    /**
009     * 
010     * @author Benjamin
011     *
012     */
013    public class TransientTreeModelBase extends TreeModelBase {
014            
015            /**
016             * The serialization id.
017             */
018            private static final long serialVersionUID = -7342900258702215016L;
019    
020        /**
021         * Bean constructor.
022         * @param root
023         */
024        public TransientTreeModelBase(final TreeNode root) {
025            super(root);
026            setTreeState(new TransientTreeStateBase());
027        }
028        
029        /**
030         * @return the active node of the tree
031         */
032        public TreeNode getActiveNode() {
033            TransientTreeStateBase myTreeState = (TransientTreeStateBase) getTreeState();
034            String nodeId = myTreeState.getActiveNodeId();
035            return getNodeById(nodeId);
036        }
037    
038        /**
039         * Set the active node.
040         * @param nodeId
041         */
042        public void setActiveNode(final String nodeId) {
043            TransientTreeStateBase myTreeState = (TransientTreeStateBase) getTreeState();
044            String[] nodePath = getPathInformation(nodeId);
045            String id = "";
046            for (String string : nodePath) {
047                            id += " " + string;
048                    }
049            myTreeState.expandPath(nodePath);
050        }
051    
052    }
