295            /**
296             * JSF callback.
297             * @return a String.
298             */
299            public String restart() {
300                    Map<String, Object> resettables = BeanUtils.getBeansOfClass(Resettable.class);
301                    for (String name : resettables.keySet()) {
302                            if (logger.isDebugEnabled()) {
303                                    logger.debug("trying to reset bean [" + name + "]...");
304                            }
305                            Object bean = resettables.get(name);
306                            if (bean == null) {
307                                    throw new ConfigException("bean [" + name 
308                                                    + "] is null, " 
309                                                    + "application can not be restarted.");
310                            }
311                            if (!(bean instanceof Resettable)) {
312                                    throw new ConfigException("bean [" + name 
313                                                    + "] does not implement Resettable, " 
314                                                    + "application can not be restarted.");
315                            }
316                            ((Resettable) bean).reset();
317                            if (logger.isDebugEnabled()) {
318                                    logger.debug("bean [" + name + "] was reset.");
319                            }
320                    }
321                    ExceptionUtils.unmarkExceptionCaught();
322                    return "applicationRestarted";
323            }
