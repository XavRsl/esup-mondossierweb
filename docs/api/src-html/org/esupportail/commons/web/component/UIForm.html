004    package org.esupportail.commons.web.component;
005    
006    import java.io.IOException;
007    import java.util.List;
008    
009    import javax.faces.application.Application;
010    import javax.faces.component.UIComponent;
011    import javax.faces.context.FacesContext;
012    
013    import org.apache.myfaces.component.html.ext.HtmlGraphicImage;
014    import org.apache.myfaces.component.html.ext.HtmlOutputText;
015    import org.apache.myfaces.custom.div.Div;
016    import org.apache.myfaces.shared_impl.renderkit.JSFAttr;
017    import org.apache.myfaces.shared_tomahawk.renderkit.RendererUtils;
018    
019    
020    /**
021     * This class represents e:form elements.
022     */
023    public class UIForm extends javax.faces.component.UIForm {
024    
025            /**
026             * The type Component.
027             */
028            public static final String COMPONENT_TYPE = "org.esupportail.commons.component.form";
029            
030            /**
031             * The id of the submit popup.
032             */
033            public static final String SUBMIT_POPUP_ID = "submitPopup";
034            
035            /**
036             * The submitPopupText attribute.
037             */
038            public static final String SUBMIT_POPUP_TEXT = "submitPopupText";
039            
040            /**
041             * The submitPopupImage attribute.
042             */
043            public static final String SUBMIT_POPUP_IMAGE = "submitPopupImage";
044            
045            /**
046             * Constructor.
047             */
048            public UIForm() {
049                    super();
050            }
051    
052            /**
053             * @return true if the form already has a submit popup.
054             */
055            @SuppressWarnings("unchecked")
056            protected boolean hasSubmitPopup() {
057                    for (UIComponent c : (List<UIComponent>) getChildren()) {
058                            if (SUBMIT_POPUP_ID.equals(c.getId())) {
059                                    return true;
060                            }
061                    }
062                    return false;
063            }
064    
065            /**
066             * Add the submitPopup.
067             * @param context
068             */
069            @SuppressWarnings("unchecked")
070            protected void addSubmitPopupIfNeeded(final FacesContext context) {
071                    Application application = context.getApplication();
072                    // add a div
073                    Div div = new Div();
074                    div.setId("submitPopup");
075                    div.setStyleClass("form-submit-popup");
076                    div.setStyle("visibility: hidden");
077                    div.setRendered(true);
078                    getChildren().add(div);
079                    // add the image to the div
080                    HtmlGraphicImage img = new HtmlGraphicImage();
081                    img.setId("submitPopupImage");
082                    img.setValue(getAttributes().get(SUBMIT_POPUP_IMAGE));
083                    img.setStyle("visibility: hidden");
084                    img.setRendered(true);
085                    div.getChildren().add(img);
086                    // add the text to the div
087                    HtmlOutputText text = new HtmlOutputText();
088                    text.setId("submitPopupText");
089                    text.setValueBinding(JSFAttr.VALUE_ATTR,  
090                                    application.createValueBinding(
091                                                    getAttributes().get(SUBMIT_POPUP_TEXT).toString()));
092                    text.setStyle("visibility: hidden");
093                    text.setRendered(true);
094                    div.getChildren().add(text);
095            }
096    
097            /**
098             * @see javax.faces.component.UIComponentBase#encodeBegin(javax.faces.context.FacesContext)
099             */
100            @Override
101            public void encodeBegin(final FacesContext context) throws IOException {
102                    if (!hasSubmitPopup()) {
103                            addSubmitPopupIfNeeded(context);
104                    }
105                    super.encodeBegin(context);
106            }
107    
108            /**
109             * @see javax.faces.component.UIComponentBase#getRendersChildren()
110             */
111            @Override
112            public boolean getRendersChildren() {
113                    return true;
114            }
115    
116            /**
117             * @see javax.faces.component.UIComponentBase#encodeChildren(javax.faces.context.FacesContext)
118             */
119            @Override
120            public void encodeChildren(final FacesContext context) throws IOException {
121                    RendererUtils.renderChildren(context, this);
122            }
123            
124    }
