004    package org.esupportail.commons.web.renderers;
005    
006    import java.io.IOException;
007    
008    import javax.faces.component.UIComponent;
009    import javax.faces.context.FacesContext;
010    import javax.faces.context.ResponseWriter;
011    
012    import org.apache.myfaces.renderkit.html.ext.HtmlButtonRenderer;
013    import org.apache.myfaces.shared_impl.renderkit.html.HTML;
014    import org.esupportail.commons.web.tags.PageTag;
015    import org.esupportail.commons.web.tags.config.TagsConfigurator;
016    
017    /**
018     * ESUP-Portail renderer for the 'menuItem' tag.
019     */
020    public class MenuItemRenderer extends HtmlButtonRenderer {
021            /**
022             * The renderer type.
023             */
024            public static final String RENDERER_TYPE = "org.esupportail.MenuItemRenderer";
025    
026            /**
027             * Constructor.
028             *
029             */
030            public MenuItemRenderer() {
031                    super();
032            }
033    
034            /**
035             * @param facesContext 
036             * @param component 
037             * @return true if the compoent is selected.
038             */
039            private boolean isSelected(final FacesContext facesContext, final UIComponent component) {
040                    String id = component.getClientId(facesContext);
041                    if (id == null) {
042                            return false;
043                    }
044                    String [] idTokens = id.split(":");
045                    if (idTokens.length < 1) {
046                            return false;
047                    }
048                    String shortId = idTokens[idTokens.length - 1];
049                    return shortId.equals(PageTag.getCurrentMenuItem());
050            }
051            
052            /**
053             * @see javax.faces.render.Renderer#encodeBegin(
054             * javax.faces.context.FacesContext, javax.faces.component.UIComponent)
055             */
056            @Override
057            public void encodeBegin(final FacesContext facesContext, final UIComponent component) throws IOException {
058                    TagsConfigurator tagsConfigurator = TagsConfigurator.getInstance();
059                    ResponseWriter writer = facesContext.getResponseWriter();
060                    writer.startElement(HTML.LI_ELEM, component);
061                    String styleClass;
062                    if (isSelected(facesContext, component)) {
063                            styleClass = tagsConfigurator.getMenuActiveItemStyleClass();
064                    } else {
065                            styleClass = tagsConfigurator.getMenuItemStyleClass();
066                    }
067                    writer.writeAttribute(HTML.CLASS_ATTR, styleClass, null);
068                    super.encodeBegin(facesContext, component);
069            }
070    
071            /**
072             * @see org.apache.myfaces.shared_tomahawk.renderkit.html.HtmlButtonRendererBase#encodeEnd(
073             * javax.faces.context.FacesContext, javax.faces.component.UIComponent)
074             */
075            @Override
076            public void encodeEnd(final FacesContext facesContext, final  UIComponent component) throws IOException {
077                    super.encodeEnd(facesContext, component);
078                    ResponseWriter writer = facesContext.getResponseWriter();
079                    writer.endElement(HTML.LI_ELEM);
080            }
081    }
