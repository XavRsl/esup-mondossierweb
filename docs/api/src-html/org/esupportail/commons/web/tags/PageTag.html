054    /**
055     * ESUP-Portail implementation of the page tag.
056     * 
057     * This class is a very few-modified copy of class org.apache.myfaces.taglib.core.ViewTag.java.
058     * It would be much simpler if it could inherit from it, but it is impossible:
059     * - ViewTag.doEndTag() calls responseWriter.endDocument(), which throws an exception since
060     *   custom tags are not allowed to flush.
061     * - One could think of overriding doEndTag(), but it is impossible since we must call
062     *   private method UIComponentBase.release(), and this only possible by calling ViewTag.doEndTag()
063     *   (since super.super.doEndTag() is not possible in Java).
064     * So the only things changed is method doEndTag(), as well as doStartTag().
065     */
066    public class PageTag extends UIComponentBodyTag {

146            /**
147             * @see javax.faces.webapp.UIComponentTag#setProperties(javax.faces.component.UIComponent)
148             */
149            @Override
150            protected void setProperties(final UIComponent component) {
151                    if (!isAuthorized()) {
152                            throw new WebFlowException("Page not allowed");
153                    }
154                    super.setProperties(component);
155                    FacesContext context = FacesContext.getCurrentInstance();
156                    Application application = context.getApplication();
157                    if (localeString != null) {
158                            if (UIComponentTag.isValueReference(localeString)) {
159                                    ValueBinding vb = application.createValueBinding(localeString);
160                                    Object localeValue = vb.getValue(context);
161                                    if (localeValue instanceof Locale) {
162                                            locale = (Locale) localeValue;
163                                    } else if (localeValue instanceof String) {
164                                            locale = LocaleUtils.toLocale((String) localeValue);
165                                    } else {
166                                            if (localeValue != null) {
167                                                    throw new IllegalArgumentException(
168                                                                    "Locale or String class expected. Expression: " + localeString
169                                                                    + ". Class: " + localeValue.getClass().getName());
170                                            }
171                                            locale = getDefaultLocale();
172                                    }
173                            } else {
174                                    locale = LocaleUtils.toLocale(localeString);
175                            }
176                    } else {
177                            locale = getDefaultLocale();
178                    }
179                    if (footer != null) {
180                            if (UIComponentTag.isValueReference(footer)) {
181                                    ValueBinding vb = application.createValueBinding(footer);
182                                    Object footerValue = vb.getValue(context);
183                                    footer = (String) footerValue;
184                            }
185                    }
186                    // set the locale of the view
187                    UIViewRoot view = (UIViewRoot) component;
188                    view.setLocale(locale);
189                    Config.set((ServletRequest) getFacesContext().getExternalContext().getRequest(),
190                                    Config.FMT_LOCALE, locale);
191    
192                    component.setRendered(true);
193    
194                    // set the strings variable
195                    if (!StringUtils.hasText(stringsVar)) {
196                            stringsVar = DEFAULT_STRINGS_VAR;
197                            logger.warn(getClass() + ": no stringsVar set, using default ["
198                                            + stringsVar + "]");
199                    }
200    
201                    downloadId = null;
202                    if (downloadIdString != null) {
203                            if (UIComponentTag.isValueReference(downloadIdString)) {
204                                    ValueBinding vb = application.createValueBinding(downloadIdString);
205                                    Object downloadIdValue = vb.getValue(context);
206                                    if (downloadIdValue != null) {
207                                            if (downloadIdValue instanceof Long) {
208                                                    downloadId = (Long) downloadIdValue;
209                                            } else if (downloadIdValue instanceof String) {
210                                                    downloadId = Long.valueOf(downloadIdValue.toString());
211                                            } else {
212                                                    throw new IllegalArgumentException(
213                                                                    "Long or String class expected. Expression: " + downloadIdString
214                                                                    + ". Class: " + downloadIdValue.getClass().getName());
215                                            }
216                                            locale = getDefaultLocale();
217                                    }
218                            }
219                    }
220            }

245            /**
246             * @return true if the page is authorized.
247             */
248            private boolean isAuthorized() {
249                    Boolean authorized = TagUtils.getBooleanAttributeValue("authorized", authorizedString);
250                    if (authorized == null) {
251                            return true;
252                    }
253                    return authorized.booleanValue();

256            /**
257             * @see javax.faces.webapp.UIComponentTag#doStartTag()
258             */
259            @SuppressWarnings("unchecked")
260            @Override
261            public int doStartTag() throws JspException {
262                    if (logger.isDebugEnabled()) {
263                            logger.debug("entering PageTag.doStartTag");
264                    }
265                    super.doStartTag();
266                    FacesContext facesContext = FacesContext.getCurrentInstance();
267                    TagsConfigurator tagsConfigurator = TagsConfigurator.getInstance();
268                    facesContext.getExternalContext().getRequestMap().put(stringsVar, tagsConfigurator.getStrings(locale));
269                    ResponseWriter responseWriter = facesContext.getResponseWriter();
270                    if (logger.isDebugEnabled()) {
271                            logger.debug("doStartTag().facesContext=" + facesContext);
272                            logger.debug("doStartTag().responseWriter=" + responseWriter.getClass());
273                    }
274                    boolean isPortlet = PortletUtil.isPortletRequest(facesContext);
275                    if (logger.isDebugEnabled()) {
276                            logger.debug("isPortlet=" + isPortlet);
277                    }
278                    try {
279                            responseWriter.startDocument();
280                            UIViewRoot view = (UIViewRoot) getComponentInstance();
281                            if (!isPortlet) {
282                                    responseWriter.write(tagsConfigurator.getDoctype());
283                                    responseWriter.startElement(HTML_TAG, view);
284                                    responseWriter.writeAttribute(HTML.LANG_ATTR, view.getLocale().getLanguage(), null);
285                                    responseWriter.startElement(HEAD_TAG, view);
286                                    responseWriter.startElement(TITLE_TAG, null);
287                                    String title;
288                                    if (tagsConfigurator.getDocumentTitleI18nKey() != null) {
289                                            Application application = facesContext.getApplication();
290                                            String exp = TagUtils.makeELExpression(
291                                                            getStringsVar() + "['"
292                                                                                + tagsConfigurator.getDocumentTitleI18nKey() + "']");
293                                            ValueBinding valueBinding = application.createValueBinding(exp);
294                                            title = (String) valueBinding.getValue(facesContext);
295                                    } else {
296                                            title = tagsConfigurator.getDocumentTitle();
297                                    }
298                                    responseWriter.write(title);
299                                    responseWriter.endElement(TITLE_TAG);
300                                    for (String stylesheet : tagsConfigurator.getStylesheets()) {
301                                            responseWriter.startElement(HTML.LINK_ELEM, null);
302                                            responseWriter.writeAttribute(HTML.REL_ATTR, "stylesheet", null);
303                                            responseWriter.writeAttribute(HTML.TYPE_ATTR, "text/css", null);
304                                            String stylesheetUrl;
305                                            if (stylesheet.startsWith("http://")
306                                                            || stylesheet.startsWith(".")
307                                                            || stylesheet.startsWith("/")) {
308                                                    stylesheetUrl = stylesheet;
309                                            } else {
310                                                    stylesheetUrl = tagsConfigurator.getMediaPath()
311                                                    + "/" + stylesheet;
312                                            }
313                                            responseWriter.writeAttribute(HTML.HREF_ATTR, stylesheetUrl, null);
314                                            responseWriter.endElement(HTML.LINK_ELEM);
315                                    }
316                                    responseWriter.startElement("meta", null);
317                                    responseWriter.writeAttribute("http-equiv", "X-UA-Compatible", null);
318                                    responseWriter.writeAttribute("content", "IE=7", null);
319                                    responseWriter.endElement("meta");
320                            }
321                            if (logger.isDebugEnabled()) {
322                                    logger.debug("writing scripts...");
323                            }
324                            for (String script : tagsConfigurator.getScripts()) {
325                                    if (logger.isDebugEnabled()) {
326                                            logger.debug("script=" + script);
327                                    }
328                                    responseWriter.startElement(HTML.SCRIPT_ELEM, null);
329                                    responseWriter.writeAttribute(HTML.TYPE_ATTR, "text/javascript", null);
330                                    String scriptUrl;
331                                    if (script.startsWith("http://") || script.startsWith(".") || script.startsWith("/")) {
332                                            scriptUrl = script;
333                                    } else {
334                                            scriptUrl = tagsConfigurator.getMediaPath() + "/" + script;
335                                    }
336                                    responseWriter.writeAttribute(HTML.SRC_ATTR, scriptUrl, null);
337                                    responseWriter.endElement(HTML.SCRIPT_ELEM);
338                            }
339                            if (isPortlet) {
340                                    responseWriter.startElement(HTML.DIV_ELEM, view);
341                            } else {
342                                    responseWriter.endElement(HEAD_TAG);
343                                    responseWriter.startElement(BODY_TAG, view);
344                            }
345                            addBodyStyleInformation(tagsConfigurator, responseWriter);
346                            responseWriter.flush();
347                    } catch (IOException e) {
348                            throw new JspException("Error writing startDocument", e);
349                    }
350    
351                    if (logger.isDebugEnabled()) {
352                            logger.debug("leaving PageTag.doStartTag");
353                    }
354                    return BodyTag.EVAL_BODY_BUFFERED;
355            }

376            /**
377             * @see javax.faces.webapp.UIComponentTag#doEndTag()
378             */
379            @Override
380            public int doEndTag() throws JspException {
381                    if (logger.isDebugEnabled()) {
382                            logger.debug("entering PageTag.doEndTag");
383                    }
384                    FacesContext facesContext = FacesContext.getCurrentInstance();
385                    ResponseWriter responseWriter = facesContext.getResponseWriter();
386                    boolean isPortlet = PortletUtil.isPortletRequest(facesContext);
387                    UIComponent uiComponent = getComponentInstance();
388                    if (logger.isDebugEnabled()) {
389                            logger.debug("doStartTag().facesContext=" + facesContext);
390                    }
391    
392                    // this was added
393                    try {
394                            addFooter(facesContext, uiComponent);
395                            if (downloadId != null) {
396                                    responseWriter.startElement(HTML.SCRIPT_ELEM, null);
397                                    responseWriter.writeAttribute(HTML.TYPE_ATTR, "text/javascript", null);
398                                    Application application = facesContext.getApplication();
399                                    String popupBlockerMessage = application.createValueBinding(
400                                                    "#{" + stringsVar + "['POPUP_BLOCKER_MESSAGE']}"
401                                    ).getValue(facesContext).toString();
402                                    responseWriter.write("openDownload('"
403                                                    + DownloadUtils.getDownloadUrl(downloadId) + "', '"
404                                                    + popupBlockerMessage + "');");
405                                    responseWriter.endElement(HTML.SCRIPT_ELEM);
406                            }
407                            if (isPortlet) {
408                                    responseWriter.endElement(HTML.DIV_ELEM);
409                            } else {
410                                    responseWriter.endElement(BODY_TAG);
411                                    responseWriter.endElement(HTML_TAG);
412                            }
413                    } catch (IOException e) {
414                            throw new JspException("Error writing endDocument", e);
415                    }
416    
417                    try {
418                            responseWriter.endDocument();
419                    } catch (IOException e) {
420                            logger.error("Error writing endDocument", e);
421                            throw new JspException(e);
422                    }
423    
424                    if (logger.isDebugEnabled()) {
425                            logger.debug("leaving PageTag.doEndTag");
426                    }
427                    return super.doEndTag();
428            }

430            /**
431             * @param facesContext
432             * @param uiComponent
433             * @throws IOException
434             */
435            protected void addFooter(FacesContext facesContext, UIComponent uiComponent) throws IOException {
436                    if (footer != null) {
437                            FooterRenderer.encodeFooter(facesContext, uiComponent, footer);
438                    } else {
439                            FooterRenderer.encodeFooter(
440                                            facesContext, getComponentInstance(),
441                                            TagsConfigurator.getInstance().getFooterText());
442                    }
443            }

445            /**
446             * @see javax.faces.webapp.UIComponentBodyTag#doAfterBody()
447             */
448            @Override
449            public int doAfterBody() throws JspException {
450                    if (logger.isDebugEnabled()) {
451                            logger.debug("entering PageTag.doAfterBody");
452                    }
453                    try {
454                            BodyContent myBodyContent = getBodyContent();
455                            if (myBodyContent != null) {
456                                    FacesContext facesContext = FacesContext.getCurrentInstance();
457                                    StateManager stateManager = facesContext.getApplication().getStateManager();
458                                    StateManager.SerializedView serializedView
459                                    = stateManager.saveSerializedView(facesContext);
460                                    if (serializedView != null) {
461                                            //until now we have written to a buffer
462                                            ResponseWriter bufferWriter = facesContext.getResponseWriter();
463                                            bufferWriter.flush();
464                                            //now we switch to real output
465                                            ResponseWriter realWriter = bufferWriter.cloneWithWriter(getPreviousOut());
466                                            facesContext.setResponseWriter(realWriter);
467    
468                                            String bodyStr = myBodyContent.getString();
469                                            if (stateManager.isSavingStateInClient(facesContext)) {
470                                                    int formMarker = bodyStr.indexOf(JspViewHandlerImpl.FORM_STATE_MARKER);
471                                                    int urlMarker = bodyStr.indexOf(HtmlLinkRendererBase.URL_STATE_MARKER);
472                                                    int lastMarkerEnd = 0;
473                                                    while ((formMarker != -1) || (urlMarker != -1)) {
474                                                            if ((urlMarker == -1) || ((formMarker != -1) && (formMarker < urlMarker))) {
475                                                                    //replace form_marker
476                                                                    realWriter.write(bodyStr, lastMarkerEnd, formMarker - lastMarkerEnd);
477                                                                    stateManager.writeState(facesContext, serializedView);
478                                                                    lastMarkerEnd = formMarker + JspViewHandlerImpl.FORM_STATE_MARKER_LEN;
479                                                                    formMarker = bodyStr.indexOf(JspViewHandlerImpl.FORM_STATE_MARKER, lastMarkerEnd);
480                                                            } else {
481                                                                    //replace url_marker
482                                                                    realWriter.write(bodyStr, lastMarkerEnd, urlMarker - lastMarkerEnd);
483                                                                    if (stateManager instanceof MyfacesStateManager) {
484                                                                            ((MyfacesStateManager) stateManager).writeStateAsUrlParams(facesContext,
485                                                                                            serializedView);
486                                                                    } else {
487                                                                            logger.error(
488                                                                                            "Current StateManager is no MyfacesStateManager and does not support saving state in url parameters.");
489                                                                    }
490                                                                    lastMarkerEnd = urlMarker + HtmlLinkRendererBase.URL_STATE_MARKER_LEN;
491                                                                    urlMarker = bodyStr.indexOf(HtmlLinkRendererBase.URL_STATE_MARKER, lastMarkerEnd);
492                                                            }
493                                                    }
494                                                    realWriter.write(bodyStr, lastMarkerEnd, bodyStr.length() - lastMarkerEnd);
495                                            } else {
496                                                    realWriter.write(bodyStr);
497                                            }
498                                    } else {
499                                            myBodyContent.writeOut(getPreviousOut());
500                                    }
501                            }
502                    } catch (IOException e) {
503                            throw new JspException("Error writing body content", e);
504                    }
505                    if (logger.isDebugEnabled()) {
506                            logger.debug("leaving PageTag.doAfterBody");
507                    }
508                    return super.doAfterBody();
509            }
