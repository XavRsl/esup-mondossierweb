004    package org.esupportail.commons.web.tags;
005    
006    import javax.faces.component.UIComponent;
007    
008    import org.apache.myfaces.shared_impl.renderkit.JSFAttr;
009    import org.apache.myfaces.taglib.html.ext.HtmlPanelGridTag;
010    import org.esupportail.commons.web.tags.config.TagsConfigurator;
011    
012    /**
013     * ESUP-Portail implementation of the 'panelGrid' tag.
014     */
015    public class PanelGridTag extends HtmlPanelGridTag {
016    
017            /**
018             * 'true' if the panelGrid contains alternate colors.
019             */
020            private boolean alternateColors;
021    
022            /**
023             * Constructor.
024             */
025            public PanelGridTag() {
026                    super();
027            }
028    
029            /**
030             * @see org.apache.myfaces.taglib.html.ext.HtmlPanelGridTag#setProperties(javax.faces.component.UIComponent)
031             */
032            @Override
033            protected void setProperties(final UIComponent component) {
034                    TagsConfigurator tagsConfigurator = TagsConfigurator.getInstance();
035                    String styleClass = null;
036                    if (alternateColors) {
037                            styleClass = tagsConfigurator.getDataTableRowClass() 
038                            + ", " + tagsConfigurator.getDataTableRowAlternateClass();
039                    } else {
040                            styleClass = tagsConfigurator.getDataTableRowClass();
041                    }
042                    if (styleClass != null) {
043                            setStringProperty(component, JSFAttr.ROW_CLASSES_ATTR, styleClass);
044                    }
045    
046                    styleClass = tagsConfigurator.getDataTableHeaderClass();
047                    if (styleClass != null) {
048                            setStringProperty(component, JSFAttr.HEADER_CLASS_ATTR, styleClass);
049                    }
050                    styleClass = tagsConfigurator.getDataTableFooterClass();
051                    if (styleClass != null) {
052                            setStringProperty(component, JSFAttr.FOOTER_CLASS_ATTR, styleClass);
053                    }
054                    styleClass = tagsConfigurator.getDataTableColumnClass();
055                    if (styleClass != null) {
056                            setStringProperty(component, JSFAttr.COLUMN_CLASSES_ATTR, styleClass);
057                    }
058                    super.setProperties(component);
059            }
060    
061            /**
062             * @param alternateColors the alternateColors to set
063             */
064            public void setAlternateColors(final boolean alternateColors) {
065                    this.alternateColors = alternateColors;
066            }
067    
068    }
