004    package org.esupportail.commons.web.tags;
005    
006    import javax.faces.application.Application;
007    import javax.faces.component.UIComponent;
008    import javax.faces.context.FacesContext;
009    import javax.faces.el.ValueBinding;
010    import javax.faces.webapp.UIComponentTag;
011    
012    import org.apache.myfaces.taglib.html.HtmlFormTag;
013    import org.esupportail.commons.web.component.UIForm;
014    
015    /**
016     * This class is a tag for esup-commons form element.
017     */
018    public class FormTag extends HtmlFormTag {
019    
020            /**
021             * The text of the submit popup.
022             */
023            private String submitPopupText = "#{msgs['FORM_SUBMIT_POPUP_TEXT']}";
024            
025            /**
026             * The image of the submit popup.
027             */
028            private String submitPopupImage = "/media/images/please-wait.gif";
029            
030            /**
031             * True to show the submit popup text. 
032             */
033            private String showSubmitPopupText = "true";
034            
035            /**
036             * True to show the submit popup image. 
037             */
038            private String showSubmitPopupImage = "true";
039            
040            /**
041             * True to freeze the screen on submit. 
042             */
043            private String freezeScreenOnSubmit = "true";
044            
045            /**
046             * The onsubmit property.
047             */
048            private String onsubmit;
049            
050            /**
051             * Constructor.
052             */
053            public FormTag() {
054                    super();
055            }
056    
057            /**
058             * @see org.apache.myfaces.taglib.html.HtmlFormTag#getComponentType()
059             */
060            @Override
061            public String getComponentType() {
062                    return UIForm.COMPONENT_TYPE;
063            }
064            
065            /**
066             * @return the boolean value of an expression.
067             * @param context
068             * @param application
069             * @param expr
070             * @param defaultValue 
071             */
072            protected boolean getBooleanValue(
073                            final FacesContext context,
074                            final Application application,
075                            final String expr,
076                            final boolean defaultValue) {
077            if (!UIComponentTag.isValueReference(expr)) {
078                    if (expr == null) {
079                            return defaultValue;
080                    }
081                return Boolean.valueOf(expr);
082            }
083            ValueBinding vb = application.createValueBinding(expr);
084            Object value = vb.getValue(context);
085            if (value == null) {
086                    return defaultValue;
087            }
088            if (value instanceof Boolean) {
089                return (Boolean) value;
090            }
091            if (value instanceof String) {
092                return Boolean.valueOf(value.toString());
093            }
094            throw new IllegalArgumentException("Boolean or String class expected for [" + expr + "]");
095            }
096    
097            /**
098             * @see javax.faces.webapp.UIComponentTag#setProperties(javax.faces.component.UIComponent)
099             */
100            @SuppressWarnings("unchecked")
101            @Override
102            protected void setProperties(final UIComponent component) {
103                    component.getAttributes().put(UIForm.SUBMIT_POPUP_TEXT, submitPopupText);
104                    component.getAttributes().put(UIForm.SUBMIT_POPUP_IMAGE, submitPopupImage);
105                    String theOnsubmit = "";
106                    if (onsubmit != null) {
107                            theOnsubmit += "{ " + onsubmit + " }; ";
108                    }
109                    theOnsubmit += "if (typeof(onFormSubmit2) == 'function') { onFormSubmit2(this, ";
110                    theOnsubmit += freezeScreenOnSubmit;
111                    theOnsubmit += ", ";
112                    theOnsubmit += showSubmitPopupText;
113                    theOnsubmit += ", ";
114                    theOnsubmit += showSubmitPopupImage;
115                    theOnsubmit += "); } else onFormSubmit();";
116                    super.setOnsubmit(theOnsubmit);
117                    super.setProperties(component);
118            }
119    
120            /**
121             * @return the submitPopupText
122             */
123            protected String getSubmitPopupText() {
124                    return submitPopupText;
125            }
126    
127            /**
128             * @param submitPopupText the submitPopupText to set
129             */
130            public void setSubmitPopupText(final String submitPopupText) {
131                    this.submitPopupText = submitPopupText;
132            }
133    
134            /**
135             * @return the submitPopupImage
136             */
137            protected String getSubmitPopupImage() {
138                    return submitPopupImage;
139            }
140    
141            /**
142             * @param submitPopupImage the submitPopupImage to set
143             */
144            public void setSubmitPopupImage(final String submitPopupImage) {
145                    this.submitPopupImage = submitPopupImage;
146            }
147    
148            /**
149             * @return the showSubmitPopupText
150             */
151            protected String getShowSubmitPopupText() {
152                    return showSubmitPopupText;
153            }
154    
155            /**
156             * @param showSubmitPopupText the showSubmitPopupText to set
157             */
158            public void setShowSubmitPopupText(final String showSubmitPopupText) {
159                    this.showSubmitPopupText = showSubmitPopupText;
160            }
161    
162            /**
163             * @return the showSubmitPopupImage
164             */
165            protected String getShowSubmitPopupImage() {
166                    return showSubmitPopupImage;
167            }
168    
169            /**
170             * @param showSubmitPopupImage the showSubmitPopupImage to set
171             */
172            public void setShowSubmitPopupImage(final String showSubmitPopupImage) {
173                    this.showSubmitPopupImage = showSubmitPopupImage;
174            }
175    
176            /**
177             * @return the freezeScreenOnSubmit
178             */
179            protected String getFreezeScreenOnSubmit() {
180                    return freezeScreenOnSubmit;
181            }
182    
183            /**
184             * @param freezeScreenOnSubmit the freezeScreenOnSubmit to set
185             */
186            public void setFreezeScreenOnSubmit(final String freezeScreenOnSubmit) {
187                    this.freezeScreenOnSubmit = freezeScreenOnSubmit;
188            }
189    
190            /**
191             * @return the onsubmit
192             */
193            protected String getOnsubmit() {
194                    return onsubmit;
195            }
196    
197            /**
198             * @param onsubmit the onsubmit to set
199             */
200            @Override
201            public void setOnsubmit(final String onsubmit) {
202                    this.onsubmit = onsubmit;
203            }
204    
205    }
