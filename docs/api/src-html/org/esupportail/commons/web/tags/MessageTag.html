004    package org.esupportail.commons.web.tags;
005    
006    import javax.faces.component.UIComponent;
007    
008    import org.apache.myfaces.shared_impl.renderkit.JSFAttr;
009    import org.apache.myfaces.taglib.html.HtmlMessageTag;
010    import org.esupportail.commons.web.tags.config.TagsConfigurator;
011    
012    /**
013     * An extension of h:messages.
014     */
015    public class MessageTag extends HtmlMessageTag {
016    
017            /**
018             * Constructor.
019             */
020            public MessageTag() {
021                    super();
022            }
023            
024            /**
025             * @see org.apache.myfaces.shared_impl.taglib.html.HtmlMessageTagBase#setProperties(
026             * javax.faces.component.UIComponent)
027             */
028            @Override
029            protected void setProperties(final UIComponent component) {
030                    TagsConfigurator tagsConfigurator = TagsConfigurator.getInstance();
031                    String styleClass = tagsConfigurator.getMessagesInfoClass();
032                    if (styleClass != null) {
033                            setStringProperty(component, JSFAttr.INFO_CLASS_ATTR, styleClass);
034                    }
035                    styleClass = tagsConfigurator.getMessagesWarnClass();
036                    if (styleClass != null) {
037                            setStringProperty(component, JSFAttr.WARN_CLASS_ATTR, styleClass);
038                    }
039                    styleClass = tagsConfigurator.getMessagesErrorClass();
040                    if (styleClass != null) {
041                            setStringProperty(component, JSFAttr.ERROR_CLASS_ATTR, styleClass);
042                    }
043                    styleClass = tagsConfigurator.getMessagesFatalClass();
044                    if (styleClass != null) {
045                            setStringProperty(component, JSFAttr.FATAL_CLASS_ATTR, styleClass);
046                    }
047            super.setProperties(component);
048        }
049    
050    }
