004    package org.esupportail.commons.services.remote; 
005    
006    import java.net.InetAddress;
007    import java.net.UnknownHostException;
008    
009    import javax.servlet.http.HttpServletRequest;
010    
011    import org.codehaus.xfire.transport.http.XFireServletController;
012    import org.esupportail.commons.services.logging.Logger;
013    import org.esupportail.commons.services.logging.LoggerImpl;
014    
015    /**
016     * An abstract web service that provides the client of the request.
017     */
018    public class AbstractClientProviderWebService {
019            
020            /**
021             * A logger.
022             */
023            private final Logger logger = new LoggerImpl(getClass());
024    
025            /**
026             * Bean constructor.
027             */
028            public AbstractClientProviderWebService() {
029                    super();
030            }
031    
032            /**
033             * @return the client.
034             */
035            protected InetAddress getClient() {
036                    HttpServletRequest request = XFireServletController.getRequest();
037                    if (request == null) {
038                            logger.warn("could not get the incoming request");
039                            return null;
040                    }
041                    String remoteAddr = request.getRemoteAddr();
042                    if (remoteAddr == null) {
043                            logger.warn("could not get the remote address");
044                            return null;
045                    }
046                    try {
047                            return InetAddress.getByName(remoteAddr);
048                    } catch (UnknownHostException e) {
049                            return null;
050                    }
051            }
052    
053    }
