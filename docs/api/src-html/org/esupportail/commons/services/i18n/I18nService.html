004    package org.esupportail.commons.services.i18n;
005    
006    import java.io.Serializable;
007    import java.util.Locale; 
008    import java.util.Map;
009    
010    /**
011     * The interface of i18n services. An i18n service is a layer to abstract
012     * resources bundles, that should never be accessed directly.
013     */
014    public interface I18nService extends Serializable {
015    
016            /**
017             * @return The default locale.
018             */
019            Locale getDefaultLocale();
020            
021            /**
022             * @return A map that contains the strings available for a locale.
023             * @param locale
024             */
025            Map<String, String> getStrings(Locale locale);
026    
027            /**
028             * @return A map that contains the strings available for the default locale.
029             */
030            Map<String, String> getStrings();
031    
032            /**
033             * @return The string that corresponds to a key and a locale.
034             * @param key
035             * @param locale
036             */
037            String getString(String key, Locale locale);
038    
039            /**
040             * @return The string that corresponds to a key for the default locale.
041             * @param key
042             */
043            String getString(String key);
044    
045            /**
046             * @return The string that corresponds to a key and a locale, where {0} ... {n} are replaced by args.
047             * @param key
048             * @param locale
049             * @param args
050             */
051            String getString(String key, Locale locale, Object [] args);
052    
053            /**
054             * @return The string that corresponds to a key for the default locale, where {0} ... {n} are replaced by args.
055             * @param key
056             * @param args
057             */
058            String getString(String key, Object [] args);
059    
060            /**
061             * @return The string that corresponds to a key and a locale, where {0} is replaced by arg0.
062             * @param key
063             * @param locale
064             * @param arg0
065             */
066            String getString(String key, Locale locale, Object arg0);
067    
068            /**
069             * @return The string that corresponds to a key for the default locale, 
070             * where {0} is replaced by arg0.
071             * @param key
072             * @param arg0
073             */
074            String getString(String key, Object arg0);
075    
076            /**
077             * @return The string that corresponds to a key and a locale, where {0} 
078             * and {1} are replaced by arg0 and arg1.
079             * @param key
080             * @param locale
081             * @param arg0
082             * @param arg1
083             */
084            String getString(String key, Locale locale, Object arg0, Object arg1);
085    
086            /**
087             * @return The string that corresponds to a key and the default locale, 
088             * where {0} and {1} are replaced by arg0 and arg1.
089             * @param key
090             * @param arg0
091             * @param arg1
092             */
093            String getString(String key, Object arg0, Object arg1);
094    
095            /**
096             * @return The string that corresponds to a key and a locale, where {0}, {1} 
097             * and {2} are replaced by arg0, arg1 and arg2.
098             * @param key
099             * @param locale
100             * @param arg0
101             * @param arg1
102             * @param arg2
103             */
104            String getString(String key, Locale locale, Object arg0, Object arg1, Object arg2);
105    
106            /**
107             * @return The string that corresponds to a key and the default locale, 
108             * where {0}, {1} and {2} are replaced by arg0, arg1 and arg2.
109             * @param key
110             * @param arg0
111             * @param arg1
112             * @param arg2
113             */
114            String getString(String key, Object arg0, Object arg1, Object arg2);
115    
116            /**
117             * @return The string that corresponds to a key and a locale, where {0}, {1}, 
118             * {2} and {3} are replaced by arg0, arg1, arg2 and arg3.
119             * @param key
120             * @param locale
121             * @param arg0
122             * @param arg1
123             * @param arg2
124             * @param arg3
125             */
126            String getString(String key, Locale locale, Object arg0, Object arg1, Object arg2, Object arg3);
127    
128            /**
129             * @return The string that corresponds to a key, where {0}, {1}, 
130             * {2} and {3} are replaced by arg0, arg1, arg2 and arg3.
131             * @param key
132             * @param arg0
133             * @param arg1
134             * @param arg2
135             * @param arg3
136             */
137            String getString(String key, Object arg0, Object arg1, Object arg2, Object arg3);
138    
139            /**
140             * @return The string that corresponds to a key and a locale, where {0}, {1}, 
141             * {2} and {3} are replaced by arg0, arg1, arg2, arg3 and arg4.
142             * @param key
143             * @param locale
144             * @param arg0
145             * @param arg1
146             * @param arg2
147             * @param arg3
148             * @param arg4
149             */
150            String getString(String key, Locale locale, Object arg0, Object arg1, Object arg2, Object arg3, Object arg4);
151    
152            /**
153             * @return The string that corresponds to a key and a locale, where {0}, {1}, 
154             * {2} and {3} are replaced by arg0, arg1, arg2, arg3 and arg4.
155             * @param key
156             * @param arg0
157             * @param arg1
158             * @param arg2
159             * @param arg3
160             * @param arg4
161             */
162            String getString(String key, Object arg0, Object arg1, Object arg2, Object arg3, Object arg4);
163    
164            /**
165             * @return The string that corresponds to a key and a locale, where {0}, {1}, 
166             * {2} {3}, {4} and {5} are replaced by arg0, arg1, arg2, arg3, arg4 and arg5.
167             * @param key
168             * @param locale
169             * @param arg0
170             * @param arg1
171             * @param arg2
172             * @param arg3
173             * @param arg4
174             * @param arg5
175             */
176            String getString(String key, Locale locale, 
177                            Object arg0, Object arg1, Object arg2, Object arg3, Object arg4, Object arg5);
178    
179            /**
180             * @return The string that corresponds to a key and a locale, where {0}, {1}, 
181             * {2} {3}, {4} and {5} are replaced by arg0, arg1, arg2, arg3, arg4 and arg5.
182             * @param key
183             * @param arg0
184             * @param arg1
185             * @param arg2
186             * @param arg3
187             * @param arg4
188             * @param arg5
189             */
190            String getString(String key, 
191                            Object arg0, Object arg1, Object arg2, Object arg3, Object arg4, Object arg5);
192    
193            /**
194             * @return The string that corresponds to a key and a locale, where {0}, {1}, 
195             * {2} {3}, {4} and {5} are replaced by arg0, arg1, arg2, arg3, arg4 and arg5.
196             * @param key
197             * @param locale
198             * @param arg0
199             * @param arg1
200             * @param arg2
201             * @param arg3
202             * @param arg4
203             * @param arg5
204             * @param arg6
205             */
206            String getString(String key, Locale locale, 
207                            Object arg0, Object arg1, Object arg2, Object arg3, Object arg4, 
208                            Object arg5, Object arg6);
209    
210            /**
211             * @return The string that corresponds to a key and a locale, where {0}, {1}, 
212             * {2} {3}, {4} and {5} are replaced by arg0, arg1, arg2, arg3, arg4 and arg5.
213             * @param key
214             * @param arg0
215             * @param arg1
216             * @param arg2
217             * @param arg3
218             * @param arg4
219             * @param arg5
220             * @param arg6
221             */
222            String getString(String key, 
223                            Object arg0, Object arg1, Object arg2, Object arg3, Object arg4, 
224                            Object arg5, Object arg6);

226            /**
227             * @return The string that corresponds to a key and a locale, where {0}, {1}, 
228             * {2} {3}, {4} and {5} are replaced by arg0, arg1, arg2, arg3, arg4 and arg5.
229             * @param key

231             * @param arg0

234             * @param arg3

236             * @param arg5
237             * @param arg6
238             * @param arg7

240            String getString(String key, Locale locale, 
241                            Object arg0, Object arg1, Object arg2, Object arg3, Object arg4,

244            /**
245             * @return The string that corresponds to a key and a locale, where {0}, {1}, 
246             * {2} {3}, {4} and {5} are replaced by arg0, arg1, arg2, arg3, arg4 and arg5.
247             * @param key

249             * @param arg1
250             * @param arg2
251             * @param arg3
252             * @param arg4
253             * @param arg5
254             * @param arg6
255             * @param arg7
256             */
257            String getString(String key, 
258                            Object arg0, Object arg1, Object arg2, Object arg3, Object arg4, 
259                            Object arg5, Object arg6, Object arg7);
260    
261            /**
262             * @return The string that corresponds to a key and a locale, where {0}, {1}, 
263             * {2} {3}, {4} and {5} are replaced by arg0, arg1, arg2, arg3, arg4 and arg5.
264             * @param key
265             * @param locale
266             * @param arg0
267             * @param arg1
268             * @param arg2
269             * @param arg3
270             * @param arg4
271             * @param arg5
272             * @param arg6
273             * @param arg7
274             * @param arg8
275             */
276            String getString(String key, Locale locale, 
277                            Object arg0, Object arg1, Object arg2, Object arg3, Object arg4, 
278                            Object arg5, Object arg6, Object arg7, Object arg8);
279    
280            /**
281             * @return The string that corresponds to a key and a locale, where {0}, {1}, 
282             * {2} {3}, {4} and {5} are replaced by arg0, arg1, arg2, arg3, arg4 and arg5.
283             * @param key

295                            Object arg0, Object arg1, Object arg2, Object arg3, Object arg4, 
296                            Object arg5, Object arg6, Object arg7, Object arg8);
297    
298            /**
299             * @return The string that corresponds to a key and a locale, where {0}, {1}, 
300             * {2} {3}, {4} and {5} are replaced by arg0, arg1, arg2, arg3, arg4 and arg5.
301             * @param key

303             * @param arg0

307             * @param arg4
308             * @param arg5
309             * @param arg6
310             * @param arg7
311             * @param arg8
312             * @param arg9
313             */
314            String getString(String key, Locale locale, 
315                            Object arg0, Object arg1, Object arg2, Object arg3, Object arg4, 
316                            Object arg5, Object arg6, Object arg7, Object arg8, Object arg9);
317    
318            /**
319             * @return The string that corresponds to a key and a locale, where {0}, {1}, 
320             * {2} {3}, {4} and {5} are replaced by arg0, arg1, arg2, arg3, arg4 and arg5.

324             * @param arg2

327             * @param arg5

334                            Object arg0, Object arg1, Object arg2, Object arg3, Object arg4, 
335                            Object arg5, Object arg6, Object arg7, Object arg8, Object arg9);

338             * @return The string that corresponds to a key and a locale, where {0}, {1}, 
339             * {2} {3}, {4} and {5} are replaced by arg0, arg1, arg2, arg3, arg4 and arg5.

342             * @param arg0

347             * @param arg5
348             * @param arg6
349             * @param arg7

354            String getString(String key, Locale locale, 
355                            Object arg0, Object arg1, Object arg2, Object arg3, Object arg4, 
356                            Object arg5, Object arg6, Object arg7, Object arg8, Object arg9,
357                            Object arg10);
358

360             * @return The string that corresponds to a key and a locale, where {0}, {1}, 
361             * {2} {3}, {4} and {5} are replaced by arg0, arg1, arg2, arg3, arg4 and arg5.
362             * @param key
363             * @param arg0
364             * @param arg1
365             * @param arg2

367             * @param arg4
368             * @param arg5
369             * @param arg6
370             * @param arg7
371             * @param arg8
372             * @param arg9
373             * @param arg10
374             */
375            String getString(String key, 
376                            Object arg0, Object arg1, Object arg2, Object arg3, Object arg4, 
377                            Object arg5, Object arg6, Object arg7, Object arg8, Object arg9,
378                            Object arg10);
379    
380            /**
381             * @return The date as a printable string corresponding to a Locale. The format 
382             * used depends on the current date.
383             * @param date
384             * @param locale
385             */
386            String printableRelativeDate(long date, Locale locale);
387    
388            /**
389             * @return The date as a printable string corresponding to the default Locale. 
390             * The format used depends on the current date.
391             * @param date
392             */
393            String printableRelativeDate(long date);
394    
395            /**
396             * @return The date as a printable corresponding to a Locale.
397             * @param date
398             * @param locale
399             */
400            String printableDate(long date, Locale locale);
401            
402            /**
403             * @return The date as a printable corresponding to the default Locale.
404             * @param date
405             */
406            String printableDate(long date);
