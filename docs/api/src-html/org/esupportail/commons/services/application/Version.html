004    package org.esupportail.commons.services.application; 
005    
006    
007    /**
008     * A class that provides facilities with versionning.
009     */
010    public class Version {
011    
012            /**
013             * The number of tokens in the version numbers.
014             */
015            private static final int TOKENS_NUMBER = 3;
016            
017            /**
018             * The major number.
019             */
020            private int major; 
021            
022            /**
023             * The minor number.
024             */
025            private int minor;
026            
027            /**
028             * The update number.
029             */
030            private int update;
031    
032            /**
033             * Constructor.
034             * @param versionString a textual representation of the version
035             * @throws VersionException 
036             */
037            public Version(final String versionString) throws VersionException {
038                    super();
039                    try {
040                            String [] numbers = versionString.split("\\.");
041                            if (numbers.length == TOKENS_NUMBER) {
042                                    major = Integer.parseInt(numbers[0]);
043                                    minor = Integer.parseInt(numbers[1]); 
044                                    update = Integer.parseInt(numbers[2]);
045                                    if (major >= 0 && minor >= 0 && update >= 0) {
046                                            return;
047                                    }
048                            }
049                    } catch (NumberFormatException e) {
050                            // see below
051                    } 
052                    throw new VersionException("invalid version number [" + versionString + "]");
053            }
054            
055            /**
056             * @see java.lang.Object#equals(java.lang.Object)
057             */
058            @Override
059            public boolean equals(final Object obj) {
060                    Version ref = null;
061                    if (obj == null) {
062                            return false;
063                    }
064                    if (obj instanceof Version) {
065                            ref = (Version) obj;
066                    } else if (obj instanceof String) {
067                            ref = new Version((String) obj);
068                    } else {
069                            return false;
070                    }
071                    return major == ref.major && minor == ref.minor && update == ref.update;
072            }
073    
074            /**
075             * @see java.lang.Object#hashCode()
076             */
077            @Override
078            public int hashCode() {
079                    return super.hashCode();
080            }
081    
082            /**
083             * @param ref 
084             * @return true if same major and minor number than another version.
085             */
086            public boolean isSameMajorAndMinor(final Version ref) {
087                    return major == ref.major && minor == ref.minor;
088            }
089    
090            /**
091             * @param ref
092             * @return true if older than another version.
093             */
094            public boolean isOlderThan(final Version ref) {
095                    if (major != ref.major) {
096                            return major < ref.major;
097                    }
098                    if (minor != ref.minor) {
099                            return minor < ref.minor;
100                    }
101                    return update < ref.update;
102            }
103    
104            /**
105             * @param ref
106             * @return true if older than another version.
107             */
108            public boolean isOlderThan(final String ref) {
109                    return isOlderThan(new Version(ref));
110            }
111    
112            /**
113             * @see java.lang.Object#toString()
114             */
115            @Override
116            public String toString() {
117                    return major + "." + minor + "." + update;
118            }
119    
120    }
