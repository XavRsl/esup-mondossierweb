004    package org.esupportail.commons.services.application; 
005    
006    import java.io.Serializable;
007    
008    
009    /**
010     * the interface of the beans for versionning management.
011     */
012    public interface VersionningService extends Serializable {
013    
014            /**
015             * Initialize the database.
016             */
017            void initDatabase();
018            
019            /**
020             * check the database version, silently upgrade if possible.
021             * @param throwException 
022             * @param printLatestVersion 
023             * @throws VersionException 
024             */
025            void checkVersion(
026                            final boolean throwException,
027                            final boolean printLatestVersion) throws VersionException;
028            
029            /**
030             * Upgrade the database.
031             * @return true if the method should be called again, false otherwise.
032             */
033            boolean upgradeDatabase();
034            
035    }
