004    package org.esupportail.commons.services.authentication;
005    
006    import edu.yale.its.tp.cas.client.filter.CASFilter;
007    
008    import org.esupportail.commons.utils.ContextUtils;
009    
010    /** 
011     * A CAS authenticator.
012     */
013    public class CasFilterAuthenticationService extends AbstractTypedAuthenticationService {
014    
015            /**
016             * The serialization id.
017             */
018            private static final long serialVersionUID = 943489018651202646L;
019    
020            /**
021             * Bean constructor.
022             */
023            public CasFilterAuthenticationService() {
024                    super();
025            }
026    
027            /**
028             * @see org.esupportail.commons.services.authentication.AbstractTypedAuthenticationService#getAuthId()
029             */
030            @Override
031            protected String getAuthId() {
032                    return (String) ContextUtils.getGlobalSessionAttribute(CASFilter.CAS_FILTER_USER);
033            }
034    
035            /**
036             * @see org.esupportail.commons.services.authentication.AbstractTypedAuthenticationService#getAuthType()
037             */
038            @Override
039            protected String getAuthType() {
040                    return AuthUtils.CAS;
041            }
042    
043    }
