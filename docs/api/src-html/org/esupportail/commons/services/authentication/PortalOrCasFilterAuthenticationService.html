004    package org.esupportail.commons.services.authentication;
005    
006    import java.util.ArrayList;
007    import java.util.List;
008    
009    
010    /** 
011     * A class to resolve the current user depending on the installation:
012     * - for portlet installations, rely on the portal,
013     * - for servlet installations, rely on the CAS filter. 
014     * @deprecated Use DelegatingAuthenticationService instead.
015     */
016    @Deprecated
017    public class PortalOrCasFilterAuthenticationService extends DelegatingAuthenticationService {
018    
019            /**
020             * The serialization id.
021             */
022            private static final long serialVersionUID = 1030107540760290734L;
023            
024            /**
025             * Bean constructor.
026             */
027            public PortalOrCasFilterAuthenticationService() {
028                    super();
029                    List<AuthenticationService> authenticationServices = new ArrayList<AuthenticationService>();
030                    PortalAuthenticationService portalAuthenticator = new PortalAuthenticationService();
031                    portalAuthenticator.setAuthType(AuthUtils.CAS);
032                    authenticationServices.add(portalAuthenticator);
033                    authenticationServices.add(new CasFilterAuthenticationService());
034                    super.setAuthenticationServices(authenticationServices);
035            }
036    
037            /**
038             * @return the portal authenticator.
039             */
040            protected PortalAuthenticationService getPortalAuthenticator() {
041                    return (PortalAuthenticationService) getAuthenticationServices().get(0);
042            }
043    
044            /**
045             * @param uidPortalAttribute the uidPortalAttribute to set
046             */
047            public void setUidPortalAttribute(final String uidPortalAttribute) {
048                    getPortalAuthenticator().setUidPortalAttribute(uidPortalAttribute);
049            }
050    
051            /**
052             * @param portalAuthType the portalAuthType to set
053             */
054            public void setPortalAuthType(final String portalAuthType) {
055                    getPortalAuthenticator().setAuthType(portalAuthType);
056            }
057    
058            /**
059             * @see org.esupportail.commons.services.authentication.DelegatingAuthenticationService
060             * #setAuthenticationServices(List)
061             */
062            @Override
063            public void setAuthenticationServices(
064                            @SuppressWarnings("unused")
065                            final List<AuthenticationService> authenticators) {
066                    throw new UnsupportedOperationException(
067                                    "method " + getClass() + ".setAuthenticators() should never be called.");
068            }
069    
070    }
