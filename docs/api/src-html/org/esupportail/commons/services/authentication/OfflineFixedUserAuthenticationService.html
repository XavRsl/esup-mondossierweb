004    package org.esupportail.commons.services.authentication;
005    
006    import org.esupportail.commons.utils.Assert;
007    import org.esupportail.commons.utils.strings.StringUtils;
008    
009    /** 
010     * This authenticator is to be used when working offline, it always returns the same user.
011     */
012    public class OfflineFixedUserAuthenticationService extends AbstractTypedAuthenticationService {
013            
014            /**
015             * The serialization id.
016             */
017            private static final long serialVersionUID = 1397301690881143218L;
018    
019            /**
020             * The id of the fixed user.
021             */
022            private String authId;
023    
024            /**
025             * Bean constructor.
026             */
027            public OfflineFixedUserAuthenticationService() {
028                    super();
029            }
030    
031            /**
032             * @see org.springframework.beans.factory.InitializingBean#afterPropertiesSet()
033             */
034            @Override
035            public void afterPropertiesSet() {
036                    super.afterPropertiesSet();
037                    Assert.hasText(this.authId, 
038                                    "property authId of class " + this.getClass().getName() 
039                                    + " can not be null");
040            }
041    
042            /**
043             * @see org.esupportail.commons.services.authentication.AbstractRealAuthenticationService#getAuthId()
044             */
045            @Override
046            protected String getAuthId() {
047                    return authId;
048            }
049    
050            /**
051             * @param authId the authId to set
052             */
053            public void setAuthId(final String authId) {
054                    this.authId = StringUtils.nullIfEmpty(authId);
055            }
056    
057    }
