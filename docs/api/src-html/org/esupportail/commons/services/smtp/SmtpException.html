004    package org.esupportail.commons.services.smtp; 
005    
006    import org.esupportail.commons.exceptions.EsupException;
007    
008    /**
009     * An class to represent the exceptions thrown when sendin mails.
010     */
011    public class SmtpException extends EsupException {
012    
013            /**
014             * the id for serialization.
015             */
016            private static final long serialVersionUID = 9093603917198471034L;
017    
018            /**
019             * Bean constructor.
020             */
021            public SmtpException() {
022                    super();
023            }
024    
025            /**
026             * @param message
027             */
028            public SmtpException(final String message) {
029                    super(message);
030            }
031    
032            /**
033             * @param cause
034             */
035            public SmtpException(final Exception cause) {
036                    super(cause);
037            }
038    
039            /**
040             * @param message
041             * @param cause
042             */
043            public SmtpException(final String message, final Exception cause) {
044                    super(message, cause);
045            }
046    
047    }
