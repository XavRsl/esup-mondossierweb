197            /**
198             * @param separator
199             * @return a StringBuffer.
200             */
201            private StringBuffer getTextReportInformation(
202                            final String separator) {
203                    StringBuffer sb = new StringBuffer();
204                    sb.append(TEXT_HRULE);
205                    sb.append(TEXT_CR + getString("EXCEPTION.HEADER.INFORMATION"));
206                    sb.append(TEXT_HRULE);
207                    sb.append(TEXT_CR + getString("EXCEPTION.INFORMATION.APPLICATION") 
208                                    + separator + applicationName);
209                    sb.append(TEXT_CR + getString("EXCEPTION.INFORMATION.VERSION") 
210                                    + separator + applicationVersion);
211                    if (server == null) {
212                            server = getString("EXCEPTION.INFORMATION.SERVER.UNKNOWN");
213                    }
214                    sb.append(TEXT_CR + getString("EXCEPTION.INFORMATION.SERVER") 
215                                    + separator + server);
216                    String dateStr;
217                    if (date == null) { 
218                            dateStr = getString("EXCEPTION.INFORMATION.DATE.UNKNOWN");
219                    } else {
220                            dateStr = printableDate(date.longValue());
221                    }
222                    sb.append(TEXT_CR + getString("EXCEPTION.INFORMATION.DATE") 
223                                    + separator + dateStr);
224                    if (ContextUtils.isWeb()) {
225                            if (userId == null) {
226                                    userId = getString("EXCEPTION.INFORMATION.USER_ID.UNKNOWN");
227                            }
228                            sb.append(TEXT_CR + getString("EXCEPTION.INFORMATION.USER_ID") 
229                                            + separator + userId);
230                            if (portal == null) {
231                                    portal = getString("EXCEPTION.INFORMATION.PORTAL.UNKNOWN");
232                            } else {
233                                    if (quickStart != null && quickStart.booleanValue()) {
234                                            portal = portal + " " 
235                                            + getString("EXCEPTION.INFORMATION.PORTAL.QUICK_START");
236                                    }
237                            }
238                            sb.append(TEXT_CR + getString("EXCEPTION.INFORMATION.PORTAL") 
239                                            + separator + portal);
240                            if (client == null) {
241                                    client = getString("EXCEPTION.INFORMATION.CLIENT.UNKNOWN");
242                            }
243                            sb.append(TEXT_CR + getString("EXCEPTION.INFORMATION.CLIENT") 
244                                            + separator + client);
245                            if (queryString == null) {
246                                    queryString = 
247                                            getString("EXCEPTION.INFORMATION.QUERY_STRING.UNKNOWN");
248                            }
249                            sb.append(TEXT_CR + getString("EXCEPTION.INFORMATION.QUERY_STRING") 
250                                            + separator + queryString);
251                            if (userAgent == null) {
252                                    userAgent = getString("EXCEPTION.INFORMATION.USER_AGENT.UNKNOWN");
253                            } else {
254                                    userAgent = StringUtils.escapeHtml(userAgent);
255                            }
256                            sb.append(TEXT_CR + getString("EXCEPTION.INFORMATION.USER_AGENT") 
257                                            + separator + userAgent);
258                    }
259                    return sb;
260            }

262            /**
263             * @param separator 
264             * @return a StringBuffer.
265             */
266            private StringBuffer getTextReportException(
267                            final String separator) {
268                    StringBuffer sb = new StringBuffer();
269                    if (throwable != null) {
270                            Throwable cause = ExceptionUtils.getRealCause(throwable);
271                            sb.append(TEXT_HRULE);
272                            sb.append(TEXT_CR + getString("EXCEPTION.HEADER.EXCEPTION"));
273                            sb.append(TEXT_HRULE);
274                            sb.append(TEXT_CR + getString("EXCEPTION.EXCEPTION.NAME") 
275                                            + separator + cause.getClass().getSimpleName());
276                            sb.append(TEXT_CR + getString("EXCEPTION.EXCEPTION.MESSAGE") 
277                                            + separator + cause.getMessage());
278                            sb.append(TEXT_CR + getString("EXCEPTION.EXCEPTION.SHORT_STACK_TRACE") 
279                                            + separator);
280                            sb.append(TEXT_CR + ExceptionUtils.getShortPrintableStackTrace(throwable));
281                            sb.append(TEXT_CR + getString("EXCEPTION.EXCEPTION.STACK_TRACE") 
282                                            + separator);
283                            sb.append(TEXT_CR + ExceptionUtils.getPrintableStackTrace(throwable));
284                    }
285                    return sb;
286            }

288            /**
289             * @param set 
290             * @return a printable String for text outputs.
291             */
292            private String text(final Set<String> set) {
293                    if (set == null || set.isEmpty()) {
294                            return null;
295                    }
296                    StringBuffer result = new StringBuffer();
297                    String separator = "";
298                    for (String string : set) {
299                            result.append(separator).append(string);
300                            separator = "\n";
301                    }
302                    return result.toString();
303            }

305            /**
306             * @return a StringBuffer.
307             */
308            private StringBuffer getTextReportRequest() {
309                    String globalSessionAttributesStr = text(globalSessionAttributes);
310                    String sessionAttributesStr = text(sessionAttributes);
311                    String requestAttributesStr = text(requestAttributes);
312                    String requestHeadersStr = text(requestHeaders);
313                    String requestParametersStr = text(requestParameters);
314                    String cookiesStr = text(cookies);
315                    StringBuffer sb = new StringBuffer();
316                    sb.append(TEXT_HRULE);
317                    sb.append(TEXT_CR + getString("EXCEPTION.HEADER.REQUEST_ATTRIBUTES"));
318                    sb.append(TEXT_HRULE);
319                    if (requestAttributesStr == null) {
320                            requestAttributesStr = getString("EXCEPTION.REQUEST_ATTRIBUTES.NONE");
321                    }
322                    sb.append(TEXT_CR + requestAttributesStr);
323                    sb.append(TEXT_HRULE);
324                    sb.append(TEXT_CR + getString("EXCEPTION.HEADER.SESSION_ATTRIBUTES"));
325                    sb.append(TEXT_HRULE);
326                    if (sessionAttributesStr == null) {
327                            sessionAttributesStr = getString("EXCEPTION.SESSION_ATTRIBUTES.NONE");
328                    }
329                    sb.append(TEXT_CR + sessionAttributesStr);
330                    if (ContextUtils.isPortlet()) {
331                            sb.append(TEXT_HRULE);
332                            sb.append(TEXT_CR + getString("EXCEPTION.HEADER.GLOBAL_SESSION_ATTRIBUTES"));
333                            sb.append(TEXT_HRULE);
334                            if (globalSessionAttributesStr == null) {
335                                    globalSessionAttributesStr = getString("EXCEPTION.GLOBAL_SESSION_ATTRIBUTES.NONE");
336                            }
337                            sb.append(TEXT_CR + globalSessionAttributesStr);
338                    }
339                    sb.append(TEXT_HRULE);
340                    sb.append(TEXT_CR + getString("EXCEPTION.HEADER.REQUEST_HEADERS"));
341                    sb.append(TEXT_HRULE);
342                    if (requestHeadersStr == null) {
343                            requestHeadersStr = getString("EXCEPTION.REQUEST_HEADERS.NONE");
344                    }
345                    sb.append(TEXT_CR + requestHeadersStr);
346                    sb.append(TEXT_HRULE);
347                    sb.append(TEXT_CR + getString("EXCEPTION.HEADER.REQUEST_PARAMETERS"));
348                    sb.append(TEXT_HRULE);
349                    if (requestParametersStr == null) {
350                            requestParametersStr = getString("EXCEPTION.REQUEST_PARAMETERS.NONE");
351                    }
352                    sb.append(TEXT_CR + requestParametersStr);
353                    sb.append(TEXT_HRULE);
354                    sb.append(TEXT_CR + getString("EXCEPTION.HEADER.COOKIES"));
355                    sb.append(TEXT_HRULE);
356                    if (cookiesStr == null) {
357                            cookiesStr = getString("EXCEPTION.COOKIES.NONE");
358                    }
359                    sb.append(TEXT_CR + cookiesStr);
360                    return sb;
361            }

377                    return sb;

381            /**
382             * @return a plain/text report of the throwable.
383             */
384            protected String getTextReport() {
385                    StringBuffer sb = new StringBuffer(getString("EXCEPTION.TITLE"));
386                    String separator = getString("EXCEPTION.SEPARATOR");
387                    sb.append(getTextReportInformation(separator));
388                    sb.append(getTextReportException(separator));
389                    if (ContextUtils.isWeb()) {
390                            sb.append(getTextReportRequest());
391                    }
392                    sb.append(getTextReportProperties());
393                    sb.append(TEXT_HRULE);
394                    return sb.toString();
395            }

397            /**
398             * @see org.esupportail.commons.services.exceptionHandling.ExceptionService#setParameters(
399             * java.lang.Throwable)
400             */
401            public void setParameters(
402                            final Throwable t) throws ExceptionHandlingException {
403                    try {
404                            throwable = t;
405                            applicationName = getApplicationService().getName();
406                            applicationVersion = getApplicationService().getVersion();
407                            date = new Long(System.currentTimeMillis());
408                            if (ContextUtils.isPortlet()) {
409                                    portal = HttpUtils.getPortalInfo();
410                            }
411                            quickStart = getApplicationService().isQuickStart();
412                            systemProperties = SystemUtils.getSystemPropertiesStrings();
413                            freeMemory = SystemUtils.getFreeMemory();
414                            maxMemory = SystemUtils.getMaxMemory();
415                            totalMemory = SystemUtils.getTotalMemory();
416                            if (ContextUtils.isWeb()) {
417                                    server = HttpUtils.getServerString();
418                                    if (authenticationService != null) {
419                                            AuthInfo authInfo = null;
420                                            try {
421                                                    authInfo = authenticationService.getAuthInfo();
422                                                    if (authInfo != null) {
423                                                            userId = authInfo.getId();
424                                                    }
425                                            } catch (Throwable e) {
426                                                    // forget it...
427                                            }
428                                    }
429                                    client = HttpUtils.getClientString();
430                                    queryString = HttpUtils.getQueryString();
431                                    userAgent = HttpUtils.getUserAgent();
432                                    requestAttributes = ContextUtils.getRequestAttributesStrings();
433                                    sessionAttributes = ContextUtils.getSessionAttributesStrings();
434                                    if (ContextUtils.isPortlet()) {
435                                            globalSessionAttributes = ContextUtils.getGlobalSessionAttributesStrings();
436                                    }
437                                    requestHeaders = HttpUtils.getRequestHeadersStrings();
438                                    requestParameters = HttpUtils.getRequestParametersStrings();
439                                    cookies = HttpUtils.getCookiesStrings();
440                            }
441                            if (server == null) {
442                                    server = SystemUtils.getServer();
443                            }
444                    } catch (Throwable t2) {
445                            logger.error("The following error was thrown by the application:", t);
446                            logger.error("Another throwable was thrown while logging the first one (giving up):", t2);
447                            throw new ExceptionHandlingException(t2);
448                    }
449            }

451            /**
452             * Log a text report.
453             * @param t
454             * @param textReport
455             */
456            protected void logTextReport(final Throwable t, final String textReport) {
457                    if (SimpleExceptionServiceFactoryImpl.ERROR.equals(logLevel.toLowerCase())) {
458                            logger.error(textReport);
459                    } else if (SimpleExceptionServiceFactoryImpl.WARN.equals(logLevel.toLowerCase())) {
460                            logger.error(ExceptionUtils.getShortPrintableStackTrace(t));
461                            logger.warn(textReport);
462                    } else if (SimpleExceptionServiceFactoryImpl.INFO.equals(logLevel.toLowerCase())) {
463                            logger.error(ExceptionUtils.getShortPrintableStackTrace(t));
464                            logger.info(textReport);
465                    } else if (SimpleExceptionServiceFactoryImpl.TRACE.equals(logLevel.toLowerCase())) {
466                            logger.error(ExceptionUtils.getShortPrintableStackTrace(t));
467                            if (logger.isTraceEnabled()) {
468                                    logger.trace(textReport);
469                            }
470                    } else {
471                            logger.error(ExceptionUtils.getShortPrintableStackTrace(t));
472                            if (logger.isDebugEnabled()) {
473                                    logger.debug(textReport);
474                            }
475                    }
476            }

478            /**
479             * @see org.esupportail.commons.services.exceptionHandling.ExceptionService#handleException()
480             */
481            public void handleException() throws ExceptionHandlingException {
482                    // try to log the throwable
483                    try {
484                            logTextReport(getThrowable(), getTextReport());
485                    } catch (Throwable t) {
486                            logger.error("The following error was thrown by the application:", throwable);
487                            logger.error("Another throwable was thrown while logging the first one (giving up):", t);
488                            throw new ExceptionHandlingException(t);
489                    }
490            }

625            /**
626             * @see org.esupportail.commons.services.exceptionHandling.ExceptionService#getExceptionView()
627             */
628            @SuppressWarnings("unchecked")
629            public String getExceptionView() {
630                    if (throwable != null) {
631                            for (Class clazz : exceptionViews.keySet()) {
632                                    for (Throwable cause : ExceptionUtils.getCauses(throwable)) {
633                                            Class causeClass = cause.getClass(); 
634                                            if (clazz.isAssignableFrom(causeClass)) {
635                                                    return exceptionViews.get(clazz);
636                                            }
637                                    }
638                            }
639                    }
640                    return exceptionViews.get(Exception.class);
641            }
