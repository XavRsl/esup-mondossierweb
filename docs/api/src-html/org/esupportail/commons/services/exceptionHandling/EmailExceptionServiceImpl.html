109            /**
110             * @return a HTML String that represents a row of one 2-cols header cell.
111             * @param title
112             */
113            private String header(final String title) {
114                    StringBuffer sb = new StringBuffer();
115                    sb.append("\n<tr>");
116                    sb.append("\n<th colspan=\"2\" align=\"left\">");
117                    sb.append(title);
118                    sb.append("\n</th>");
119                    sb.append("\n</tr>");
120                    return sb.toString();
121            }

173            /**
174             * @return a StringBuffer.
175             */
176            private StringBuffer getHtmlReportInformation() {
177                    StringBuffer sb = new StringBuffer();
178                    sb.append(header(getString("EXCEPTION.HEADER.INFORMATION")));
179                    sb.append(row2(getString("EXCEPTION.INFORMATION.APPLICATION"), 
180                                    StringUtils.escapeHtml(getApplicationName())));
181                    sb.append(row2(getString("EXCEPTION.INFORMATION.VERSION"), 
182                                    StringUtils.escapeHtml(getApplicationVersion().toString())));
183                    String serverStr;
184                    if (getServer() == null) {
185                            serverStr = em(getString("EXCEPTION.INFORMATION.SERVER.UNKNOWN"));
186                    } else {
187                            serverStr = StringUtils.escapeHtml(getServer());
188                    }
189                    sb.append(row2(getString("EXCEPTION.INFORMATION.SERVER"), serverStr));
190                    String dateStr = null;
191                    if (getDate() == null) {
192                            dateStr = em(getString("EXCEPTION.INFORMATION.DATE.UNKNOWN"));
193                    } else {
194                            dateStr = StringUtils.escapeHtml(printableDate(getDate()));
195                    }
196                    sb.append(row2(getString("EXCEPTION.INFORMATION.DATE"), dateStr));
197                    if (ContextUtils.isWeb()) {
198                            String userIdStr;
199                            if (getUserId() == null) {
200                                    userIdStr = em(getString("EXCEPTION.INFORMATION.USER_ID.UNKNOWN"));
201                            } else {
202                                    userIdStr = StringUtils.escapeHtml(getUserId());
203                            }
204                            sb.append(row2(getString("EXCEPTION.INFORMATION.USER_ID"), userIdStr));
205                            String portalStr;
206                            if (getPortal() == null) {
207                                    portalStr = 
208                                            em(getString("EXCEPTION.INFORMATION.PORTAL.UNKNOWN"));
209                            } else {
210                                    portalStr = StringUtils.escapeHtml(getPortal());
211                                    if (getQuickStart() != null && getQuickStart().booleanValue()) {
212                                            portalStr = portalStr + " " 
213                                            + getString("EXCEPTION.INFORMATION.PORTAL.QUICK_START");
214                                    }
215                                    sb.append(row2(getString(
216                                                    "EXCEPTION.INFORMATION.PORTAL"), portalStr));
217                                    String clientStr;
218                                    if (getClient() == null) {
219                                            clientStr = em(getString(
220                                                            "EXCEPTION.INFORMATION.CLIENT.UNKNOWN"));
221                                    } else {
222                                            clientStr = StringUtils.escapeHtml(getClient());
223                                    }
224                                    sb.append(row2(getString("EXCEPTION.INFORMATION.CLIENT"), clientStr));
225                                    String queryStringStr;
226                                    if (getQueryString() == null) {
227                                            queryStringStr = 
228                                                    em(getString(
229                                                                    "EXCEPTION.INFORMATION.QUERY_STRING.UNKNOWN"));
230                                    } else {
231                                            queryStringStr = StringUtils.escapeHtml(getQueryString());
232                                    }
233                                    sb.append(row2(
234                                                    getString("EXCEPTION.INFORMATION.QUERY_STRING"), 
235                                                    queryStringStr));
236                                    String userAgentStr;
237                                    if (getUserAgent() == null) {
238                                            userAgentStr = em(getString(
239                                                            "EXCEPTION.INFORMATION.USER_AGENT.UNKNOWN"));
240                                    } else {
241                                            userAgentStr = StringUtils.escapeHtml(getUserAgent());
242                                    }
243                                    sb.append(row2(getString(
244                                                    "EXCEPTION.INFORMATION.USER_AGENT"), userAgentStr));
245                            }
246                    }
247                    return sb;
248            }

250            /**
251             * @param strings
252             * @return a printable String for text outputs.
253             */
254            private String html(final Collection<String> strings) {
255                    if (strings == null || strings.isEmpty()) {
256                            return null;
257                    }
258                    StringBuffer result = new StringBuffer();
259                    String separator = "";
260                    for (String string : strings) {
261                            result.append(separator).append(StringUtils.escapeHtml(string));
262                            separator = "<br>";
263                    }
264                    return result.toString();
265            }

267            /**
268             * @return a StringBuffer.
269             */
270            private StringBuffer getHtmlReportException() {
271                    StringBuffer sb = new StringBuffer();
272                    sb.append(header(getString("EXCEPTION.HEADER.EXCEPTION")));
273                    String name = "";
274                    String message = "";
275                    String stackTrace = "";
276                    String shortStackTrace = "";
277                    Throwable t = getThrowable();
278                    if (t != null) {
279                            Throwable cause = ExceptionUtils.getRealCause(t);
280                            name = StringUtils.escapeHtml(cause.getClass().getSimpleName());
281                            message = StringUtils.escapeHtml(cause.getMessage());
282                            shortStackTrace = html(ExceptionUtils.getShortStackTraceStrings(t));
283                            stackTrace = html(ExceptionUtils.getStackTraceStrings(t));
284                    }
285                    sb.append(row2(getString("EXCEPTION.EXCEPTION.NAME"), 
286                                    name));
287                    sb.append(row2(getString("EXCEPTION.EXCEPTION.MESSAGE"), 
288                                    message));
289                    sb.append(row2(getString("EXCEPTION.EXCEPTION.SHORT_STACK_TRACE"), 
290                                    shortStackTrace));
291                    sb.append(row2(getString("EXCEPTION.EXCEPTION.STACK_TRACE"), 
292                                    stackTrace));
293                    return sb;
294            }

296            /**
297             * @return a StringBuffer.
298             */
299            private StringBuffer getHtmlReportRequest() {
300                    String requestAttributesStr = html(getRequestAttributes());
301                    String sessionAttributesStr = html(getSessionAttributes());
302                    String globalSessionAttributesStr = html(getGlobalSessionAttributes());
303                    String requestHeadersStr = html(getRequestHeaders());
304                    String requestParametersStr = html(getRequestParameters());
305                    String cookiesStr = html(getCookies());
306                    StringBuffer sb = new StringBuffer();
307                    sb.append(header(getString("EXCEPTION.HEADER.REQUEST_ATTRIBUTES")));
308                    if (requestAttributesStr == null) {
309                            requestAttributesStr = 
310                                    em(getString("EXCEPTION.REQUEST_ATTRIBUTES.NONE"));
311                    }
312                    sb.append(row(requestAttributesStr));
313                    sb.append(header(getString("EXCEPTION.HEADER.SESSION_ATTRIBUTES")));
314                    if (sessionAttributesStr == null) {
315                            sessionAttributesStr = 
316                                    em(getString("EXCEPTION.SESSION_ATTRIBUTES.NONE"));
317                    }
318                    sb.append(row(sessionAttributesStr));
319                    sb.append(header(getString("EXCEPTION.HEADER.GLOBAL_SESSION_ATTRIBUTES")));
320                    if (globalSessionAttributesStr == null) {
321                            globalSessionAttributesStr = 
322                                    em(getString("EXCEPTION.GLOBAL_SESSION_ATTRIBUTES.NONE"));
323                    }
324                    sb.append(row(globalSessionAttributesStr));
325                    sb.append(header(getString("EXCEPTION.HEADER.REQUEST_HEADERS")));
326                    if (requestHeadersStr == null) {
327                            requestHeadersStr = em(getString("EXCEPTION.REQUEST_HEADERS.NONE"));
328                    }
329                    sb.append(row(requestHeadersStr));
330                    sb.append(header(getString("EXCEPTION.HEADER.REQUEST_PARAMETERS")));
331                    if (requestParametersStr == null) {
332                            requestParametersStr = 
333                                    em(getString("EXCEPTION.REQUEST_PARAMETERS.NONE"));
334                    }
335                    sb.append(row(requestParametersStr));
336                    sb.append(header(getString("EXCEPTION.HEADER.COOKIES")));
337                    if (cookiesStr == null) {
338                            cookiesStr = em(getString("EXCEPTION.COOKIES.NONE"));
339                    }
340                    sb.append(row(cookiesStr));
341                    return sb;
342            }

386            /**
387             * @return the subject of the email.
388             */
389            public String getReportSubject() {
390                    String serverStr = getServer();
391                    if (serverStr == null) {
392                            serverStr = getString("EXCEPTION.INFORMATION.SERVER.UNKNOWN");
393                    }
394                    String className = null;
395                    if (getThrowable() != null) {
396                            className = ExceptionUtils.getRealCause(getThrowable()).getClass().getSimpleName();
397                    }
398                    return getString(
399                                    "EXCEPTION.EMAIL_SUBJECT", 
400                                    getApplicationName(), 
401                                    getApplicationVersion(), 
402                                    className, 
403                                    serverStr);
404            }

406            /**
407             * Send an exception report. This method has been added to allow 
408             * it to be overriden (see org.esupportail.commons.services.exceptionHandling.CachingEmailExceptionServiceImpl).
409             * @param intercept
410             * @param t
411             * @param to the recipient
412             * @param emailSubject the subject of the mail
413             * @param htmlReport the HTML body
414             * @param textReport the plain text body
415             */
416            @SuppressWarnings("unchecked")
417            protected void sendEmail(
418                            final boolean intercept,
419                            final Throwable t,
420                            final InternetAddress to,
421                            final String emailSubject,
422                            final String htmlReport,
423                            final String textReport) {
424                    for (Throwable cause : ExceptionUtils.getCauses(t)) {
425                            Class causeClass = cause.getClass(); 
426                            for (Class clazz : noEmailExceptions) {
427                                    if (clazz.isAssignableFrom(causeClass)) {
428                                            if (logger.isDebugEnabled()) {
429                                                    logger.debug("no email for exception class [" + causeClass + "]");
430                                            }
431                                            return;
432                                    }
433                            }
434                    }
435                    if (logger.isDebugEnabled()) {
436                            logger.debug("sending the exception report to '" + to + "'...");
437                    }
438                    if (intercept) {
439                            smtpService.send(to, emailSubject, htmlReport, textReport);
440                    } else {
441                            smtpService.sendDoNotIntercept(to, emailSubject, htmlReport, textReport);
442                    }
443                    logger.debug("done.");
444            }

446            /**
447             * @see org.esupportail.commons.services.exceptionHandling.SimpleExceptionServiceImpl#handleException()
448             */
449            @Override
450            public void handleException() throws ExceptionHandlingException {
451                    String textReport = null;
452                    // try to logger the exception
453                    try {
454                            textReport = getTextReport();
455                            logTextReport(getThrowable(), textReport);
456                    } catch (Exception e) {
457                            logger.error("The following error was thrown by the application:", getThrowable());
458                            logger.error("Another exception was thrown while logging the first one (giving up):", e);
459                            throw new ExceptionHandlingException(e);
460                    }
461                    // finally try to send emails
462                    try {
463                            String htmlReport = getHtmlReport();
464                            String emailSubject = getReportSubject();
465                            if (this.recipientEmail != null) {
466                                    InternetAddress to = null;
467                                    try {
468                                            to = new InternetAddress(this.recipientEmail);
469                                    } catch (AddressException e) {
470                                            // already tested when the bean was created
471                                    }
472                                    sendEmail(true, getThrowable(), to, emailSubject, htmlReport, textReport);
473                            }
474                            if (!this.doNotSendExceptionReportsToDevelopers && getDevelEmail() != null) {
475                                    InternetAddress to = null;
476                                    try {
477                                            to = new InternetAddress(getDevelEmail());
478                                    } catch (AddressException e) {
479                                            // already tested when the bean was created
480                                    }
481                                    sendEmail(false, getThrowable(), to, emailSubject, htmlReport, textReport);
482                            }
483                    } catch (Exception e) {
484                            logger.error("The following error was thrown by the application:", getThrowable());
485                            logger.error("another exception was thrown while sending an email alert (giving up):", e);
486                            throw new ExceptionHandlingException(e);
487                    }
488            }
