004    package org.esupportail.commons.services.cas;
005    
006    import java.io.Serializable;
007    
008    
009    /** 
010     * The interface of the CAS service used to retrieve PTs from the CAs server.
011     */
012    public interface CasService extends Serializable {
013            
014            /**
015             * @param targetService The service the PT should be sent to.
016             * @return a PT.
017             * @throws CasException 
018             */
019            String getProxyTicket(String targetService) throws CasException;
020            
021            /**
022             * validate the ticket (ST or PT) to get a PGT. This method is 
023             * automatically called when retrieving the first PT for a remote
024             * service, but it can be called manually to prevent from the
025             * peremption of the ticket passed to the application.  
026             * @throws CasException 
027             */
028            void validate() throws CasException;
029            
030    }
