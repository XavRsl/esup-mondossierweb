004    package org.esupportail.commons.services.ldap;
005    
006    import java.util.List;
007    
008    import org.esupportail.commons.exceptions.ObjectNotFoundException;
009    
010    /**
011     * The interface of LDAP services for any entity.
012     */
013    public interface LdapEntityService extends BasicLdapService {
014    
015            /**
016             * Search an entity in the LDAP directory from a unique identifier.
017             * @param id the identifier
018             * @return the LdapEntity that corresponds to the given id.
019             * @throws LdapException 
020             * @throws ObjectNotFoundException
021             */
022            LdapEntity getLdapEntity(String id) throws LdapException, ObjectNotFoundException;
023    
024            /**
025             * Tell if an entity matches a filter.
026             * @param id the entity's unique identifier in the LDAP directory
027             * @param filter the filter
028             * @return true if the entity matches the filter.
029             * @throws LdapException
030             */
031            boolean entityMatchesFilter(String id, String filter) throws LdapException;
032    
033            /**
034             * @param filterExpr
035             * @return The list of LdapEntity that corresponds to a filter.
036             * @throws LdapException 
037             */
038            List<LdapEntity> getLdapEntitiesFromFilter(String filterExpr) throws LdapException;
039            
040    }
