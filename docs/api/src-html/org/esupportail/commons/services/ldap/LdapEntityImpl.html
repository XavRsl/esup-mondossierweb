004    package org.esupportail.commons.services.ldap;
005    
006    import java.util.ArrayList;
007    import java.util.HashMap;
008    import java.util.List;
009    import java.util.Map;
010    
011    /**
012     * A basic implementation of LdapEntity.
013     */
014    public class LdapEntityImpl implements LdapEntity {
015    
016    
017            /**
018             * The id for serialization.
019             */
020            private static final long serialVersionUID = 5921570970584589329L;
021    
022            /**
023             * The identifier of the LDAP user.
024             */
025            private String id;
026    
027    
028            /**
029             * The attributes of the LDAP user.
030             */
031            private Map<String, List<String>> attributes = new HashMap<String, List<String>>();
032    
033            /**
034             * Constructor.
035             */
036            public LdapEntityImpl() {
037                    super();
038            }
039    
040            /**
041             * @see java.lang.Object#toString()
042             */
043            @Override
044            public String toString() {
045                    return getClass().getSimpleName() + "#" + hashCode() + "[" 
046                    + "id=[" + id + "], " 
047                    + "attributeNames=" + getAttributeNames() + ", "
048                    + "attributes=" + attributes 
049                    + "]";
050            }
051    
052            /**
053             * @see org.esupportail.commons.services.ldap.LdapEntity#getId()
054             */
055            public String getId() {
056                    return id;
057            }
058    
059            /**
060             * @see org.esupportail.commons.services.ldap.LdapEntity#setId(java.lang.String)
061             */
062            public void setId(final String id) {
063                    this.id = id;
064            }
065    
066            /**
067             * @see org.esupportail.commons.services.ldap.LdapEntity#getAttributes()
068             */
069            public Map<String, List<String>> getAttributes() {
070                    return attributes;
071            }
072    
073            /**
074             * @see org.esupportail.commons.services.ldap.LdapEntity#setAttributes(java.util.Map)
075             */
076            public void setAttributes(
077                            final Map<String, List<String>> attributes) {
078                    this.attributes = attributes;
079            }
080    
081            /**
082             * @see org.esupportail.commons.services.ldap.LdapEntity#getAttributeNames()
083             */
084            public List<String> getAttributeNames() {
085                    return new ArrayList<String>(attributes.keySet()); 
086            }
087    
088            /**
089             * @see org.esupportail.commons.services.ldap.LdapEntity#getAttributes(java.lang.String)
090             */
091            public List<String> getAttributes(final String name) {
092                    List<String> result = attributes.get(name);
093                    if (result == null) {
094                            result = new ArrayList<String>();
095                    }
096                    return result;
097            }
098    
099            /**
100             * @see org.esupportail.commons.services.ldap.LdapEntity#getAttribute(java.lang.String)
101             */
102            public String getAttribute(final String name) {
103                    List<String> values = getAttributes(name);
104                    if (values.size() < 1) {
105                            return null;
106                    }
107                    return values.get(0);
108            }
109    
110    }
