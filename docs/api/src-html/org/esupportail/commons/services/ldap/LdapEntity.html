004    package org.esupportail.commons.services.ldap;
005    
006    import java.io.Serializable;
007    import java.util.List;
008    import java.util.Map;
009    
010    /**
011     * A LDAP entity, as returned by LDAP searches.
012     */
013    public interface LdapEntity extends Serializable {
014    
015            /**
016             * @return the map of all the attributes.
017             */
018            Map<String, List<String>> getAttributes();
019    
020            /**
021             * @param attributes the attributes to set
022             */
023            void setAttributes(Map<String, List<String>> attributes);
024    
025    
026            /**
027             * @return the id
028             */
029            String getId();
030            
031            /**
032             * 
033             * @param id the id to set
034             */
035            void setId(String id);
036            
037            /**
038             * @return the set of all the attribute names.
039             */
040            List<String> getAttributeNames();
041            
042            /**
043             * @param name 
044             * @return the values for an attribute.
045             */
046            List<String> getAttributes(String name);
047            
048            /**
049             * @param name 
050             * @return the value for an attribute.
051             */
052            String getAttribute(String name);
053            
054    }
