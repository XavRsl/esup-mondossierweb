004    package org.esupportail.commons.services.ldap;
005    
006    import java.util.ArrayList;
007    import java.util.List;
008    
009    
010    /**
011     * A basic implementation of LdapGroup.
012     */
013    public class LdapGroupImpl extends LdapEntityImpl implements LdapGroup {
014    
015            /**
016             * The id for serialization.
017             */
018            private static final long serialVersionUID = -7065179465595505379L;
019    
020            /**
021             * Constructor.
022             */
023            public LdapGroupImpl() {
024                    super();
025            }
026            
027            /**
028             * @param ldapEntity
029             * @return a LdapGroup, created using a LdapEntity.
030             */
031            public static LdapGroup createLdapGroup(final LdapEntity ldapEntity) {
032                    LdapGroupImpl ldapGroup = new LdapGroupImpl();
033                    ldapGroup.setId(ldapEntity.getId());
034                    ldapGroup.setAttributes(ldapEntity.getAttributes());
035                    return ldapGroup;
036            }
037    
038            /**
039             * @param ldapEntities
040             * @return a list of LdapGroup, created using ldapEntities.
041             */
042            public static List<LdapGroup> createLdapGroups(final List<LdapEntity> ldapEntities) {
043                    List<LdapGroup> ldapGroups = new ArrayList<LdapGroup>();
044                    for (LdapEntity ldapEntity : ldapEntities) {
045                            ldapGroups.add(LdapGroupImpl.createLdapGroup(ldapEntity));
046                    }
047                    return ldapGroups;
048            }
049    
050    }
