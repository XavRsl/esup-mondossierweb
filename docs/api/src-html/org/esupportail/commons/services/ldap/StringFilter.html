004    package org.esupportail.commons.services.ldap;
005    
006    import java.io.Serializable;
007    
008    import org.springframework.ldap.support.filter.AbstractFilter;
009    
010    /**
011     * A LdapTemplate filter, build with a string (that represents an already-encoded filter).
012     */
013    public class StringFilter extends AbstractFilter implements Serializable {
014            
015            /**
016             * The serialization id.
017             */
018            private static final long serialVersionUID = 7312355712619088000L;
019    
020            /**
021             * The filter string.
022             */
023            private String filterExpr;
024    
025            /**
026             * Bean constructor.
027             * @param filterExpr an already-encoded LDAP filter
028             */
029            public StringFilter(final String filterExpr) {
030                    super();
031                    this.filterExpr = "(" + filterExpr + ")";
032            }
033    
034            /**
035             * @see org.springframework.ldap.support.filter.AbstractFilter#encode(java.lang.StringBuffer)
036             */
037            @Override
038            public StringBuffer encode(final StringBuffer buff) {
039                    buff.append(filterExpr);
040                    return buff;
041            }
042    
043    }
