004    package org.esupportail.commons.services.ldap;
005    
006    import java.io.Serializable;
007    
008    /**
009     * The interface of writeable LDAP user services.
010     */
011    public interface WriteableLdapUserService extends Serializable {
012    
013            /** 
014             * Update a LDAP user in the LDAP directory.
015             * @param ldapUser
016             */
017            void updateLdapUser(LdapUser ldapUser);
018            
019            /** 
020             * Create a LDAP user.
021             * @param ldapUser
022             */
023            void createLdapUser(LdapUser ldapUser);
024            
025            /** 
026             * Delete a LDAP user.
027             * @param ldapUser
028             */
029            void deleteLdapUser(LdapUser ldapUser);
030            
031            /** 
032             * Define credentials which will be used for LDAP binding.
033             * Usefull when you ask credentials via a forms in your application
034             * and you want to bind to LDAP with those credentials.
035             * @param userId
036             * @param password
037             */
038            void setAuthenticatedContext(String userId, String password);
039            
040            /** 
041             * Disable WriteableLdapUserService write capabilities
042             * so that it can't write in LDAP directory in the sequel.
043             */
044            void defineAnonymousContext();
045    
046    }
