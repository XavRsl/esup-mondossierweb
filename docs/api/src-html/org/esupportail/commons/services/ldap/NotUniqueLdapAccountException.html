007    /**
008     * A class to represent the exceptions thrown when expected objects 
009     * do not exists (either retrieved from the database or configuration files).
010     */
011    public class NotUniqueLdapAccountException extends LdapException {
012    
013            /**
014             * The serialization id.
015             */
016            private static final long serialVersionUID = 5844758511471738202L;
017    
018            /**
019             * @param message
020             */
021            public NotUniqueLdapAccountException(final String message) {
022                    super(message);
023            }
024    
025            /**
026             * @param cause
027             */
028            public NotUniqueLdapAccountException(final Exception cause) {
029                    super(cause);
030            }
031    
032            /**
033             * @param message
034             * @param cause
035             */
036            public NotUniqueLdapAccountException(final String message, final Exception cause) {
037                    super(message, cause);
038            }
039    
040    }
