004    package org.esupportail.commons.services.ldap;
005    
006    import java.io.Serializable;
007    import java.util.List;
008    import java.util.Locale;
009    
010    /**
011     * The interface of LDAP services for any entity.
012     */
013    interface BasicLdapService extends Serializable {
014    
015            /**
016             * @param filterExpr
017             * @return null if the filter is correct, an error message otherwise.
018             * @throws LdapException 
019             */
020            String testLdapFilter(String filterExpr) throws LdapException;
021            
022            /**
023             * @return true if the class supports the retrieval of statictics. If false,
024             * calls to methods getStatistics() and resetStatistics() should throw an 
025             * exception.
026             */
027            boolean supportStatistics();
028            
029            /**
030             * Reset the statistics.
031             */
032            void resetStatistics();
033            
034            /**
035             * @return the statistics, as a list of Strings.
036             * @param locale
037             */
038            List<String> getStatistics(Locale locale);
039            
040            /**
041             * @return true if the class supports testing. If false, calls to method 
042             * test() should throw an exception.
043             */
044            boolean supportsTest();
045            
046            /**
047             * Test the LDAP connection.
048             */
049            void test();
050            
051    }
