016    package jp.sf.pal.facesresponse.renderkit;
017    
018    import org.apache.commons.logging.Log;
019    import org.apache.commons.logging.LogFactory;
020    
021    import java.util.Iterator;
022    
023    import javax.faces.context.FacesContext;
024    import javax.faces.render.RenderKit;
025    import javax.faces.render.RenderKitFactory;
026    
027    public class BufferedResponseRenderKitFactoryImpl extends RenderKitFactory {
028        /**
029         * Logger for this class
030         */
031        private static final Log log = LogFactory
032                .getLog(BufferedResponseRenderKitFactoryImpl.class);
033    
034        private RenderKitFactory parent;
035    
036        public BufferedResponseRenderKitFactoryImpl(RenderKitFactory parent) {
037            if (log.isDebugEnabled()) {
038                log
039                        .debug("ExtensionsRenderKitFactoryImpl(RenderKitFactory) parent="
040                                + parent);
041            }
042            this.parent = parent;
043        }
044    
045        public void addRenderKit(String renderKitId, RenderKit renderKit) {
046            parent.addRenderKit(renderKitId, renderKit);
047        }
048    
049        public RenderKit getRenderKit(FacesContext context, String renderKitId) {
050            return new BufferedResponseRenderKit(parent.getRenderKit(context,
051                    renderKitId));
052        }
053    
054        public Iterator getRenderKitIds() {
055            return parent.getRenderKitIds();
056        }
057    
058    }
