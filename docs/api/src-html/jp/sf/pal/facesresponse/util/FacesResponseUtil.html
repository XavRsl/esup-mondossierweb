016    package jp.sf.pal.facesresponse.util;
017    
018    import javax.faces.context.FacesContext;
019    import javax.portlet.PortletRequest;
020    
021    /**
022     * @author shinsuke
023     * 
024     */
025    public class FacesResponseUtil {
026        public static final String MYFACES_CURRENT_FACES_CONTEXT = "org.apache.myfaces.portlet.MyFacesGenericPortlet.CURRENT_FACES_CONTEXT";
027    
028        public static boolean isMyFacesFacesContext(PortletRequest request) {
029            if (request.getPortletSession().getAttribute(
030                    MYFACES_CURRENT_FACES_CONTEXT) != null) {
031                return true;
032            }
033            return false;
034        }
035    
036        public static boolean isMyFacesFacesContext() {
037            FacesContext facesContext = FacesContext.getCurrentInstance();
038            if (facesContext != null
039                    && facesContext.getExternalContext().getSessionMap().get(
040                            MYFACES_CURRENT_FACES_CONTEXT) != null) {
041                return true;
042            }
043            return false;
044        }
045    
046        public static FacesContext getMyFacesFacesContext(PortletRequest request) {
047            return (FacesContext) request.getPortletSession().getAttribute(
048                    MYFACES_CURRENT_FACES_CONTEXT);
049        }
050    }
